\section{Graded Call-By-Push-Value}

A new calculus to fix the issues we identified with the graded modal type theory.
The idea is that we have a new value judgement $\Gamma \tee v :^q A$,
where, informally, $q\cdot\Gamma \tee v :^q A$ if and only if
$\Gamma \tee v : A$ in the graded modal type theory.

\paragraph{Syntax}

\begin{align*}
  A &::= U\, \underline{B} \mid \boxempty^q A \mid A \times A' \mid A + A' \mid \unit \\
  \underline{B} &::= F^q\, A \mid \,^q A \rightarrow \underline{B} \\
  v &::= x \mid \thunk\, e \mid \bbox_q\, v \mid (v_1, v_2) \mid () \mid \inl\, v \mid \inr\, v \\
  e &::= \return\, v \mid \lambda x.\, e \mid e\, v
      \mid \llet x \leftarrow e_1 \lin e_2 \\
    &\mid \llet_{q_1} \bbox_{q_2} x = v \lin e \\
    &\mid \llet_q (x, y) = v \lin e
      \mid \case_q\, v\, \{x.e_1;\; y.e_2\} \\
    &\mid \absurd \, e
\end{align*}

\begin{mathpar}

  } \and

\section{Laws}

The laws that must hold for the algebra are:

\begin{law}
\label{law:add-assoc}
  $q_1 + (q_2 + q_3) = (q_1 + q_2) + q_3$.
\end{law}

\begin{law}
\label{law:add-comm}
  $q_1 + q_2 = q_2 + q_1$.
\end{law}

\begin{law}
\label{law:mul-assoc-1}
  $q_1 \cdot (q_2 \cdot q_3) \le (q_1 \cdot q_2) \cdot q_3$.
\end{law}

\begin{law}
\label{law:mul-distr-add-left-1}
  $q_1 \scale q_2 + q_1 \scale q_3 \le q_1 \scale (q_2 + q_3)$.
\end{law}

\begin{law}
\label{law:mul-distr-add-right-1}
  $q_1 \cdot q_3 + q_2 \cdot q_3 \le (q_1 + q_2) \cdot q_3$.
\end{law}

\begin{law}
\label{law:add-mono}
  If $q_1 \le q_3$ and $q_2 \le q_4$, then $q_1 + q_2 \le q_3 + q_4$.
\end{law}

\begin{law}
\label{law:mul-mono-left}
  If $q_2 \le q_3$, then $q_1 \scale q_2 \le q_1 \scale q_3$.
\end{law}

The laws that must hold for forward substitution are:

\begin{law}
\label{law:mul-assoc-2}
  $(q_1 \cdot q_2) \cdot q_3 \le q_1 \cdot (q_2 \cdot q_3)$.
 
  FOR BOX INTRO
\end{law}

\begin{law}
\label{law:mul-distr-add-left-2}
  $q_1 \scale (q_2 + q_3) \le q_1 \scale q_2 + q_1 \scale q_3$.
\end{law}

\begin{law}
\label{law:mul-distr-add-right-2}
  $(q_1 + q_2) \boxop q_3 \le q_1 \boxop q_3 + q_2 \boxop q_3$.
  
  FOR BOX INTRO
\end{law}

\begin{law}
\label{law:mul-mono-right}
  If $q_1 \le q_2$, then $q_1 \boxop q_3 \le q_2 \boxop q_3$.
\end{law}

\section{Proofs}

\begin{lemma}
\label{lem:mul-left-inverse-div-1}
  $q^\prime \le (q \scale q^\prime) \setminus q$.
\end{lemma}

\begin{proof}
  Consider the set $M = \{q^{\prime\prime} : q \cdot q^{\prime\prime} \le q \cdot q^\prime\}$, the candidates for $q^{\prime\prime} = q \cdot q^\prime \setminus q$.

  Since $q \cdot q^\prime \le q \cdot q^\prime$, it follows that $q^\prime \in M$.
  
  Thus, since $q \cdot q^\prime \setminus q$ is given by $\bigvee M$, $q^\prime \le q \cdot q^\prime \setminus q$, as required.
  
\end{proof}

\begin{lemma}
\label{lem:mul-left-inverse-div-2}
  $q \scale (q^\prime \setminus q) \le q^\prime$.
\end{lemma}

\begin{proof}
  By the definition of division, $q^\prime \setminus q = \bigvee\{q^{\prime\prime} : q \cdot q^{\prime\prime} \le q^\prime\}$.

  Thus, $q \cdot (q^\prime \setminus q) \le q^\prime$, as required.
  
\end{proof}

\begin{lemma}
\label{lem:div-mono}
  If $q_1 \le q_2$, then $q_1 \setminus q \le q_2 \setminus q$.
\end{lemma}

\begin{proof}
  Consider the sets $M_1 = \{q^\prime : q \cdot q^\prime \le q_1\}$ and $M_2 = \{q^\prime : q \cdot q^\prime = q_2\}$.

  Since $q_1 \le q_2$, every element in $M_1$ is in $M_2$, so $M_1 \subseteq M_2$.
  
  Thus, since $q_1 \setminus q$ and $q_2 \setminus q$ are given by $\bigvee M_1$ and $\bigvee M_2$, $q_1 \setminus q \le q_2 \setminus q$, as required.
  
\end{proof}

\begin{lemma}
\label{lem:div-distr-add}
  $q_1 \setminus q_3 + q_2 \setminus q_3 \le (q_1 + q_2) \setminus q_3$.
\end{lemma}

\begin{proof}
  By \ref{lem:mul-left-inverse-div-2}, have \textbf{(1)} $q_3 \cdot (q_1 \setminus q_3) \le q_1$ and \textbf{(2)} $q_3 \cdot (q_2 \setminus q_3) \le q_2$.

  By law \ref{law:add-mono} on \textbf{(1)} and \textbf{(2)}, have \textbf{(3)} $q_3 \cdot (q_1 \setminus q_3) + q_3 \cdot (q_2 \setminus q_3) \le q_1 + q_2$.

  By law \ref{law:mul-distr-add-left-2} on \textbf{(3)}, have \textbf{(4)} $q_3 \cdot (q_1 \setminus q_3 + q_2 \setminus q_3) \le q_1 + q_2$.

  Now consider the candidate set for $(q_1 + q_2) \setminus q_3$, $M = \{q : q_3 \cdot q \le q_1 + q_2\}$.

  Since $\textbf{(3)} \in M$, $\textbf{(3)} \le (q_1 + q_2) \setminus q_3$, as required.
  
\end{proof}

\begin{lemma}
\label{lem:new-modal-mode-split-2}
  If $\gamma\Gamma \tee v :^{q_1 + q_2} A$, then there exists $\gamma_1 + \gamma_2 \le \gamma$ such that $\gamma_1\Gamma \tee v :^{q_1} A$ and $\gamma_2\Gamma \tee v :^{q_2} A$.
\end{lemma}

\begin{proof}
  Proceed by induction over the structure of the derivation of $(\gamma_1 + \gamma_2)\Gamma \tee v :^{q_1 + q_2} A$.
  \begin{description}
    \item[Case VAR.]
    By VAR, $\underline{0}\Gamma, x :^{q_1} A, \underline{0}\Gamma^\prime \tee x :^{q_1}$ and $\underline{0}\Gamma, x :^{q_2} A, \underline{0}\Gamma^\prime \tee x :^{q_2} A$ as required.
    \\
    \item[Case VSUB.]
    By assumption, have \textbf{(1)} $\gamma^\prime\Gamma \tee v :^{q_1 + q_2} B$.

    By VSUB inversion on \textbf{(1)}, have \textbf{(2)} $\gamma \le \gamma^\prime$ and \textbf{(3)} $\gamma\Gamma \tee v :^{q_1 + q_2} B$.

    By the inductive hypothesis on \textbf{(3)}, have $\gamma_1 + \gamma_2 \le \gamma$ such that \textbf{(4)} $\gamma_1\Gamma \tee v :^{q_1} B$ and \textbf{(5)} $\gamma_2\Gamma \tee v :^{q_2} B$.

    For the exists statement, let $\gamma_1 = \gamma_1$ and $\gamma_2 = \gamma_2$.

    $\gamma_1 + \gamma_2 \le \gamma \le \gamma^\prime$, as required.

    \textbf{(4)} and \textbf{(5)} are as required.
    \\
    \item[Case THUNK.]
    By assumption, have \textbf{(1)} $(q \scale \gamma)\Gamma \tee \thunk e :^{q_1 + q_2} U \, \underline{B}$.

    By THUNK inversion, have \textbf{(2)} $\gamma\Gamma \tee e : \underline{B}$.

    For the exists statement, let $\gamma_1 = q_1 \scale \gamma$ and $\gamma_2 = q_2 \scale \gamma$.

    By law \ref{law:mul-distr-add-left-1}, $q_1 \scale \gamma + q_2 \scale \gamma \le (q_1 + q_2) \scale \gamma = q \scale \gamma$, as required.

    By THUNK on \textbf{(2)}, have \textbf{(3)} $q_1 \scale \Gamma \tee \thunk e :^{q_1} U \, \underline{B}$ and \textbf{(4)} $q_2 \scale \Gamma \tee \thunk e :^{q_2} U \, \underline{B}$, as required.
    \\
    \item[Case UNIT.]
    By UNIT, $\underline{0}\Gamma \tee () :^{q_1} \unit$ and $\underline{0}\Gamma \tee () :^{q_2} \unit$, as required.
    \\
    \item[Case PAIR.]
    By assumption, have \textbf{(1)} $(\gamma_1 + \gamma_2)\Gamma \tee (v_1, v_2) :^{q_1 + q_2} A \times B$.

    By PAIR inversion on \textbf{(1)}, have \textbf{(2)} $\gamma_1\Gamma \tee v_1 :^{q_1 + q_2} A$ and \textbf{(3)} $\gamma_2\Gamma \tee v_2 :^{q_1 + q_2} B$.

    By the inductive hypothesis on each of \textbf{(2)} and \textbf{(3)}, have $\gamma_{1_1} + \gamma_{1_2} \le \gamma_1$ and $\gamma_{2_1} + \gamma_{2_2} \le \gamma_2$ such that \textbf{(4)} $\gamma_{1_1}\Gamma \tee v_1 :^{q_1} A$, \textbf{(5)} $\gamma_{1_2}\Gamma \tee v_1 :^{q_2} A$, \textbf{(6)} $\gamma_{2_1}\Gamma \tee v_2 :^{q_1} B$ and \textbf{(7)} $\gamma_{2_2}\Gamma \tee v_2 :^{q_2} B$.

    For the exists statement, let $\gamma_1 = \gamma_{1_1} + \gamma_{2_1}$ and $\gamma_2 = \gamma_{1_2} + \gamma_{2_2}$.

    By \ref{law:add-mono}, $\gamma_{1_1} + \gamma_{2_1} + \gamma_{1_2} + \gamma_{2_2} \le \gamma_1 + \gamma_2$, as required.

    By PAIR on \textbf{(4)} with \textbf{(6)}, and \textbf{(5)} with \textbf{(7)}, $(\gamma_{1_1} + \gamma_{2_1})\Gamma \tee (v_1, v_2) :^{q_1} A \times B$ and $(\gamma_{1_2} + \gamma_{2_2})\Gamma \tee (v_1, v_2) :^{q_2} A \times B$, as required.
    \\
    \item[Case INL.]
    By assumption, have \textbf{(1)} $\gamma\Gamma \tee \inl v :^{q_1 + q_2} A + B$.

    By INL inversion on \textbf{(1)}, have \textbf{(2)} $\gamma\Gamma \tee v :^{q_1 + q_2} A$.

    By the inductive hypothesis on \textbf{(2)}, have \textbf{(3)} $\gamma_1 + \gamma_2 \le \gamma$ such that \textbf{(4)} $\gamma_1\Gamma \tee v :^{q_2} A$ and \textbf{(5)} $\gamma_2\Gamma \tee v :^{q_2} A$.

    For the exists statement, let $\gamma_1 = \gamma_1$ and $\gamma_2 = \gamma_2$.

    \textbf{(3)} is as required.

    By INL on each of \textbf{(4)} and \textbf{(5)}, $\gamma_1\Gamma \tee \inl v :^{q_1} A + B$ and $\gamma_2\Gamma \tee \inl v :^{q_2} A + B$, as required.
    \\
    \item[Case INR.]
    By assumption, have \textbf{(1)} $\gamma\Gamma \tee \inr v :^{q_1 + q_2} A + B$.

    By INR inversion on \textbf{(1)}, have \textbf{(2)} $\gamma\Gamma \tee v :^{q_1 + q_2} B$.

    By the inductive hypothesis on \textbf{(2)}, have \textbf{(3)} $\gamma_1 + \gamma_2 \le \gamma$ such that \textbf{(4)} $\gamma_1\Gamma \tee v :^{q_2} B$ and \textbf{(5)} $\gamma_2\Gamma \tee v :^{q_2} B$.

    For the exists statement, let $\gamma_1 = \gamma_1$ and $\gamma_2 = \gamma_2$.

    \textbf{(3)} is as required.

    By INR on each of \textbf{(4)} and \textbf{(5)}, $\gamma_1\Gamma \tee \inr v :^{q_1} A + B$ and $\gamma_2\Gamma \tee \inl v :^{q_2} A + B$, as required.
    \\
    \item[Case $\boxempty$I.] 
    By assumption, have \textbf{(1)} $\gamma\Gamma \tee \bbox_q v :^{q_1 + q_2} \boxempty^q A$.

    By $\boxempty$I inversion on \textbf{(1)}, have \textbf{(2)} $\gamma\Gamma \tee v :^{(q_1 + q_2) \boxop q} A$.

    By weakening using law \ref{law:mul-distr-add-right-2} on \textbf{(2)}, have \textbf{(3)} $\gamma\Gamma \tee v :^{q_1 \boxop q + q_2 \boxop q} A$.

    By the inductive hypothesis on \textbf{(3)}, have \textbf{(4)} $\gamma_1 + \gamma_2 \le \gamma$ such that \textbf{(5)} $\gamma_1\Gamma \tee v :^{q_1 \boxop q} A$ and \textbf{(6)} $\gamma_2\Gamma \tee v :^{q_2 \boxop q} A$.

    For the exists statement, let $\gamma_1 = \gamma_1$ and $\gamma_2 = \gamma_2$.

    \textbf{(4)} is as required.

    By $\boxempty$I on each of \textbf{(4)} and \textbf{(5)}, $\gamma_1\Gamma \tee \bbox_q v :^{q_1} \boxempty^q A$ and $\gamma_2\Gamma \tee \bbox_q v :^{q_2} \boxempty^q A$, as required.
  \end{description}
\end{proof}

\begin{lemma}
\label{lem:new-modal-succedent-weakening}
  If $\Gamma \tee v :^q A$ and $q^\prime \le q$, then $\Gamma \tee v :^{q^\prime} A$.
\end{lemma}

\begin{proof}
  Proceed by induction over the structure of the derivation of $\Gamma \tee v :^q A$.
  \begin{description}
    \item[Case VAR.] 
    By assumption, have $\underline{0}\Gamma, x :^q A, \underline{0}\Gamma^\prime \tee x :^q A$ and \textbf{(1)} $q^\prime \le q$.

    By VAR, have \textbf{(2)} $\underline{0}\Gamma, x :^{q^\prime} A, \underline{0}\Gamma^\prime \tee x :^{q^\prime} A$.

    By weakening on \textbf{(1)} and \textbf{(2)}, $\underline{0}\Gamma, x :^q A, \underline{0}\Gamma^\prime \tee x :^{q^\prime} A$, as required.
    \\
    \item[Case VSUB.] 
    By assumption, have \textbf{(1)} $\gamma^\prime\Gamma \tee v :^q A$ and \textbf{(2)} $q^\prime \le q$.
    
    By VSUB inversion, have \textbf{(3)} $\gamma \le \gamma^\prime$ and \textbf{(4)} $\gamma\Gamma, x :^{q^\prime} A, \Gamma^\prime \tee v :^{q_1 \cdot q_2} B$.

\begin{lemma}[Forward Substitution for New Type Theory Values]
\label{lem:new-modal-forward-substitution-vals}
  If $\gamma_1\Gamma \tee v :^q A$ and $\gamma_2\Gamma, x :^q A, \gamma_2^\prime\Gamma^\prime \tee v^\prime :^{q^\prime} B$, then $(\gamma_1 + \gamma_2)\Gamma, \gamma_2^\prime\Gamma^\prime \tee v^\prime[v/x] :^{q^\prime} B$.
\end{lemma}

\begin{proof}
  Proceed by induction over the structure of the derivation of $\gamma_2\Gamma, x :^q A, \gamma_2^\prime\Gamma^\prime \tee v^\prime :^{q^\prime} B$.

  \begin{description}
    \item[Case VAR.]
    Since $\gamma_2 = \gamma_2^\prime = \underline{0}$, by assumption, $\Gamma_1 \tee v :^q A$, as required.
    \\
    \item[Case VSUB.]
    By assumption, have \textbf{(1)} $\gamma_1\Gamma \tee v :^{q_2} A$ and \textbf{(2)} $\gamma_2\Gamma, x :^{q_2} A, \gamma_2^\prime\Gamma^\prime \tee v^\prime :^q B$.

    By VSUB inversion on \textbf{(2)}, have \textbf{(3)} $(\gamma_2^\prime, q_1, \gamma_2^{\prime\prime}) \le (\gamma_2, q_2, \gamma_2^\prime)$ and \textbf{(4)} $\gamma_2^\prime\Gamma, x :^{q_1} A, \gamma_2^{\prime\prime}\Gamma^\prime \tee v^\prime :^q B$.

    By \ref{lem:new-modal-succedent-weakening} on \textbf{(1)} using \textbf{(3)}, have \textbf{(5)} $\gamma_1\Gamma \tee v :^{q_1} A$.

    By the inductive hypothesis on \textbf{(4)} and \textbf{(5)}, have \textbf{(6)} $(\gamma_1 + \gamma_2^\prime)\Gamma, \gamma_2^{\prime\prime}\Gamma^\prime \tee v^\prime[v/x] :^q B$.

    By VSUB on \textbf{(3)} and \textbf{(6)}, $(\gamma_1 + \gamma_2)\Gamma, \gamma_2^\prime\Gamma^\prime \tee v^\prime[v/x] :^q B$, as required.
    \\
    \item[Case THUNK.]
    By assumption, have \textbf{(1)} $\gamma_1\Gamma \tee v :^{q^\prime \scale q} A$ and \textbf{(2)} $(q^\prime \scale \gamma_2)\Gamma, x :^{q^\prime \scale q} A, \gamma_2^\prime\Gamma^\prime \tee \thunk e :^{q^\prime} U \, \underline{B}$.

    By THUNK inversion on \textbf{(2)}, have \textbf{(3)} $\gamma_2\Gamma, x :^q A, \gamma_2^\prime\Gamma^\prime \tee e : \underline{B}$.

    By \ref{lem:new-modal-div-both-sides} on \textbf{(1)}, have \textbf{(4)} $(\gamma_1 \setminus q^\prime)\Gamma \tee v :^q A$.

    By \ref{lem:new-modal-forward-substitution-exprs} on \textbf{(3)} and \textbf{(4)}, have \textbf{(5)} $(\gamma_1 \setminus q^\prime + \gamma_2)\Gamma, \gamma_2^\prime\Gamma^\prime \tee e[v/x] : \underline{B}$.

    \\
    \item[Case UNIT.]

    \\
    \item[Case PAIR.]

    \\
    \item[Case INR.]

\end{lemma}

\begin{proof}
  Proceed by induction over the structure of the derivation of $\gamma_2\Gamma, x :^q A, \gamma_2^\prime\Gamma^\prime \tee e : B$.
  
  \begin{description}
    \item[Case RETURN.]
    By assumption, have \textbf{(1)} $\gamma_1\Gamma \tee v :^q A$ and $\gamma_2\Gamma, x :^q A, \gamma_2^\prime\Gamma^\prime \tee \return v^\prime : F^{q^\prime} A$.

    By RETURN inversion on \textbf{(2)}, have \textbf{(3)} $\gamma_2\Gamma, x :^q A, \gamma_2^\prime\Gamma^\prime \tee v :^{q^\prime} A$.

    By \ref{lem:new-modal-forward-substitution-vals} on \textbf{(1)} and \textbf{(3)}, have \textbf{(4)} $(\gamma_1 + \gamma_2)\Gamma, \gamma_2^\prime\Gamma^\prime \tee v^\prime[v/x] :^{q^\prime} A$.

    Thus, by RETURN on \textbf{(4)}, $(\gamma_1 + \gamma_2)\Gamma, \gamma_2^\prime\Gamma^\prime \tee \return v^\prime[v/x] : F^{q^\prime} A$, as required.
    \\
    \item[Case FORCE.]
    By assumption, have \textbf{(1)} $\gamma_1\Gamma \tee v :^q A$ and \textbf{(2)} $\gamma_2\Gamma, x :^q A, \gamma_2^\prime\Gamma^\prime \tee \force v^\prime : \underline{B}$.

    By FORCE inversion on \textbf{(2)}, have \textbf{(3)} $\gamma_2\Gamma, x :^q A, \gamma_2^\prime\Gamma^\prime \tee v^\prime :^1 U \, \underline{B}$.

    By \ref{lem:new-modal-forward-substitution-vals} on \textbf{(1)} and \textbf{(3)}, have \textbf{(4)} $(\gamma_1 + \gamma_2)\Gamma, \gamma_2^\prime\Gamma^\prime \tee v^\prime[v/x] :^1 U \, \underline{B}$.

    By FORCE on \textbf{(4)}, $(\gamma_1 + \gamma_2)\Gamma, \gamma_2^\prime\Gamma^\prime \tee \force v^\prime[v/x] : \underline{B}$, as required.
    \\
    \item[Case ESUB.]
    By assumption, have \textbf{(1)} $\gamma_1\Gamma \tee v :^{q_2} A$ and \textbf{(2)} $\gamma_2\Gamma, x :^{q_2} A, \gamma_2^\prime\Gamma^\prime \tee e : B$.

    By ESUB inversion on \textbf{(2)}, have \textbf{(3)} $(\gamma_2^\prime, q_1, \gamma_2^{\prime\prime}) \le (\gamma_2, q_2, \gamma_2^\prime)$ and \textbf{(4)} $\gamma_2^\prime\Gamma, x :^{q_1} A, \gamma_2^{\prime\prime}\Gamma^\prime \tee e : B$.

    By \ref{lem:new-modal-succedent-weakening} on \textbf{(1)} using \textbf{(3)}, have \textbf{(5)} $\gamma_1\Gamma \tee v :^{q_1} A$.

    By the inductive hypothesis on \textbf{(4)} and \textbf{(5)}, have \textbf{(6)} $(\gamma_1 + \gamma_2^\prime)\Gamma, \gamma_2^{\prime\prime}\Gamma^\prime \tee e[v/x] : B$.

    By ESUB on \textbf{(3)} and \textbf{(6)}, $(\gamma_1 + \gamma_2)\Gamma, \gamma_2^\prime\Gamma^\prime \tee e[v/x] : B$, as required.
    \\
    \item[Case LAM.]
    By assumption, have \textbf{(1)} $\gamma_1\Gamma \tee v :^q A$ and \textbf{(2)} $\gamma_2\Gamma, x :^q A, \gamma_2^\prime\Gamma^\prime \tee \lambda y.e : \,^{q^\prime} B_1 \rightarrow \underline{B_2}$.

    By LAM inversion on \textbf{(2)}, have \textbf{(3)} $\gamma_2\Gamma, x :^q A, \gamma_2^\prime\Gamma^\prime, y :^{q^\prime} B_1 \tee e : \underline{B_2}$.

    By the inductive hypothesis on \textbf{(3)}, have \textbf{(4)} $(\gamma_1 + \gamma_2)\Gamma, \gamma_2^\prime\Gamma^\prime, y :^{q^\prime} B_1 \tee e[v/x] : \underline{B_2}$.

    By LAM on \textbf{(4)}, $(\gamma_1 + \gamma_2)\Gamma, \gamma_2^\prime\Gamma^\prime \tee \lambda y.e :^{q^\prime} B_1 \rightarrow \underline{B_2}$, as required.
    \\
    \item[Case APP.]
    By assumption, have \textbf{(1)} $\gamma_1\Gamma \tee v :^{q_1 + q_2} A$ and \textbf{(2)} $(\gamma_{2_1} + \gamma_{2_2})\Gamma, x :^{q_1 + q_2} A, (\gamma_{2_1}^\prime + \gamma_{2_2}^\prime)\Gamma^\prime \tee e \, v : \underline{B}$.

    By APP inversion on \textbf{(2)}, have \textbf{(3)} $\gamma_{2_1}\Gamma, x :^{q_1} A, \gamma_{2_1}^\prime\Gamma^\prime \tee e : \,^{q^\prime} B_1 \rightarrow \underline{B_2}$ and \textbf{(4)} $\gamma_{2_2}\Gamma, x :^{q_2} A, \gamma_{2_2}^\prime\Gamma^\prime \tee v :^{q^\prime} B_1$.

    By \ref{lem:new-modal-mode-split-2} on \textbf{(1)}, have some \textbf{(5)} $\gamma_{1_1} + \gamma_{1_2} \le \gamma_1$ such that \textbf{(6)} $\gamma_{1_1}\Gamma \tee v :^{q_1^\prime} A$ and \textbf{(7)} $\gamma_{1_2}\Gamma \tee v :^{q_2^\prime} A$.
    
    By the inductive hypothesis on \textbf{(3)} and \textbf{(6)}, have \textbf{(8)} $(\gamma_{1_1} + \gamma_{2_1})\Gamma, \gamma_{2_1}^\prime\Gamma^\prime \tee e[v/x] : \,^{q^\prime} B_1 \rightarrow \underline{B_2}$. 

    By \ref{lem:new-modal-forward-substitution-vals} on \textbf{(4)} and \textbf{(7)}, have \textbf{(9)} $(\gamma_{1_2} + \gamma_{2_2}), \gamma_{2_2}^\prime\Gamma^\prime \tee v^\prime[v/x] :^{q^\prime} B_1$.

    By APP on \textbf{(8)} and \textbf{(9)}, have \textbf{(10)} $(\gamma_{1_1} + \gamma_{1_2} + \gamma_2)\Gamma, \gamma_2^\prime\Gamma^\prime \tee e[v/x] \, v^\prime[v/x] : \underline{B_2}$.

    By weakening on \textbf{(10)} using \textbf{(5)}, $(\gamma_1 + \gamma_2)\Gamma, \gamma_2^\prime\Gamma^\prime \tee e[v/x] \, v^\prime[v/x] : \underline{B_2}$, as required.
    \\
    \item[Case BIND.]
    By assumption, have \textbf{(1)} $\gamma_1\Gamma \tee v :^{q_1 + q_2} A$ and \textbf{(2)} $(\gamma_{2_1} + \gamma_{2_2})\Gamma, x :^{q_1 + q_2}, (\gamma_{2_1}^\prime + \gamma_{2_2}^\prime)\Gamma^\prime \tee \llet y \leftarrow e_1 \lin e_2 : \underline{B_2}$.

    By BIND inversion on \textbf{(2)}, have \textbf{(3)} $\gamma_{2_1}\Gamma, x :^{q_1} A, \gamma_{2_1}^\prime\Gamma^\prime \tee e_1 : F^{q^\prime} B_1$ and \textbf{(4)} $\gamma_{2_2}\Gamma, x :^{q_2} A, \gamma_{2_2}^\prime\Gamma^\prime, y :^{q^\prime} B_1 \tee e_2 : \underline{B_2}$.
    
    By \ref{lem:new-modal-mode-split-2} on \textbf{(1)}, have some \textbf{(5)} $\gamma_{1_1} + \gamma_{1_2} \le \gamma_1$ such that \textbf{(6)} $\gamma_{1_1}\Gamma \tee v :^{q_1^\prime} A$ and \textbf{(7)} $\gamma_{1_2}\Gamma \tee v :^{q_2^\prime} A$.

    By the inductive hypothesis on \textbf{(3)} with \textbf{(6)} and \textbf{(4)} with \textbf{(7)}, have \textbf{(8)} $(\gamma_{1_1} + \gamma_{2_1})\Gamma, \gamma_{2_1}^\prime\Gamma^\prime \tee e_1[v/x] : F^{q^\prime} B_1$ and \textbf{(9)} $(\gamma_{1_2} + \gamma_{2_2})\Gamma, \gamma_{2_2}^\prime\Gamma^\prime, y :^{q^\prime} B_1 \tee e_2[v/x] : \underline{B_2}$.

    By BIND on \textbf{(8)} and \textbf{(9)}, have \textbf{(10)} $(\gamma_{1_1} + \gamma_{1_2} + \gamma_2)\Gamma, \gamma_2^\prime\Gamma^\prime \tee \llet y \leftarrow e_1 \lin e_2 : \underline{B_2}$.

    By weakening on \textbf{(10)} using \textbf{(5)}, $(\gamma_1 + \gamma_2)\Gamma, \gamma_2^\prime\Gamma^\prime \tee \llet y \leftarrow e_1 \lin e_2 : \underline{B_2}$, as required.
    \\
    \item[Case $\boxempty$E.]
    By assumption, have \textbf{(1)} $\gamma_1\Gamma \tee v :^{q_1 + q_2} A$ and \textbf{(2)} $(\gamma_{2_1} + \gamma_{2_2})\Gamma^\prime, x :^{q_1^\prime + q_2^\prime} A, (\gamma_{2_1}^\prime + \Gamma_{2_2}^\prime)\Gamma^\prime \tee \llet_{q_1} \bbox_{q_2} y = \bbox_{q_2} v^\prime \lin e : \underline{B_2}$.

    By $\boxempty$E inversion on \textbf{(2)}, have \textbf{(3)} $\gamma_{2_1}\Gamma, x :^{q_1^\prime} A, \gamma_{2_1}^\prime\Gamma^\prime \tee \bbox_{q_2} v^\prime :^{q_1} \boxempty^{q_2} B_1$ and \textbf{(4)} $\gamma_{2_2}\Gamma, x :^{q_2^\prime} A, \gamma_{2_2}^\prime\Gamma^\prime, y :^{q_1 \boxop q_2} B_1 \tee e : \underline{B_2}$.

    By \ref{lem:new-modal-mode-split-2} on \textbf{(1)}, have some \textbf{(5)} $\gamma_{1_1} + \gamma_{1_2} \le \gamma_1$ such that \textbf{(6)} $\gamma_{1_1}\Gamma \tee v :^{q_1^\prime} A$ and \textbf{(7)} $\gamma_{1_2}\Gamma \tee v :^{q_2^\prime} A$.

    By \ref{lem:new-modal-forward-substitution-vals} on \textbf{(3)} and \textbf{(6)}, have \textbf{(8)} $(\gamma_{1_1} + \gamma_{2_1})\Gamma, \gamma_{2_1}^\prime\Gamma^\prime \tee \bbox_{q_2} v^\prime[v/x] :^{q_1} \boxempty^{q_2} B_1$. 

    By the inductive hypothesis on \textbf{(4)} and \textbf{(7)}, have \textbf{(9)} $(\gamma_{1_2} + \gamma_{2_2})\Gamma, \gamma_{2_2}^\prime\Gamma^\prime, y :^{q_1 \scale q_2} B_1 \tee e[v/x] : B_2$.

    By $\boxempty$E on \textbf{(8)} and \textbf{(9)}, have \textbf{(10)} $(\gamma_{1_1} + \gamma_{1_2} + \gamma_2)\Gamma, \gamma_2^\prime\Gamma^\prime \tee \llet_{q_1} \bbox_{q_2} y = \bbox_{q_2} v^\prime[v/x] \lin e[v/x] : B_2$.

    By weakening on \textbf{(10)} using \textbf{(5)}, $(\gamma_1 + \gamma_2)\Gamma, \gamma_2^\prime\Gamma^\prime \tee \llet_{q_1} \bbox_{q_2} y = \bbox_{q_2} v^\prime[v/x] \lin e[v/x] : B_2$, as required.
    \\
    \item[Case SPLIT.]
    By assumption, have \textbf{(1)} $\gamma_1\Gamma \tee v :^{q_1 + q_2} A$ and $(\gamma_{2_1} + \gamma_{2_2})\Gamma, x^{q_1 + q_2} A, (\gamma_{2_1}^\prime + \gamma_{2_2}^\prime)\Gamma^\prime \tee \llet_{q^\prime} (y, z) = v^\prime \lin e : \underline{C}$.

    By SPLIT inversion on \textbf{(2)}, have \textbf{(3)} $\gamma_{2_1}\Gamma, x :^{q_1}, \gamma_{2_1}^\prime\Gamma^\prime \tee v :^{q^\prime} B_1 \times B_2$ and \textbf{(4)} $\gamma_{2_2}\Gamma, x :^{q_2}, \gamma_{2_2}^\prime\Gamma^\prime, y :^{q^\prime} B_1, z :^{q^\prime} B_2 \tee e : \underline{C}$.

    By \ref{lem:new-modal-mode-split-2} on \textbf{(1)}, have some \textbf{(5)} $\gamma_{1_1} + \gamma_{1_2} \le \gamma_1$ such that \textbf{(6)} $\gamma_{1_1}\Gamma \tee v :^{q_1^\prime} A$ and \textbf{(7)} $\gamma_{1_2}\Gamma \tee v :^{q_2^\prime} A$.
    
    By \ref{lem:new-modal-forward-substitution-vals} on \textbf{(3)} and \textbf{(6)}, have \textbf{(8)} $(\gamma_{1_1} + \gamma_{2_1})\Gamma, \gamma_{2_1}^\prime\Gamma^\prime \tee v^\prime[v/x] :^{q^\prime} B_1 \times B_2$. 

    By the inductive hypothesis on \textbf{(4)} and \textbf{(7)}, have \textbf{(9)} $(\gamma_{1_2} + \gamma_{2_2})\Gamma, \gamma_{2_2}^\prime\Gamma^\prime, y :^{q^\prime} B_1, z :^{q^\prime} B_2 \tee e[v/x] : \underline{C}$.

    By SPLIT on \textbf{(8)} and \textbf{(9)}, have \textbf{(10)} $(\gamma_{1_1} + \gamma_{1_2} + \gamma_2)\Gamma, \gamma_2^\prime\Gamma^\prime \tee \llet_{q^\prime} (y, z) = v^\prime[v/x] \lin e[v/x] : \underline{C}$.

    By weakening on \textbf{(10)} using \textbf{(5)}, $(\gamma_1 + \gamma_2)\Gamma, \gamma_2^\prime\Gamma^\prime \tee \llet_{q^\prime} (y, z) = v^\prime[v/x] \lin e[v/x] : \underline{C}$, as required.
    \\
    \item[Case CASE.]
    By assumption, have \textbf{(1)} $\gamma_1 \tee v :^{q_1 + q_2} A$ and $(\gamma_{2_1} + \gamma_{2_2})\Gamma, x :^{q_1 + q_2} A, (\gamma_{2_1}^\prime + \gamma_{2_2}^\prime)\Gamma^\prime \tee \case_{q^\prime} \, v^\prime \{y.e_1; \, z.e_2\} : \underline{C}$.

    By CASE inversion on \textbf{(2)}, have \textbf{(3)} $\gamma_{2_1}\Gamma, x :^{q_1} A, \gamma_{2_1}^\prime\Gamma^\prime \tee v^\prime :^{q^\prime} B_1 + B_2$, \textbf{(4)} $\gamma_{2_2}\Gamma, x :^{q_2} A, \gamma_{2_2}^\prime\Gamma^\prime, y :^{q^\prime} B_1 \tee e_1 : \underline{C}$ and \textbf{(5)} $\gamma_{2_2}\Gamma, x :^{q_2} A, \gamma_{2_2}^\prime\Gamma^\prime, z :^{q^\prime} B_2 \tee e_2 : \underline{C}$.

    By \ref{lem:new-modal-mode-split-2} on \textbf{(1)}, have some \textbf{(6)} $\gamma_{1_1} + \gamma_{1_2} \le \gamma_1$ such that \textbf{(7)} $\gamma_{1_1}\Gamma \tee v :^{q_1^\prime} A$ and \textbf{(8)} $\gamma_{1_2}\Gamma \tee v :^{q_2^\prime} A$.
    
    By \ref{lem:new-modal-forward-substitution-vals} on \textbf{(3)} and \textbf{(7)}, have \textbf{(9)} $(\gamma_{1_1} + \gamma_{2_1})\Gamma, \gamma_{2_1}^\prime\Gamma^\prime \tee v^\prime[v/x] :^{q^\prime} B_1 + B_2$. 

    By the inductive hypothesis on each of \textbf{(4)} and \textbf{(5)} with \textbf{(8)}, have \textbf{(10)} $(\gamma_{1_2} + \gamma_{2_2})\Gamma, \gamma_{2_2}^\prime\Gamma, y :^{q^\prime} B_1 \tee e_1[v/x] : \underline{C}$ and \textbf{(11)} $(\gamma_{1_2} + \gamma_{2_2})\Gamma, \gamma_{2_2}^\prime\Gamma^\prime, z :^{q^\prime} B_2 \tee e_2[v/x] : \underline{C}$.

    By CASE on \textbf{(9)}, \textbf{(10)}, \textbf{(11)}, have \textbf{(12)} $(\gamma_{1_1} + \gamma_{1_2} + \gamma_2)\Gamma, \gamma_2^\prime\Gamma^\prime \tee \case_{q^\prime} \, v^\prime[v/x] \{y.e_1[v/x]; \, z.e_2[v/x]\} : \underline{C}$.
    
    By weakening on \textbf{(12)} using \textbf{(6)}, $(\gamma_1 + \gamma_2)\Gamma, \gamma_2^\prime\Gamma^\prime \tee \case_{q^\prime} \, v^\prime[v/x] \{y.e_1[v/x]; \, z.e_2[v/x]\}$, as required.
    \\
    \item[Case ABSURD.]
    $(\gamma_2, q, \gamma_2^\prime)$ has a different shape to $\top$, and so this case cannot occur.
  \end{description}
\end{proof}

Assume that $\Gamma \tee \llet_{q_1} \bbox_{q_2} x = \bbox_{q_2} v \lin e : \underline{B}$.
By the box-elim rule, we have $\Gamma = \Gamma_1 + \Gamma_2$ for some $\Gamma_1$ and $\Gamma_2$,
and $\Gamma_1 \tee \bbox_{q_2} v :^{q_1} \boxempty^{q_2} A$,
and $\Gamma_2, x :^{q_1 \cdot q_2} A \tee e : \underline{B}$.
By the box-intro rule, we have $\Gamma_1 \tee v :^{q_1 \cdot q_2} A$.
We use the forward substitution lemma, to obtain $\Gamma_1 + \Gamma_2 \tee e[v/x] : \underline{B}$.
