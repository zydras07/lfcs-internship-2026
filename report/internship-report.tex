\section{Calculi}

\subsection{Graded Modal Type Theory}

A fine-grained call-by-value calculus in the style of \cite{Abel:unified,Choudhury:graded}.

\paragraph{Syntax}

\begin{align*}
  A &::= \,^q A \rightarrow B \mid \boxempty^q A \mid A \times B \mid A + B \mid \unit \\
  v &::= x \mid \lambda x.\, e \mid \bbox_q\, v \mid (v_1, v_2) \mid () \mid \inl\, v \mid \inr\, v \\
  e &::= v \mid e\, v
      \mid \llet x \leftarrow e_1 \lin e_2 \\
    &\mid \llet_{q_1} \bbox_{q_2} x = v \lin e \\
    &\mid \llet_q (x, y) = v \lin e
      \mid \case_q\, v\, \{x.e_1;\; y.e_2\}
\end{align*}

\paragraph{Value typing rules} \hfill \framebox{$\Gamma \tee v : A$}

\begin{mathpar}
  \inferrule*[right=var]{
    \phantom{.}
  }{
    \Gamma, x :^1 A, \Gamma' \tee x : A
  } \and
  \inferrule*[right=sub]{
    q \leq q' \\
    \Gamma, x :^q A, \Gamma' \tee v : B
  }{
    \Gamma, x :^{q'} A, \Gamma' \tee v : B
  } \and
  \inferrule*[right=lam]{
    \Gamma, x :^q A \tee e : B
  }{
    \Gamma \tee \lambda x.\, e : \,^q A \rightarrow B
  } \and
  \inferrule*[right=unit]{
    \phantom{.}
  }{
    \Gamma \tee () : \unit
  } \and
  \inferrule*[right=pair]{
    \Gamma_1 \tee v_1 : A \\
    \Gamma_2 \tee v_2 : B
  }{
    \Gamma_1 + \Gamma_2 \tee (v_1, v_2) : A \times B
  } \and
  \inferrule*[right=inl]{
    \Gamma \tee v : A
  }{
    \Gamma \tee \inl\, v : A + B
  } \and
  \inferrule*[right=inr]{
    \Gamma \tee v : B
  }{
    \Gamma \tee \inr\, v : A + B
  } \and
  \inferrule*[right=$\boxempty$I]{
    \Gamma \tee v : A
  }{
    q\cdot\Gamma \tee \bbox_q\, v : \boxempty^q A
  }
\end{mathpar}

\paragraph{Expression typing rules} \hfill \framebox{$\Gamma \tee e : B$}

\begin{mathpar}
  \inferrule*[right=app]{
    \Gamma_1 \tee e : \,^q A \rightarrow B \\
    \Gamma_2 \tee v : A
  }{
    \Gamma_1 + q\cdot\Gamma_2 \tee e\, v : B
  } \and
  \inferrule*[right=bind]{
    \Gamma_1 \tee e_1 : A \\
    \Gamma_2, x :^1 A \tee e_2 : B
  }{
    \Gamma_1 + \Gamma_2 \tee \llet x \leftarrow e_1 \lin e_2 : B
  } \and
  \inferrule*[right=$\boxempty$E]{
    \Gamma_1 \tee v : \boxempty^{q_2} A \\
    \Gamma_2, x :^{q_1 \cdot q_2} A \tee e : B
  }{
    q_1\cdot\Gamma_1 + \Gamma_2 \tee \llet_{q_1} \bbox_{q_2} x = v \lin e : B
  } \and
  \inferrule*[right=split]{
    \Gamma_1 \tee v : A_1 \times A_2 \\
    \Gamma_2, x :^q A_1, y :^q A_2 \tee e : B
  }{
    q\cdot\Gamma_1 + \Gamma_2 \tee \llet_q (x, y) = v \lin e : B
  } \and
  \inferrule*[right=case]{
    \Gamma_1 \tee v : A + B \\
    \Gamma_2, x :^q A \tee e_1 : C \\
    \Gamma_2, y :^q B \tee e_2 : C
  }{
    q\cdot\Gamma_1 + \Gamma_2 \tee \case_q\, v\, \{x.e_1;\; y.e_2\} : C
  }
\end{mathpar}

  \case_q\, (\inr\, v)\, \{x.e_1;\; y.e_2\} &\red e_2[v/y]
\end{align*}

\paragraph{Parameterisation}

Graded modal type theory is usually parameterised by a pre-ordered semiring $(S, 0, 1, +, \cdot, \leq)$ of grades.
Aside from the usual semiring axioms, we require that $+$ and $\cdot$ are monotone with respect to $\leq$.

For local and global modes, we take the three-element semiring $\{0, 1, \mathbb{G}\}$ with
$0 \leq 1 \leq \mathbb{G}$, $+$ as maximum, and $\mathbb{G} \cdot \mathbb{G} = \mathbb{G}$.

We also have a division operator $\setminus$, which is defined by saying that $q_1 \setminus q_2$ is the maximum $q$ such that $q_2 \circ q = q_1$. $q_1 \setminus q_2$ is not defined if $q_1 < q_2$. $\setminus$ over contexts is defined point-wise over variables.

\subsection{Mode Qualifiers}

\paragraph{Syntax}

\begin{align*}
  A, B &::= A \at \mu_1 \rightarrow B \at \mu_2 \mid \boxempty^\nu A \mid A \times B \mid A + B \mid \unit \\
  e &::= x \mid \lambda x.\, e \mid e_1\, e_2
      \mid \bbox_\nu\, e \mid \unbox_\nu\, e
      \mid (e_1, e_2) \mid () \mid \inl\, e \mid \inr\, e \\
    &\mid \llet x = e_1 \lin e_2
      \mid \llet_\mu (x, y) = e \lin e
      \mid \case_\mu\, e\, \{x.e_1;\; y.e_2\} \\
  \Gamma &::= \cdot \mid \Gamma, x : A \at \mu \mid \Gamma, \lock_\mu
\end{align*}

We define $\mathrm{locks}(\Gamma)$ as the composition of all locks in $\Gamma$:
\begin{align*}
  \mathrm{locks}(\cdot) &= \mathrm{id} \\
  \mathrm{locks}(\Gamma, x :_\mu A) &= \mathrm{locks}(\Gamma) \\
  \mathrm{locks}(\Gamma, \lock_\mu) &= \mathrm{locks}(\Gamma) \circ \mu
\end{align*}

\paragraph{Typing rules} \hfill \framebox{$\Gamma \tee e : A \at \mu$}

\begin{mathpar}
  \inferrule*[right=var]{
    \mu_1 \geq \mathrm{locks}(\Gamma') \circ \mu_2
  }{
    \Gamma, x : A \at \mu_1, \Gamma' \tee x : A \at \mu_2
  } \and
  \inferrule*[right=lam]{
    \Gamma, \lock_\mu, x : A \at \mu_1 \tee e : B \at \mu_2
  }{
    \Gamma \tee \lambda x.\, e : (A \at \mu_1 \rightarrow B \at \mu_2) \at \mu
  } \and
  \inferrule*[right=app]{
    \Gamma_1 \tee e_1 : (A \at \mu_1 \rightarrow B \at \mu_2) \at \_ \\\\
    \Gamma_2 \tee e_2 : A \at \mu_1 \\
    \mu_2 \geq \mu_3
  }{
    \Gamma_1 + \Gamma_2 \tee e_1\, e_2 : B \at \mu_3
  } \and
  \inferrule*[right=$\boxempty$I]{
    \Gamma \tee e : A \at \mu \circ \nu
  }{
    \Gamma \tee \bbox_\nu\, e : \boxempty^\nu A \at \mu
  } \and
  \inferrule*[right=$\boxempty$E]{
    \Gamma \tee e : \boxempty^\nu A \at \mu
  }{
    \Gamma \tee \unbox_\nu\, e : A \at \mu \circ \nu
  } \and
  \inferrule*[right=let]{
    \Gamma_1 \tee e_1 : A \at \mu_1 \\
    \Gamma_2, x : A \at \mu_1 \tee e_2 : B \at \mu_2
  }{
    \Gamma_1 + \Gamma_2 \tee \llet x = e_1 \lin e_2 : B \at \mu_2
  } \and
  \inferrule*[right=unit]{
    \phantom{.}
  }{
    \Gamma \tee () : \unit \at \mu
  } \and
  \inferrule*[right=pair]{
    \Gamma_1 \tee e_1 : A \at \mu \\
    \Gamma_2 \tee e_2 : B \at \mu
  }{
    \Gamma_1 + \Gamma_2 \tee (e_1, e_2) : A \times B \at \mu
  } \and
  \inferrule*[right=split]{
    \Gamma_1 \tee e_1 : A \times B \at \mu_1 \\\\
    \Gamma_2, x : A \at \mu_1, y : B \at \mu_1 \tee e_2 : C \at \mu_2
  }{
    \Gamma_1 + \Gamma_2 \tee \llet_{\mu_1} (x, y) = e_1 \lin e_2 : C \at \mu_2
  } \and
  \inferrule*[right=inl]{
    \Gamma \tee e : A \at \mu
  }{
    \Gamma \tee \inl\, e : A + B \at \mu
  } \and
  \inferrule*[right=inr]{
    \Gamma \tee e : B \at \mu
  }{
    \Gamma \tee \inr\, e : A + B \at \mu
  } \and
  \inferrule*[right=case]{
    \Gamma_1 \tee e : A + B \at \mu_1 \\\\
    \Gamma_2, x : A \at \mu_1 \tee e_1 : C \at \mu_2 \\\\
    \Gamma_2, y : B \at \mu_1 \tee e_2 : C \at \mu_2
  }{
    \Gamma_1 + \Gamma_2 \tee \case_{\mu_1}\, e\, \{x.e_1;\; y.e_2\} : C \at \mu_2
  }
\end{mathpar}

\paragraph{Parameterisation}

Mode qualifier systems are usually parameterised by a set of axes $\mathcal{A}$.
Each axis $((A, \leq) \in \mathcal{A})$ is a total order and contains a distinguished element $1_A$.
The distinguished element $1_A$ is either the minimum element of $A$ (in which we call this axis \textit{monadic})
or the maximum element of $A$ (in which case we call this axis \textit{comonadic}).
Composition is given by maximum on monadic axes and by minimum on comonadic axes.
A mode $\mu$ is a product of axes which is partially ordered by the pointwise extension of the orders on the axes.

For the local and global modes, we take a single comonadic axis $\{1, \mathbb{G}\}$ and $1 \leq \mathbb{G}$. Note that this is dual to the subtyping relation, and is used to maintain consistency with the graded modal calculus.

Over contexts, the $\le$ relation is defined by comparing variables point-wise. If variables modes are equal when compared point-wise, then locks may be compared point-wise contrapositively, where $\mu \le \mu^\prime$ $\Gamma_1, \lock_{\mu^\prime}, \Gamma_2 \le \Gamma_1, \lock_\mu, \Gamma_2$. Two contexts are incomparable if they have a different set of variables or locks in different positions.

We also have $+$ and $\setminus$ operators for modes.

$+$ is defined with $\mu_1 + \mu_2 = \max(\mu_1, \mu_2)$. $+$ over contexts is defined point-wise over variables.

$\setminus$ is defined by saying that $\mu_1 \setminus \mu_2$ is the maximum $\mu$ such that $\mu_2 \circ \mu = \mu_1$. $\mu_1 \setminus \mu_2$ is not defined if $\mu_1 < \mu_2$. $\setminus$ over contexts is defined point-wise over variables.

For this parameterisation, it is also possible to define the lock as a left-division operator:
\begin{align*}
  \cdot,\lock_\mu &= \cdot \\
  \Gamma, x :_{\mu_1} A, \lock_{\mu_2} &= \Gamma, \lock_{\mu_2}, x :_{\mu_1} A & (\text{if } \mu_1 \leq \mu_2) \\
  \Gamma, x :_{\mu_1} A, \lock_{\mu_2} &= \Gamma, \lock_{\mu_2} & (\text{otherwise})
\end{align*}

\subsection{Converting Graded Modal Type Theory to Mode Qualifiers}

\paragraph{Interpretation of types}
%
\begin{align*}
  \sem{\unit} &= \unit \\
  \sem{A + B} &= \sem{A} + \sem{B} \\
  \sem{A \times B} &= \sem{A} \times \sem{B} \\
  \sem{\boxempty^{q} A} &= \boxempty^{q}\, \sem{A} \\
  \sem{\,^q A \rightarrow B} &= \sem{A} \at q \rightarrow \sem{B} \at 1
\end{align*}

\paragraph{Contexts}
%
\begin{align*}
  \sem{\cdot} &= \cdot \\
  \sem{\Gamma, x :^0 A} &= \sem{\Gamma} \\
  \sem{\Gamma, x :^q A} &= \sem{\Gamma},\, x : \sem{A} \at q \\
  \sem{\Gamma_1 + \Gamma_2} &= \sem{\Gamma_1} + \sem{\Gamma_2} \\
  \sem{q \cdot \Gamma} &= q \circ \sem{\Gamma}
\end{align*}

\paragraph{Translation of values}
%
\begin{align*}
  \sem{x} &= x \\
  \sem{()} &= () \\
  \sem{\lambda x.\, e} &= \lambda x.\, \sem{e} \\
  \sem{\bbox_q\, v} &= \bbox_q\; \sem{v} \\
  \sem{(v_1, v_2)} &= (\sem{v_1}, \sem{v_2}) \\
  \sem{\inl\, v} &= \inl\; \sem{v} \\
  \sem{\inr\, v} &= \inr\; \sem{v}
\end{align*}

\paragraph{Translation of expressions}
%
\begin{align*}
  \sem{v} &= \sem{v} \\
  \sem{e\, v} &= \sem{e}\; \sem{v} \\
  \sem{\llet x \leftarrow e_1 \lin e_2}
    &= \llet x = \sem{e_1} \lin \sem{e_2} \\
  \sem{\llet_{q_1} \bbox_{q_2} x = v \lin e}
    &= \llet x = \unbox_{q_2}\; \sem{v} \lin \sem{e} \\
  \sem{\llet_q (x, y) = v \lin e}
    &= \llet_q (x, y) = \sem{v} \lin \sem{e} \\
  \sem{\case_q\, v\, \{x.e_1;\; y.e_2\}}
    &= \case_q\; \sem{v}\; \{x.\sem{e_1};\; y.\sem{e_2}\}
\end{align*}

\begin{theorem}[Weakening for Graded Modal Type Theory]
  If \ $\Gamma \leq \Gamma^\prime$ and $\Gamma \tee e : A$, then $\Gamma^\prime \tee e : A$.
\end{theorem}

\begin{proof}
  By the assumption $\Gamma \leq \Gamma^\prime$, and the definition of $\le$ over contexts, have $q \le q^\prime$ for every $q$, $q^\prime$ and $x$ such that $\Gamma = \Gamma_1, x :^q A, \Gamma_2$ and $\Gamma^\prime = \Gamma_1^\prime, x :^{q^\prime} A, \Gamma_2^\prime$. Repeatedly apply SUB with each variable $x \in \Gamma$ and the above statement specialised with $x$, to obtain $\Gamma^\prime \tee e : A$, as required.
  
\end{proof}

\begin{theorem}[Weakening for Mode Qualifiers]
  If \ $\Gamma \leq \Gamma^\prime$, $q^\prime \leq q$ and $\Gamma \tee e : A \at q$, then $\Gamma^\prime \tee e : A \at q^\prime$.
\end{theorem}

\begin{proof}
  Proceed by induction over the structure of the derivation of $\Gamma \tee x : A$.\\
  \begin{description}
    \item[Case VAR.]
    By assumption, have $\mu_1 \ge \mathrm{locks}(\Gamma_2) \circ \mu_2 \ge \mathrm{locks}(\Gamma_2) \circ \mu_2^\prime$. 
    
    By this and the definition of $\le$ in contexts, we have $\mu_1^\prime \ge \mu_1 \ge \mathrm{locks}(\Gamma^\prime) \circ \mu_2^\prime$ for every $\mu_1^\prime$ and $\mu_1$ such that $\Gamma \tee x : A \at \mu_1$ and $\Gamma^\prime \tee x : A \at \mu^\prime_1$. 
    
    Thus, by VAR, have $\Gamma^\prime_1, x : A \at \mu_1^\prime, \Gamma^\prime_2 \tee x : A \at \mu_2^\prime$, as required.
    \\
    \item[Case LAM.]
    By the inductive hypothesis, have $\Gamma^\prime, \lock_{\mu^\prime}, x : A \at \mu_1 \tee e : B \at \mu_2$. 
    
    By LAM on this, $\Gamma^\prime \tee \lambda x.e : (A \at \mu_1 \rightarrow B \at \mu_2) \at \mu^\prime$, as required.
    \\
    \item[Case APP.]
    By the inductive hypothesis, have $\Gamma^\prime_1 \tee e_1 : (A \at \mu_1 \rightarrow B \at \mu_2) \at \_$ and $\Gamma^\prime_2 \tee e_2 : A \at \mu_1$. 
    
    By APP on these and assumption $\mu_2^\prime \le \mu_2$, $\Gamma^\prime_1 + \Gamma^\prime_2 \tee e_1 e_2 : B \at \mu_2^\prime$ as required.
    \\
    \item[Case $\boxempty$I.]
    By the inductive hypothesis, have $\Gamma^\prime \tee e : A \at \nu \circ \mu^\prime$.
    
    By $\boxempty$I on this, $\Gamma^\prime \tee \mathrm{box}_ve : \boxempty^vA \at \mu^\prime$, as required.
    \\
    \item[Case $\boxempty$E.]
    By the inductive hypothesis, have $\Gamma^\prime \tee e : \boxempty^vA \at \mu^\prime$.

\begin{lemma}
  If $\mu^\prime \setminus \mu$ is defined, then $\mu^\prime = \mu \circ \mu^\prime \setminus \mu$.
\end{lemma}

\begin{proof}
  Proceed by induction over $\mu$ and $\mu^\prime$.
  \begin{description}
    \item[Case $\mu = 1$.]
    $1 \circ \mu^\prime \setminus 1 = \mu^\prime \setminus 1$. Since $\mu = \mu^\prime$ is the only $\mu$ such that $1 \circ \mu = \mu^\prime$, it follows that $\mu^\prime = \mu^\prime \setminus 1$, as required.
    \item[Case $\mu^\prime = \mathbb{G}$, $\mu = \mathbb{G}$.] 
    $\mathbb{G} \circ \mathbb{G} \setminus \mathbb{G} = \mathbb{G} \setminus \mathbb{G}$. Both $\mu = 1$ and $\mu = \mathbb{G}$ satisfy $\mathbb{G} \circ \mu = \mathbb{G}$ so, since $1 \le \mathbb{G}$, $\mathbb{G} = \mathbb{G} \setminus \mu$, as required.
  \end{description}
\end{proof}

\begin{lemma}
  $q \setminus q \ge 1$.
\end{lemma}

\begin{proof}
  Proceed by cases over $q$.
  \begin{description}
      \item[Case $q = 1$.]
      $q = 1$ is the greatest (and only) q such that $1 \cdot q = 1$, so $1 \setminus 1 = 1 \ge 1$, as required.
      \item[Case $q = \mathbb{G}$.]
      $q = \mathbb{G}$ is the greatest q such that $\mathbb{G} \cdot q = \mathbb{G}$ (candidates for $q$ are 1 and $\mathbb{G}$), so $\mathbb{G} \setminus \mathbb{G} = \mathbb{G} \ge 1$, as required.
  \end{description}
\end{proof}

\begin{theorem}[Division to Locks for Mode Qualifiers]
  If $\Gamma \setminus \mu, \Gamma^\prime \tee e : A \at \mu^\prime$, then $\Gamma, \lock_\mu, \Gamma^\prime \tee e : A \at \mu^\prime$.
\end{theorem}

\begin{proof}
  Proceed by induction over the structure of $\Gamma \setminus \mu, \Gamma^\prime \tee e : A \at \mu^\prime$.
  \begin{description}
    \item[Case VAR.]
    By assumption, have $\mu_1 \setminus \mu \ge \mathrm{locks}(\Gamma^\prime) \circ \mu_2$. 
    
    By the monotonicity of $\circ$, have $\mu \circ \mu_1 \setminus \mu \ge \mu \circ \mathrm{locks}(\Gamma^\prime) \circ \mu_2$. 
    
    By (1.4), have $\mu_1 \ge \mu \circ \mathrm{locks}(\Gamma^\prime) \circ \mu_2$. 
    
    Thus, by VAR, $\Gamma, x : A \at \mu_1, \lock_\mu, \Gamma^\prime \tee x : A \at \mu_2$, as required.
    \\
    \item[Case LAM.]
    By assumption, have \textbf{(1)} $\Gamma \setminus \mu, \Gamma^\prime, \lock_\mu^\prime, x : A \at \mu_1 \tee e : B \at \mu_2$. 
    
    By inductive hypothesis on \textbf{(1)}, have \textbf{(2)} $\Gamma, \lock_\mu, \Gamma^\prime, \lock_{\mu^\prime}, x : A \at \mu_1 \tee e : B \at \mu_2$.
    
    Thus, by LAM on \textbf{(2)}, $\Gamma, \lock_\mu, \Gamma^\prime \tee \lambda x.e : (A \at \mu_1 \rightarrow B \at \mu_2) \at \mu^\prime$, as required.
    \\
    \item[Case APP.]
    By assumption, have \textbf{(1)} $\Gamma_1 \setminus \mu, \Gamma^\prime \tee e_1 : (A \at \mu_1 \rightarrow B \at \mu_2) \at \_$, \textbf{(2)} $\Gamma_2 \setminus \mu, \Gamma^\prime \tee e_2 : A \at \mu_1$ and \textbf{(3)} $\mu_2 \ge \mu_3$. 
    
    By inductive hypothesis on each of \textbf{(1)} and \textbf{(2)}, have \textbf{(4)} $\Gamma_1, \lock_\mu, \Gamma^\prime_1 \tee e_1 : (A \at \mu_1 \rightarrow B \at \mu_2) \at \_$ and \textbf{(5)} $\Gamma_2, \lock_\mu, \Gamma^\prime_2 \tee e_2 : A \at \mu_1$.
    
    Thus, by APP on \textbf{(3)}, \textbf{(4)} and \textbf{(5)}, $\Gamma, \lock_\mu, \Gamma^\prime \tee e_1 e_2 : B \at \mu_3$, as required.
    \\
    \item[Case $\boxempty$I.]
    By assumption, have \textbf{(1)} $\Gamma \setminus \mu, \Gamma^\prime \tee e : A \at \nu \circ \mu^\prime$. 
    
    By the inductive hypothesis on \textbf{(1)}, have \textbf{(2)} $\Gamma, \lock_\mu, \Gamma^\prime \tee e : A \at \nu \circ \mu^\prime$. 
    
    Thus, by $\boxempty$I on \textbf{(2)}, $\Gamma, \lock_\mu, \Gamma^\prime \tee \mathrm{box}_v e : \boxempty^\nu A \at \circ \mu^\prime$, as required.
    \\
    \item[Case $\boxempty$E.]
    By assumption, have \textbf{(1)} $\Gamma \setminus \mu, \Gamma^\prime \tee e : \boxempty^\nu A \at \mu^\prime$. 
    
    By the inductive hypothesis on \textbf{(1)}, have \textbf{(2)} $\Gamma, \lock_\mu, \Gamma^\prime \tee e : \boxempty^\nu A \at \mu^\prime$. 
    
    Thus, by $\boxempty$E on \textbf{(2)}, $\Gamma, \lock_\mu, \Gamma^\prime \tee \mathrm{unbox}_v e : A \at \nu \circ \mu^\prime$, as required.
    \\
    \item[Case LET.]
    By assumption, have \textbf{(1)} $\Gamma_1 \setminus \mu, \Gamma^\prime_1 \tee e_1 : A \at \mu_1$ and \textbf{(2)} $\Gamma_2 \setminus \mu, \Gamma^\prime_2, x : A \at \mu_1 \tee e_2 : B \at \mu_2$. 
    
    By the inductive hypothesis on each of \textbf{(1)} and \textbf{(2)}, have \textbf{(3)} $\Gamma_1, \lock_\mu, \Gamma^\prime_1 \tee e_1 : A \at \mu_1$ and \textbf{(4)} $\Gamma_2, \lock_\mu, \Gamma^\prime_2, x : A \at \mu_1 \tee e_2 : B \at \mu_2$. 
    
    Thus, by LET on \textbf{(3)} and \textbf{(4)}, $\Gamma, \lock_\mu, \Gamma^\prime \tee \llet x = e_1 \lin e_2 : B \at \mu_2$, as required.
    \\
    \item[Case UNIT.]
    $\Gamma, \lock_\mu, \Gamma^\prime \tee () : \unit \at \mu^\prime$, as required.
    \\
    \item[Case PAIR.]
    By assumption, have \textbf{(1)} $\Gamma_1 \setminus \mu, \Gamma^\prime_1 \tee e_1 : A \at \mu^\prime$ and \textbf{(2)} $\Gamma_2 \setminus \mu, \Gamma^\prime_2 \tee e_2 : B \at \mu^\prime$. 
    
    By the inductive hypothesis on each of \textbf{(1)} and \textbf{(2)}, have \textbf{(3)} $\Gamma_1, \lock_\mu, \Gamma^\prime_1 \tee e_1 : A \at \mu^\prime$ and \textbf{(4)} $\Gamma_2, \lock_\mu, \Gamma^\prime_2 \tee e_2 : B \at \mu^\prime$. 
    
    Thus, by PAIR on \textbf{(3)} and \textbf{(4)}, $\Gamma, \lock_\mu, \Gamma^\prime \tee (e_1, e_2) : A \times B \at \mu^\prime$, as required.
    \\

    \\

    By the inductive hypothesis on \textbf{(1)}, have \textbf{(2)} $\Gamma, \lock_\mu, \Gamma^\prime \tee e : A \at \mu^\prime$.

    By the inductive hypothesis on \textbf{(1)}, have \textbf{(2)} $\Gamma, \lock_\mu, \Gamma^\prime \tee e : B \at \mu^\prime$. 
    Thus, by INR on \textbf{(2)}, $\Gamma, \lock_\mu, \Gamma^\prime \tee \inr e : A + B \at \mu^\prime$, as required.
    \\
    \item[Case CASE.]

  \end{description}

\begin{theorem}[Graded Type Theory Values to Mode Qualifiers]
  If\/ $\Gamma \tee v : A$, then $q \circ \sem{\Gamma} \tee \sem{v} : \sem{A} \at q$ for all $q \neq 0$. 
\end{theorem}

\begin{proof}
  Proceed by induction over the structure of the derivation of $q \cdot \Gamma \tee v : A$.\\
  \begin{description}
    \item[Case VAR.]
    Have $q \le q$ by reflexivity of $\le$, and note that $\sem{\Gamma}$ and $\sem{\Gamma^\prime}$ must not have any locks. 
    
    Thus, by VAR, $\sem{\Gamma}, x : \sem{A} \at q, \sem{\Gamma^\prime} \tee x : \sem{A} \at q$, as required.
    \\
    \item[Case SUB.]
    By the inductive hypothesis, have $q^\prime \circ \sem{\Gamma}, x : \sem{A} \at q^\prime \circ q_1, q^\prime \circ \sem{\Gamma^\prime} \tee \sem{v} : \sem{B} \at q$. 
    
    Thus, by the assumption $q_1 \le q_2$ and weakening, $q^\prime \circ \sem{\Gamma}, x : \sem{A} \at q_2, q^\prime \circ \sem{\Gamma^\prime} \tee \sem{v} : \sem{B} \at q$, as required.
    \\
    \item[Case LAM.]
    By the inductive hypothesis, have $\sem{\Gamma}, x : \sem{A} \at q \tee \sem{e} : \sem{B} \at 1$. 
    
    By (1.5), have $q^\prime \circ \sem{\Gamma} \setminus q^\prime, x : \sem{A} \at q \tee \sem{e} : B \at 1$. 
    
    By (1.6), have $q^\prime \circ \sem{\Gamma}, \lock_q, x : \sem{A} \at q \tee \sem{e} : B \at 1$. 
    
    Thus, by LAM, $q^\prime \circ \sem{\Gamma} \tee \lambda x.\sem{e} : (\sem{A} \at q \rightarrow \sem{B} \at 1) \at q^\prime$ which, by the translation, is equivalent to $q^\prime \circ \sem{\Gamma} \tee \sem{\lambda x.e} : \sem{\,^qA \rightarrow B} \at q^\prime$, as required.
    \\
    \item[Case UNIT.]
    By UNIT, $q \circ \sem{\Gamma} \tee () : \unit \at q$ as required.
    \\ 
    \item[Case PAIR.]
    By assumption, have \textbf{(1)} $\Gamma_1 \tee v_1 : A$ and \textbf{(2)} $\Gamma_2 \tee v_2 : B$.
    
    By the inductive hypothesis on each of \textbf{(1)} and \textbf{(2)}, have \textbf{(3)} $q \circ \sem{\Gamma_1} \tee \sem{v_1} : \sem{A} \at q$ and \textbf{(4)} $q \circ \sem{\Gamma_2} \tee \sem{v_2} : \sem{B} \at q$.
    
    By PAIR on \textbf{(3)} and \textbf{(4)}, have \textbf{(5)} $q \circ \sem{\Gamma_1} + q \circ \sem{\Gamma_2} \tee (\sem{v_1}, \sem{v_2}) : \sem{A} \times \sem{B} \at q$
    
    Thus, by the left distributivity of $\circ$ over $+$, $q \circ (\sem{\Gamma_1} + \sem{\Gamma_2}) \tee (\sem{v_1}, \sem{v_2}) : \sem{A} \times \sem{B} \at q$, which, by translation, is equivalent to $q \circ (\sem{\Gamma_1} + \sem{\Gamma_2}) \tee \sem{(v_1, v_2)} : \sem{A \times B} \at q$, as required.
    \\ 
    \item[Case INL.]
    By the inductive hypothesis, have $q \circ \sem{\Gamma} \tee \sem{e} : \sem{A} \at q$. 
    
    By INL on this, $q \circ \sem{\Gamma} \tee \inl \sem{e} : \sem{A} + B^\prime \at q$. 
    
    Since, for all mode qualifier types $B^\prime$, there exists a graded modal type $B$ such that $B^\prime = \sem{B}$, we have $q \circ \sem{\Gamma} \tee \inl \sem{e} : \sem{A} + \sem{B} \at q$, and thus, by the translation, $q \circ \sem{\Gamma} \tee \sem{\inl e} : \sem{A + B} \at q$, as required.
    \\
    \item[Case INR.]
    By the inductive hypothesis, have $q \circ \sem{\Gamma} \tee \sem{e} : \sem{B} \at q$. 
    
    By INR on this, $q \circ \sem{\Gamma} \tee \inr \sem{e} : A^\prime + \sem{B} @ q$. 
    
    Since, for all mode qualifier types $A^\prime$, there exists a graded modal type $A$ such that $A^\prime = \sem{A}$, we have $q \circ \sem{\Gamma} \tee \inr \sem{e} : \sem{A} + \sem{B} \at q$, and thus, by the translation, $q \circ \sem{\Gamma} \tee \sem{\inr e} : \sem{A + B} \at q$, as required.
    \\
    \item[Case $\boxempty$I.]
    By the inductive hypothesis, have $q^\prime \circ q \circ \sem{\Gamma} \tee \sem{v} : \sem{A} \at q^\prime \circ q$. 
    
    Thus, by $\boxempty$I, $q^\prime \circ q \circ \sem{\Gamma} \tee \mathrm{box}_q \sem{v} : \boxempty^q \sem{A} \at q^\prime$, which, by the translation, is equivalent to $q^\prime \circ q \circ \sem{\Gamma} \tee \sem{\mathrm{box}_q v} : \sem{\boxempty_q A} \at q^\prime$, as required.
  \end{description}
\end{proof}

\begin{theorem}[Graded Type Theory Expressions to Mode Qualifiers]
If $\Gamma \tee e : A$, then $\sem{\Gamma} \tee \sem{e} : \sem{A} \at 1$.
\end{theorem}

\begin{proof} 
  Proceed by induction over the structure of the derivation of  $\Gamma \tee e : A$.\\
  \begin{description}
    \item[Case APP.]
    By assumption, have \textbf{(1)} $\Gamma_1 \tee e : \,^qA \rightarrow B$ and \textbf{(2)} $\Gamma_2 \tee v : A$. 
    
    By inductive hypothesis on \textbf{(1)} and (1.7) on \textbf{(2)}, have \textbf{(3)} $\sem{\Gamma_1} \tee \sem{e} : (\sem{A} \at q \rightarrow \sem{B} \at 1) \at 1$ and \textbf{(4)} $\sem{\Gamma_2} \tee \sem{v} : \sem{A} \at q$. 

    By APP on \textbf{(3)} and \textbf({4}), have $\sem{\Gamma_1} + \sem{\Gamma_2} \tee \sem{e} \sem{v} : \sem{B} \at 1$
    
    Thus, by weakening on each of \textbf{(3)} and \textbf{(4)}, have \textbf{(5)} $\sem{\Gamma_1} + q \circ \sem{\Gamma_2} \tee \sem{e} : (\sem{A} \at q \rightarrow \sem{B} \at 1) \at 1$ and \textbf{(6)} $\sem{\Gamma_1} + q \circ \sem{\Gamma_2} \tee \sem{v} : \sem{A} \at q$.
    
    Thus, by APP on \textbf{(5)} and \textbf{(6)}, $\sem{\Gamma_1} + q \circ \sem{\Gamma_2} \tee \sem{e} \sem{v} : \sem{B} \at 1$, which, by translation, is equivalent to $\sem{\Gamma_1 + q \cdot \Gamma_2} \tee \sem{e \, v} : \sem{B} \at 1$, as required.
    \\
    \item[Case BIND.]
    By assumption, have \textbf{(1)} $\Gamma_1 \tee e_1 : A$ and \textbf{(2)} $\Gamma_2, x :^1 A \tee e_2 : B$. 
    
    By inductive hypothesis on each of \textbf{(1)} and \textbf{(2)}, have \textbf{(3)} $\sem{\Gamma_1} \tee \sem{e_1} : \sem{A} \at 1$ and \textbf{(4)} $\sem{\Gamma_2}, x : \sem{A} \at 1 \tee \sem{e_2} : \sem{B} \at 1$. 
    
    By weakening on each of \textbf{(3)} and \textbf{(4)}, have \textbf{(5)} $\sem{\Gamma_1} + \sem{\Gamma_2} \tee \sem{e_1} : A \at 1$ and \textbf{(6)} $\sem{\Gamma_1} + \sem{\Gamma_2}, x : \sem{A} \at 1 \tee \sem{e_2} : \sem{B} \at 1$. 
    
    Thus, by LET on \textbf{(5)} and \textbf{(6)}, $\sem{\Gamma_1} + \sem{\Gamma_2} \tee \llet x = \sem{e_1} \lin \sem{e_2} : \sem{B} \at 1$, which, by translation, is equivalent to $\sem{\Gamma_1 + \Gamma_2} \tee \sem{\llet x \leftarrow e_1 \lin e_2} : \sem{B} \at 1$, as required.
    \\
    \item[Case $\boxempty$E.]
    By assumption, have \textbf{(1)} $\Gamma_1 \tee v : \boxempty^{q_2} A$ and \textbf{(2)} $\Gamma_2, x :^{q_1 \cdot q_2} A \tee e : B$. 
    
    By (1.7) on \textbf{(1)} and inductive hypothesis on \textbf{(2)}, have \textbf{(3)} $q_1 \circ \sem{\Gamma_1} \tee \sem{v} : \boxempty^{q_2} \sem{A} \at q_1$ and \textbf{(4)} $\sem{\Gamma_2}, x : \sem{A} \at q_1 \circ q_2 \tee \sem{e} : \sem{B} \at 1$. 
    
    By weakening on each of \textbf{(3)} and \textbf{(4)}, have \textbf{(5)} $q \circ \sem{\Gamma_1} + \sem{\Gamma_2} \tee \sem{v} : \boxempty^{q_2} \sem{A} \at q_1$ and \textbf{(6)} $q \circ \sem{\Gamma_1} + \sem{\Gamma_2}, x : \sem{A} \at q_1 \circ q_2 \tee \sem{e} : \sem{B} \at 1$. 
    
    By $\boxempty$E on \textbf{(5)}, have \textbf{(7)} $q \circ \sem{\Gamma_1} + \sem{\Gamma_2} \tee \mathrm{unbox}_{q_2}\sem{v} : \sem{A} \at q_1 \circ q_2$. 
    
    Thus, by LET on \textbf{(6)} and \textbf{(7)}, $q \circ \sem{\Gamma_1} + \sem{\Gamma_2} \tee \llet x = \mathrm{unbox}_{q_2}\sem{v} \lin \sem{e} : \sem{B} \at 1$, which, by translation, is equivalent to $\sem{q \cdot \Gamma_1 + \Gamma_2} \tee \sem{\llet_{q_1} \mathrm{box}_{q_2} x = v \lin e} : \sem{B} \at 1$, as required.
    \\
    \item[Case SPLIT.]
    By assumption, have \textbf{(1)} $\Gamma_1 \tee v : A_1 \times A_2$ and \textbf{(2)} $\Gamma_2, x :^q A_1, y :^q A_2 \tee e : B$.

    By weakening on each of \textbf{(3)} and \textbf{(4)}, have \textbf{(5)} $q \circ \sem{\Gamma_1} + \sem{\Gamma_2} \tee \sem{v} : A_1 \times A_2 \at 1$ and \textbf{(6)} $q \circ \sem{\Gamma_1} + \sem{\Gamma_2}, x : \sem{A_1} \at q, y : \sem{A_2} \at q \tee \sem{e_2} : \sem{B} \at 1$. 
    
    Thus, by SPLIT on \textbf{(5)} and \textbf{(6)}, $q \circ \sem{\Gamma_1} + \sem{\Gamma_2} \tee \llet_q(x, y) = \sem{e_1} \lin \sem{e_2} : \sem{B} \at 1$, which, by translation, is equivalent to $\sem{q \cdot \Gamma_1 + \Gamma_2} \tee \sem{\llet_q(x, y) = e_1 \lin e_2} : \sem{B} \at 1$, as required.

    \item[Case CASE.]
    By assumption, have \textbf{(1)} $\Gamma_1 \tee v : A + B$, \textbf{(2)} $\Gamma_2, x :^q A \tee e_1 : C$ and \textbf{(3)} $\Gamma_2, y :^q B \tee e_2 : C$. 
    
    By (1.7) on \textbf{(1)} and inductive hypothesis on each of \textbf{(2)} and \textbf{(3)}, have \textbf{(4)} $q \circ \sem{\Gamma_1} \tee \sem{v} : \sem{A} + \sem{B} \at q$, \textbf{(5)} $\sem{\Gamma_2}, x : \sem{A} \at q \tee \sem{e_1} : \sem{C} \at 1$ and \textbf{(6)} $\sem{\Gamma_2}, y : \sem{B} \at q \tee \sem{e_2} : \sem{C} \at 1$.

  \end{description}
\end{proof}

\begin{lemma}
  If $\Gamma_1 \tee e_1 : \boxempty^{q_1} A$ and $\Gamma_2, x^\prime :^{q_1} A \tee e_2 : \boxempty^{q_2} B$, then $\Gamma_1 + \Gamma_2 \tee \llet x \leftarrow e_1 \lin \llet \bbox_{q_1} x^\prime = x \lin e_2 : \boxempty^{q_2} B$.
\end{lemma}

\begin{proof}
  By assumption, have \textbf{(1)} $\Gamma_1 \tee e_1 : \boxempty^{q_1} A$ and \textbf{(2)} $\Gamma_2, x^\prime :^{q_1} A \tee e_2 : \boxempty^{q_2} B$.

  By VAR, have \textbf{(3)} $\cdot, x :^1 \boxempty^{q_1} A \tee x : \boxempty^{q_1} A$.

  By $\boxempty$E on \textbf{(2)} and \textbf{(3)}, have \textbf{(4)} $\cdot, x :^1 \boxempty^{q_1} A + \Gamma_2 \tee \llet \bbox_{q_1} x^\prime = x \lin e_2 : \boxempty^{q_2} B$.

  Thus, by BIND on \textbf{(1)} and \textbf{(5)}, $\Gamma_1 + \Gamma_2 \tee \llet x \leftarrow e_1 \lin \llet \bbox_{q_1} x^\prime = x \lin e_2 : \boxempty^{q_2} B$, as required.
  
\end{proof}

     \item[Case CASE.]
     By assumption, have \textbf{(1)} $\Gamma_1 \tee e_1 : A + B \at q_1$, \textbf{(2)} $\Gamma_2, x : A \at q_1 \tee e_3 : C \at q_2$, and \textbf{(3)} $\Gamma_2, y : B \at q_1 \tee e_3 : C \at q_2$.
     
     By inductive hypothesis on each of \textbf{(1)}, \textbf{(2)} and \textbf{(3)}, have \textbf{(4)} $\sem{\Gamma_1} \tee \sem{e_1} : \boxempty^{q_1} (\sem{A} + \sem{B})$, \textbf{(5)} $\sem{\Gamma_2}, x :^{q_1} \sem{A} \tee \sem{e_3} : \boxempty^{q_2} \sem{C}$ and \textbf{(6)} $\sem{\Gamma_2}, y :^{q_1} \sem{B} \tee \sem{e_3} : \boxempty^{q_2} \sem{C}$.

     By VAR, have \textbf{(7)} $\cdot, z^\prime :^1 \sem{A} + \sem{B} \tee z^\prime : \sem{A} + \sem{B}$.

     By CASE on \textbf{(5)}, \textbf{(6)} and \textbf{(7)}, have \textbf{(8)} $\sem{\Gamma_2}, z^\prime :^{q_1} \sem{A} + \sem{B} \tee \mathrm{case} \, z^\prime \{x.\sem{e_2}; y.\sem{e_3}\} : \boxempty^{q_2} \sem{C}$.
     
     Thus, by (1.9) on \textbf{(4)} and \textbf{(8)}, $\sem{\Gamma_1} + \sem{\Gamma_2} \tee \llet z \leftarrow \sem{e_1} \lin \llet \bbox_{q_1} z^\prime = z \lin \mathrm{case} \, z^\prime \{x.\sem{e_2}; y.\sem{e_3}\} : \boxempty^{q_2} \sem{C}$, which, by translation, is equivalent to $\sem{\Gamma_1 + \Gamma_2} \tee \sem{\mathrm{case}_{q_1} e_1 \{x.e_2; y.e_3\}} : \boxempty^{q_2} \sem{C}$, as required.
  \end{description}
\end{proof}

\subsection{Equational Laws for Graded Modal Type Theory}

\paragraph{Beta laws}

\begin{align*}
  (\lambda x.\, e)\, v &\equiv e[v/x] : A \\
  \llet x \leftarrow v \lin e &\equiv e[v/x] : A \\
  \llet_{q_1} \bbox_{q_2} x = \bbox_{q_2}\, v \lin e &\equiv e[v/x] : A \\
  \llet_q (x, y) = (v_1, v_2) \lin e &\equiv e[v_1/x, v_2/y] : A \\
  \case_q\, (\inl\, v)\, \{x.e_1;\; y.e_2\} &\equiv e_1[v/x] : A \\
  \case_q\, (\inr\, v)\, \{x.e_1;\; y.e_2\} &\equiv e_2[v/y] : A
\end{align*}

\begin{lemma}[Type Preservation for Beta Laws]
\label{lem:graded-modal-type-preservation-beta-laws}
  If\/ $e_1 \equiv e_2 : A$, then $\Gamma \tee e_1 : A$ if and only if $\Gamma \tee e_2 : A$.
\end{lemma}

\paragraph{Eta laws}

\begin{align*}
  \lambda x.\, v\, x &\equiv v : \,^q A \rightarrow B
    && (x \notin \mathrm{fv}(v)) \\
  \llet x \leftarrow v \lin () &\equiv v : \unit \\
  \llet_1 (x, y) = v \lin (x, y) &\equiv v : A \times B \\
  \case_1\, v\, \{x.\, \inl\, x;\; y.\, \inr\, y\} &\equiv v : A + B \\
  \llet_{1} \bbox_{q} x = v \lin \bbox_{q}\, x &\equiv v : \boxempty^{q} A
\end{align*}

\begin{lemma}[Type Preservation for Eta Laws]
\label{lem:graded-modal-type-preservation-eta-laws}
  If\/ $e_1 \equiv e_2 : A$, then $\Gamma \tee e_1 : A$ if and only if $\Gamma \tee e_2 : A$.
\end{lemma}

\paragraph{Evaluation contexts}
Recall $E ::= [\cdot] \mid E\, v \mid \llet x \leftarrow E \lin e$.

\paragraph{Commuting conversions}

\begin{align*}
  E[\llet x \leftarrow e_1 \lin e_2]
    &\equiv \llet x \leftarrow e_1 \lin E[e_2] : A
    && (x \notin \mathrm{fv}(E)) \\
  E[\llet_{q_1} \bbox_{q_2} x = v \lin e]
    &\equiv \llet_{q_1} \bbox_{q_2} x = v \lin E[e] : A
    && (x \notin \mathrm{fv}(E)) \\
  E[\llet_q (x, y) = v \lin e]
    &\equiv \llet_q (x, y) = v \lin E[e] : A
    && (x, y \notin \mathrm{fv}(E)) \\
  E[\case_q\, v\, \{x.e_1;\; y.e_2\}]
    &\equiv \case_q\, v\, \{x.\, E[e_1];\; y.\, E[e_2]\} : A
    && (x, y \notin \mathrm{fv}(E))
\end{align*}

\begin{lemma}[Type Preservation for Commuting Conversions]
\label{lem:graded-modal-type-preservation-commuting-conversions}
  If\/ $e_1 \equiv e_2 : A$, then $\Gamma \tee e_1 : A$ if and only if $\Gamma \tee e_2 : A$.
\end{lemma}

\begin{theorem}[Type Preservation for Graded Modal Type Theory]
\label{lem:graded-modal-type-preservation}
  If\/ $e_1 \equiv e_2 : A$, then $\Gamma \tee e_1 : A$ if and only if $\Gamma \tee e_2 : A$.
\end{theorem}
\begin{proof}
  By Lemmas~\ref{lem:graded-modal-type-preservation-beta-laws}, \ref{lem:graded-modal-type-preservation-eta-laws} and \ref{lem:graded-modal-type-preservation-commuting-conversions}.
\end{proof}

\subsection{Equational Laws for Mode Qualifiers}

\paragraph{Beta laws}

\begin{align*}
  (\lambda x.\, e)\, v &\equiv e[v/x] : A \at \mu_2 \\
  \llet x = v \lin e &\equiv e[v/x] : A \at \mu_2 \\
  \unbox_\nu\, (\bbox_\nu\, v) &\equiv v : A \at \nu \circ \mu \\
  \llet_{\mu_1} (x, y) = (v_1, v_2) \lin e &\equiv e[v_1/x, v_2/y] : A \at \mu_2 \\
  \case_{\mu_1}\, (\inl\, v)\, \{x.e_1;\; y.e_2\} &\equiv e_1[v/x] : A \at \mu_2 \\
  \case_{\mu_1}\, (\inr\, v)\, \{x.e_1;\; y.e_2\} &\equiv e_2[v/y] : A \at \mu_2
\end{align*}

\begin{lemma}[Type Preservation for Beta Laws]
\label{lem:mode-qualifiers-type-preservation-beta-laws}
  If\/ $e_1 \equiv e_2 : A \at \mu$, then $\Gamma \tee e_1 : A \at \mu$ if and only if $\Gamma \tee e_2 : A \at \mu$.
\end{lemma}

\paragraph{Eta laws}

\begin{align*}
  \lambda x.\, v\, x &\equiv v : (A \at \mu_1 \rightarrow B \at \mu_2) \at \mu
    && (x \notin \mathrm{fv}(v)) \\
  \llet x = e \lin () &\equiv e : \unit \at \mu \\
  \llet_1 (x, y) = e \lin (x, y) &\equiv e : A \times B \at \mu \\
  \case_1\, e\, \{x.\, \inl\, x;\; y.\, \inr\, y\} &\equiv e : A + B \at \mu \\
  \bbox_\nu\, (\unbox_\nu\, e) &\equiv e : \boxempty^\nu A \at \mu
\end{align*}

\begin{lemma}[Type Preservation for Eta Laws]
\label{lem:mode-qualifiers-type-preservation-eta-laws}
  If\/ $e_1 \equiv e_2 : A \at \mu$, then $\Gamma \tee e_1 : A \at \mu$ if and only if $\Gamma \tee e_2 : A \at \mu$.
\end{lemma}

\paragraph{Evaluation contexts}
Recall:
\begin{align*}
  E &::= [\cdot] \mid \bbox_\nu\, E \mid (E, e) \mid (v, E) \mid \inl\, E \mid \inr\, E \\
    &\mid E\, e_2 \mid v_1\, E \mid \unbox_\nu\, E \\
    &\mid \llet x = E \lin e \mid \llet_\mu (x, y) = E \lin e \\
    &\mid \case_\mu\, E\, \{x.e_1;\; y.e_2\}
\end{align*}

\paragraph{Commuting conversions}

\begin{align*}
  E[\llet x = e_1 \lin e_2]
    &\equiv \llet x = e_1 \lin E[e_2] : A \at \mu_2
    && (x \notin \mathrm{fv}(E)) \\
  E[\llet_{\mu_1} (x, y) = e_1 \lin e_2]
    &\equiv \llet_{\mu_1} (x, y) = e_1 \lin E[e_2] : A \at \mu_2
    && (x, y \notin \mathrm{fv}(E)) \\
  E[\case_{\mu_1}\, e\, \{x.e_1;\; y.e_2\}]
    &\equiv \case_{\mu_1}\, e\, \{x.\, E[e_1];\; y.\, E[e_2]\} : A \at \mu_2
    && (x, y \notin \mathrm{fv}(E))
\end{align*}

\begin{lemma}[Type Preservation for Commuting Conversions]
\label{lem:mode-qualifiers-type-preservation-commuting-conversions}
  If\/ $e_1 \equiv e_2 : A \at \mu$, then $\Gamma \tee e_1 : A \at \mu$ if and only if $\Gamma \tee e_2 : A \at \mu$.
\end{lemma}

\begin{theorem}[Type Preservation for Mode Qualifiers]
  If\/ $e_1 \equiv e_2 : A \at \mu$, then $\Gamma \tee e_1 : A \at \mu$ if and only if $\Gamma \tee e_2 : A \at \mu$.
\end{theorem}
\begin{proof}
  By Lemmas~\ref{lem:mode-qualifiers-type-preservation-beta-laws}, \ref{lem:mode-qualifiers-type-preservation-eta-laws} and \ref{lem:mode-qualifiers-type-preservation-commuting-conversions}.
\end{proof}
