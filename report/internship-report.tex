\section{Calculi}

\subsection{Graded Modal Type Theory}

A fine-grained call-by-value calculus in the style of \cite{Abel:unified,Choudhury:graded}.

\paragraph{Syntax}

\begin{align*}
  A &::= \,^q A \rightarrow B \mid \boxempty^q A \mid A \times B \mid A + B \mid \unit \\
  v &::= x \mid \lambda x.\, e \mid \bbox_q\, v \mid (v_1, v_2) \mid () \mid \inl\, v \mid \inr\, v \\
  e &::= v \mid e\, v
      \mid \llet x \leftarrow e_1 \lin e_2 \\
    &\mid \llet_{q_1} \bbox_{q_2} x = v \lin e \\
    &\mid \llet_q (x, y) = v \lin e
      \mid \case_q\, v\, \{x.e_1;\; y.e_2\} \\
    &\mid \absurd \, e
\end{align*}

\paragraph{Value typing rules} \hfill \framebox{$\Gamma \tee v : A$}

\begin{mathpar}
  \inferrule*[right=var]{
    \phantom{.}
  }{
    0 \cdot \Gamma, x :^1 A, 0 \cdot \Gamma' \tee x : A
  } \and
  \inferrule*[right=sub]{
    q \leq q' \\
    \Gamma, x :^q A, \Gamma' \tee v : B
  }{
    \Gamma, x :^{q'} A, \Gamma' \tee v : B
  } \and
  \inferrule*[right=lam]{
    \Gamma, x :^q A \tee e : B
  }{
    \Gamma \tee \lambda x.\, e : \,^q A \rightarrow B
  } \and
  \inferrule*[right=unit]{
    \phantom{.}
  }{
    \Gamma \tee () : \unit
  } \and
  \inferrule*[right=pair]{
    \Gamma_1 \tee v_1 : A \\
    \Gamma_2 \tee v_2 : B
  }{
    \Gamma_1 + \Gamma_2 \tee (v_1, v_2) : A \times B
  } \and
  \inferrule*[right=inl]{
    \Gamma \tee v : A
  }{
    \Gamma \tee \inl\, v : A + B
  } \and
  \inferrule*[right=inr]{
    \Gamma \tee v : B
  }{
    \Gamma \tee \inr\, v : A + B
  } \and
  \inferrule*[right=$\boxempty$I]{
    \Gamma \tee v : A
  }{
    q\cdot\Gamma \tee \bbox_q\, v : \boxempty^q A
  }
\end{mathpar}

\paragraph{Expression typing rules} \hfill \framebox{$\Gamma \tee e : B$}

\begin{mathpar}
  \inferrule*[right=app]{
    \Gamma_1 \tee e : \,^q A \rightarrow B \\
    \Gamma_2 \tee v : A
  }{
    \Gamma_1 + q\cdot\Gamma_2 \tee e\, v : B
  } \and
  \inferrule*[right=bind]{
    \Gamma_1 \tee e_1 : A \\
    \Gamma_2, x :^1 A \tee e_2 : B
  }{
    \Gamma_1 + \Gamma_2 \tee \llet x \leftarrow e_1 \lin e_2 : B
  } \and
  \inferrule*[right=$\boxempty$E]{
    \Gamma_1 \tee v : \boxempty^{q_2} A \\
    \Gamma_2, x :^{q_1 \cdot q_2} A \tee e : B
  }{
    q_1\cdot\Gamma_1 + \Gamma_2 \tee \llet_{q_1} \bbox_{q_2} x = v \lin e : B
  } \and
  \inferrule*[right=split]{
    \Gamma_1 \tee v : A_1 \times A_2 \\
    \Gamma_2, x :^q A_1, y :^q A_2 \tee e : B
  }{
    q\cdot\Gamma_1 + \Gamma_2 \tee \llet_q (x, y) = v \lin e : B
  } \and
  \inferrule*[right=case]{
    \Gamma_1 \tee v : A + B \\
    \Gamma_2, x :^q A \tee e_1 : C \\
    \Gamma_2, y :^q B \tee e_2 : C
  }{
    q\cdot\Gamma_1 + \Gamma_2 \tee \case_q\, v\, \{x.e_1;\; y.e_2\} : C
  } \and
  \inferrule*[right=absurd]{
    \Gamma \tee e : \zero
  }{
    \top\cdot\Gamma \tee \absurd \, e : A
  }
\end{mathpar}

\paragraph{Semantics} \hfill \framebox{$e \red e'$}

\begin{align*}
  (\lambda x.\, e)\, v &\red e[v/x] \\
  \llet x \leftarrow v \lin e &\red e[v/x] \\
  \llet_{q_1} \bbox_{q_2} x = \bbox_{q_2}\, v \lin e &\red e[v/x] \\
  \llet_q (x, y) = (v_1, v_2) \lin e &\red e[v_1/x, v_2/y] \\
  \case_q\, (\inl\, v)\, \{x.e_1;\; y.e_2\} &\red e_1[v/x] \\
  \case_q\, (\inr\, v)\, \{x.e_1;\; y.e_2\} &\red e_2[v/y]
\end{align*}

\subsection{Mode Qualifiers}

\paragraph{Syntax}

\begin{align*}
  A, B &::= A \at \mu_1 \rightarrow B \at \mu_2 \mid \boxempty^\nu A \mid A \times B \mid A + B \mid \unit \\
  e &::= x \mid \lambda x.\, e \mid e_1\, e_2
      \mid \bbox_\nu\, e \mid \unbox_\nu\, e
      \mid (e_1, e_2) \mid () \mid \inl\, e \mid \inr\, e \mid \absurd \, e \\
    &\mid \llet x = e_1 \lin e_2
      \mid \llet_\mu (x, y) = e \lin e
      \mid \case_\mu\, e\, \{x.e_1;\; y.e_2\} \\
  \Gamma &::= \cdot \mid \Gamma, x : A \at \mu \mid \Gamma, \lock_\mu
\end{align*}

We define $\mathrm{locks}(\Gamma)$ as the composition of all locks in $\Gamma$:
\begin{align*}
  \mathrm{locks}(\cdot) &= \mathrm{id} \\
  \mathrm{locks}(\Gamma, x :_\mu A) &= \mathrm{locks}(\Gamma) \\
  \mathrm{locks}(\Gamma, \lock_\mu) &= \mathrm{locks}(\Gamma) \cdot \mu
\end{align*}

\paragraph{Typing rules} \hfill \framebox{$\Gamma \tee e : A \at \mu$}

\begin{mathpar}
  \inferrule*[right=var]{
    \mu_1 \geq \mathrm{locks}(\Gamma') \cdot \mu_2
  }{
    0 \cdot \Gamma, x : A \at \mu_1, 0 \cdot \Gamma' \tee x : A \at \mu_2
  } \and
  \inferrule*[right=lam]{
    \Gamma, \lock_\mu, x : A \at \mu_1 \tee e : B \at \mu_2
  }{
    \Gamma \tee \lambda x.\, e : (A \at \mu_1 \rightarrow B \at \mu_2) \at \mu
  } \and
  \inferrule*[right=app]{
    \Gamma_1 \tee e_1 : (A \at \mu_1 \rightarrow B \at \mu_2) \at \_ \\\\
    \Gamma_2 \tee e_2 : A \at \mu_1 \\
    \mu_2 \geq \mu_3
  }{
    \Gamma_1 + \Gamma_2 \tee e_1\, e_2 : B \at \mu_3
  } \and
  \inferrule*[right=$\boxempty$I]{
    \Gamma \tee e : A \at \mu \cdot \nu
  }{
    \Gamma \tee \bbox_\nu\, e : \boxempty^\nu A \at \mu
  } \and
  \inferrule*[right=$\boxempty$E]{
    \Gamma \tee e : \boxempty^\nu A \at \mu
  }{
    \Gamma \tee \unbox_\nu\, e : A \at \mu \cdot \nu
  } \and
  \inferrule*[right=let]{
    \Gamma_1 \tee e_1 : A \at \mu_1 \\
    \Gamma_2, x : A \at \mu_1 \tee e_2 : B \at \mu_2
  }{
    \Gamma_1 + \Gamma_2 \tee \llet x = e_1 \lin e_2 : B \at \mu_2
  } \and
  \inferrule*[right=unit]{
    \phantom{.}
  }{
    \Gamma \tee () : \unit \at \mu
  } \and
  \inferrule*[right=pair]{
    \Gamma_1 \tee e_1 : A \at \mu \\
    \Gamma_2 \tee e_2 : B \at \mu
  }{
    \Gamma_1 + \Gamma_2 \tee (e_1, e_2) : A \times B \at \mu
  } \and
  \inferrule*[right=split]{
    \Gamma_1 \tee e_1 : A \times B \at \mu_1 \\\\
    \Gamma_2, x : A \at \mu_1, y : B \at \mu_1 \tee e_2 : C \at \mu_2
  }{
    \Gamma_1 + \Gamma_2 \tee \llet_{\mu_1} (x, y) = e_1 \lin e_2 : C \at \mu_2
  } \and
  \inferrule*[right=inl]{
    \Gamma \tee e : A \at \mu
  }{
    \Gamma \tee \inl\, e : A + B \at \mu
  } \and
  \inferrule*[right=inr]{
    \Gamma \tee e : B \at \mu
  }{
    \Gamma \tee \inr\, e : A + B \at \mu
  } \and
  \inferrule*[right=case]{
    \Gamma_1 \tee e : A + B \at \mu_1 \\\\
    \Gamma_2, x : A \at \mu_1 \tee e_1 : C \at \mu_2 \\\\
    \Gamma_2, y : B \at \mu_1 \tee e_2 : C \at \mu_2
  }{
    \Gamma_1 + \Gamma_2 \tee \case_{\mu_1}\, e\, \{x.e_1;\; y.e_2\} : C \at \mu_2
  } \and
  \inferrule*[right=absurd]{
    \Gamma \tee e : \zero \at \mu_1 \\
  }{
    \Gamma \tee \absurd \, e : A \at \mu_2
  }
\end{mathpar}

\subsection{Parameterisation of the Calculi}

The graded modal calculus and the mode qualifier calculus have a similar parameterisation. Both are characterised by an algebra $(S, +, \cdot, \le, 0, 1)$, where 0 is the additive identity and multiplicative zero and 1 is the multiplicative identity. 

$\Gamma_1 + \Gamma_2$, where $\Gamma_1$ and $\Gamma_2$ are contexts, is defined by adding modes point-wise over variables, and $q \cdot \Gamma$, where $q$ is a mode and $\Gamma$ is a context, is defined by multiplying modes point-wise over the variables in $\Gamma$.

$\Gamma_1 \le \Gamma_2$ is also defined by comparing variables point-wise. In the mode qualifier calculus, if variable modes are equal when compared point-wise, then locks may be compared point-wise contrapositively, where $\mu \le \mu^\prime$ $\Gamma_1, \lock_{\mu^\prime}, \Gamma_2 \le \Gamma_1, \lock_\mu, \Gamma_2$. Two contexts are incomparable if they have a different set of variables or locks in different positions.

We also have a division operator $\setminus$, which is defined by saying that $q_1 \setminus q_2$ is the maximum $q$ such that $q_2 \cdot q \le q_1$. $\Gamma \setminus q$ is defined point-wise over the variables in $\Gamma$.
\\

The laws that must hold for the algebra are:

\begin{law}
\label{law:add-assoc}
  $q_1 + (q_2 + q_3) = (q_1 + q_2) + q_3$.
\end{law}

\begin{law}
\label{law:add-comm}
  $q_1 + q_2 = q_2 + q_1$.
\end{law}

\begin{law}
\label{law:mul-assoc-1}
  $q_1 \cdot (q_2 \cdot q_3) \le (q_1 \cdot q_2) \cdot q_3$.
\end{law}

\begin{law}
\label{law:mul-distr-add-left-1}
  $q_1 \cdot q_2 + q_1 \cdot q_3 \le q_1 \cdot (q_2 + q_3)$.
\end{law}

\begin{law}
\label{law:mul-distr-add-right-1}
  $q_1 \cdot q_3 + q_2 \cdot q_3 \le (q_1 + q_2) \cdot q_3$.
\end{law}

\begin{law}
\label{law:mul-inverse-div}
  $q_2 \cdot (q_1 \setminus q_2) \le q_1$.
\end{law}

\begin{law}
\label{law:add-mono}
  If $q_1 \le q_3$ and $q_2 \le q_4$, then $q_1 + q_2 \le q_3 + q_4$.
\end{law}

\begin{law}
\label{law:mul-mono-left}
  If $q_2 \le q_3$, then $q_1 \cdot q_2 \le q_1 \cdot q_3$.
\end{law}

\begin{law}
\label{law:riesz-decomposition-dual}
  If $q_1 + q_2 \le q_3$, then there exists $q_1^\prime$ and $q_2^\prime$ where $q_1^\prime + q_2^\prime = q_3$ such that $q_1 \le q_1^\prime$ and $q_2 \le q_2^\prime$.
\end{law}

The laws that must hold for the beta-laws are:

\begin{law}
\label{law:mul-assoc-2}
  $(q_1 \cdot q_2) \cdot q_3 \le q_1 \cdot (q_2 \cdot q_3)$.
\end{law}

\begin{law}
\label{law:mul-distr-add-left-2}
  $q_1 \cdot (q_2 + q_3) \le q_1 \cdot q_2 + q_1 \cdot q_3$.
\end{law}

\begin{law}
\label{law:mul-distr-add-right-2}
  $(q_1 + q_2) \cdot q_3 = q_1 \cdot q_3 + q_2 \cdot q_3$.
\end{law}

For local and global modes, we have S = $\{\top, 0, 1, \mathbb{G}\}$ with
$\top \le 0 \le 1 \le \mathbb{G}$, $+$ as maximum, $\top$ as the multiplicative zero, 0 as the multiplicative zero when not considering $\top$ and $\mathbb{G} \cdot \mathbb{G} = \mathbb{G}$.

\subsection{Converting Graded Modal Type Theory to Mode Qualifiers}

\paragraph{Interpretation of types}
%
\begin{align*}
  \sem{\zero} &= \zero \\
  \sem{\unit} &= \unit \\
  \sem{A + B} &= \sem{A} + \sem{B} \\
  \sem{A \times B} &= \sem{A} \times \sem{B} \\
  \sem{\boxempty^{q} A} &= \boxempty^{q}\, \sem{A} \\
  \sem{\,^q A \rightarrow B} &= \sem{A} \at q \rightarrow \sem{B} \at 1
\end{align*}

\paragraph{Contexts}
%
\begin{align*}
  \sem{\cdot} &= \cdot \\
  \sem{\Gamma, x :^0 A} &= \sem{\Gamma} \\
  \sem{\Gamma, x :^q A} &= \sem{\Gamma},\, x : \sem{A} \at q \\
  \sem{\Gamma_1 + \Gamma_2} &= \sem{\Gamma_1} + \sem{\Gamma_2} \\
  \sem{q \cdot \Gamma} &= q \cdot \sem{\Gamma}
\end{align*}

\paragraph{Translation of values}
%
\begin{align*}
  \sem{x} &= x \\
  \sem{()} &= () \\
  \sem{\lambda x.\, e} &= \lambda x.\, \sem{e} \\
  \sem{\bbox_q\, v} &= \bbox_q\; \sem{v} \\
  \sem{(v_1, v_2)} &= (\sem{v_1}, \sem{v_2}) \\
  \sem{\inl\, v} &= \inl\; \sem{v} \\
  \sem{\inr\, v} &= \inr\; \sem{v}
\end{align*}

\paragraph{Translation of expressions}
%
\begin{align*}
  \sem{v} &= \sem{v} \\
  \sem{e\, v} &= \sem{e}\; \sem{v} \\
  \sem{\llet x \leftarrow e_1 \lin e_2}
    &= \llet x = \sem{e_1} \lin \sem{e_2} \\
  \sem{\llet_{q_1} \bbox_{q_2} x = v \lin e}
    &= \llet x = \unbox_{q_2}\; \sem{v} \lin \sem{e} \\

    &= \llet_q (x, y) = \sem{v} \lin \sem{e} \\
  \sem{\case_q\, v\, \{x.e_1;\; y.e_2\}}
    &= \case_q\; \sem{v}\; \{x.\sem{e_1};\; y.\sem{e_2}\} \\
  \sem{\absurd \, e} &= \absurd \, \sem{e}
\end{align*}

\begin{lemma}
\label{lem:mul-left-inverse-div-le}
  $q^\prime \le (q \cdot q^\prime) \setminus q$.
\end{lemma}

\begin{proof}
  Consider the set $M = \{q^{\prime\prime} : q \cdot q^{\prime\prime} = q \cdot q^\prime\}$, the candidates for $q^{\prime\prime} = q \cdot q^\prime \setminus q$.

  Since $q \cdot q^\prime = q \cdot q^\prime$, it follows that $q^\prime \in M$.
  
  Thus, since $q \cdot q^\prime \setminus q$ is given by $\max(M)$, $q^\prime \le q \cdot q^\prime \setminus q$, as required.
  
\end{proof}

\begin{lemma}
\label{lem:mul-left-inverse-div-le-contexts}
  $\Gamma \le (q \cdot \Gamma) \setminus q$.
\end{lemma}

\begin{proof}
  By the definition of $\setminus$ over contexts being point-wise, this follows from \ref{lem:mul-left-inverse-div-le}.
  
\end{proof}

\begin{lemma}
\label{lem:div-identity}
  $1 \le q \setminus q$.
\end{lemma}

\begin{proof}
  Consider the set $M = \{q^\prime : q \cdot q^\prime = q\}$, the candidates for $q^\prime = q \setminus q$.

  Since 1 is the identity for $\cdot$, it follows that $1 \in M$.
  
  Thus, since $q \setminus q$ is given by $\max(M)$, $1 \le q \setminus q$, as required.
  
\end{proof}

\begin{theorem}[Weakening for Graded Modal Type Theory]
\label{thm:graded-modal-weakening}
  If \ $\Gamma \leq \Gamma^\prime$ and $\Gamma \tee e : A$, then $\Gamma^\prime \tee e : A$.
\end{theorem}

\begin{proof}
  By the assumption $\Gamma \leq \Gamma^\prime$, and the definition of $\le$ over contexts, have $q \le q^\prime$ for every $q$, $q^\prime$ and $x$ such that $\Gamma = \Gamma_1, x :^q A, \Gamma_2$ and $\Gamma^\prime = \Gamma_1^\prime, x :^{q^\prime} A, \Gamma_2^\prime$. Repeatedly apply SUB with each variable $x \in \Gamma$ and the above statement specialised with $x$, to obtain $\Gamma^\prime \tee e : A$, as required.
  
\end{proof}

\begin{theorem}[Weakening for Mode Qualifiers]
\label{thm:mode-qualifiers-weakening}
  If \ $\Gamma \leq \Gamma^\prime$, $\mu^\prime \leq \mu$ and $\Gamma \tee e : A \at \mu$, then $\Gamma^\prime \tee e : A \at \mu^\prime$.
\end{theorem}

\begin{proof}
  Proceed by induction over the structure of the derivation of $\Gamma \tee x : A$.

  For each case involving an addition of two contexts, use law \ref{law:riesz-decomposition-dual} to assume and prove the statement for a decomposition of $\Gamma^\prime$.
  \begin{description}
    \item[Case VAR.]
    By assumption, have $\mu_1 \ge \mathrm{locks}(\Gamma_2) \cdot \mu_2 \ge \mathrm{locks}(\Gamma_2) \cdot \mu_2^\prime$. 
    
    By this and the definition of $\le$, we have $\mu_1^\prime \ge \mu_1 \ge \mathrm{locks}(\Gamma^\prime) \cdot \mu_2^\prime$ for every $\mu_1^\prime$ and $\mu_1$ such that $\Gamma \tee x : A \at \mu_1$ and $\Gamma^\prime \tee x : A \at \mu^\prime_1$. 
    
    Thus, by VAR, have $0 \cdot \Gamma^\prime_1, x : A \at \mu_1^\prime, 0 \cdot \Gamma^\prime_2 \tee x : A \at \mu_2^\prime$, as required.
    \\
    \item[Case LAM.]
    By the inductive hypothesis, have $\Gamma^\prime, \lock_{\mu^\prime}, x : A \at \mu_1 \tee e : B \at \mu_2$. 
    
    By LAM on this, $\Gamma^\prime \tee \lambda x.e : (A \at \mu_1 \rightarrow B \at \mu_2) \at \mu^\prime$, as required.
    \\
    \item[Case APP.]
    By the inductive hypothesis, have $\Gamma^\prime_1 \tee e_1 : (A \at \mu_1 \rightarrow B \at \mu_2) \at \_$ and $\Gamma^\prime_2 \tee e_2 : A \at \mu_1$. 
    
    By APP on these and assumption $\mu_2^\prime \le \mu_2$, $\Gamma^\prime_1 + \Gamma^\prime_2 \tee e_1 e_2 : B \at \mu_2^\prime$ as required.
    \\
    \item[Case $\boxempty$I.]
    By the inductive hypothesis, have $\Gamma^\prime \tee e : A \at \nu \cdot \mu^\prime$.
    
    By $\boxempty$I on this, $\Gamma^\prime \tee \mathrm{box}_ve : \boxempty^vA \at \mu^\prime$, as required.
    \\
    \item[Case $\boxempty$E.]
    By the inductive hypothesis, have $\Gamma^\prime \tee e : \boxempty^vA \at \mu^\prime$.
    
    By $\boxempty$E on this, $\Gamma^\prime \tee \mathrm{unbox}_ve : A \at \nu \cdot \mu^\prime$, as required.
    \\
    \item[Case LET.]
    By the inductive hypothesis, have \textbf{(1)} $\Gamma^\prime_1 \tee e_1 : A \at \mu_1$ and \textbf{(2)} $\Gamma^\prime_2, x : A \at \mu_1 \tee e_2 : B \at \mu_2^\prime$.
    
    By LET on \textbf{(1)} and \textbf{(2)}, $\Gamma^\prime_1 + \Gamma^\prime_2 \tee \llet x = e_1 \lin e_2 : B \at \mu_2^\prime$, as required.
    \\
    \item[Case UNIT.]
    By UNIT, $\Gamma^\prime \tee () : \unit \at \mu^\prime$, as required.
    \\
    \item[Case PAIR.]
    By the inductive hypothesis, have \textbf{(1)} $\Gamma^\prime_1 \tee e_1 : A \at \mu^\prime$ and \textbf{(2)} $\Gamma^\prime_2 \tee e_2 : B \at \mu^\prime.$
    
    By PAIR on \textbf{(1)} and \textbf{(2)}, $\Gamma^\prime_1 + \Gamma^\prime_2 \tee (e_1, e_2) : A \times B \at \mu^\prime$, as required.
    \\
    \item[Case SPLIT.]
    By the inductive hypothesis, have \textbf{(1)} $\Gamma^\prime_1 \tee e : A \times B \at \mu_1$ and \textbf{(2)} $\Gamma^\prime_2, x : A \at \mu_1, y : B \at \mu_1 \tee e_2 : C \at \mu_2^\prime$.
    
    By SPLIT on \textbf{(1)} and \textbf{(2)}, $\Gamma^\prime_1 + \Gamma^\prime_2 \tee \llet_{\mu_1} (x, y) = e_1 \lin e_2 : C \at \mu_2^\prime$, as required.
    \\
    \item[Case INL.]
    By the inductive hypothesis, have $\Gamma^\prime \tee e : A \at \mu^\prime$.
    
    By INL on this, $\Gamma^\prime \tee \inl e : A + B \at \mu^\prime$, as required.
    \\
    \item[Case INR.]
    By the inductive hypothesis, have $\Gamma^\prime \tee e : B \at \mu^\prime$.
    
    By INR on this, $\Gamma^\prime \tee \inr e : A + B \at \mu^\prime$, as required.
    \\
    \item[Case CASE.]
    By the inductive hypothesis, have \textbf{(1)} $\Gamma^\prime_1 \tee e : A + B \at \mu_1$, \textbf{(2)} $\Gamma^\prime_2, x : A \at \mu_1 \tee e_1 : C \at \mu_2^\prime$ and \textbf{(3)} $\Gamma^\prime_2, y : B \at \mu_1 \tee e_2 : C \at \mu_2^\prime$.
    
    By CASE on \textbf{(1)}, \textbf{(2)} and \textbf{(3)}, $\Gamma^\prime_1 + \Gamma^\prime_2 \tee \case_{\mu_1} e \{x.e_1; y.e_2\} : C \at \mu_2^\prime$, as required.
    \\
    \item[Case ABSURD.]
    Because we should not be able to construct a term of type Void, we can weaken a context from an ABSURD derivation as required.
  \end{description}
\end{proof}

\begin{theorem}[Division to Locks for Mode Qualifiers]
\label{thm:mode-qualifiers-div-to-lock}
  If $\Gamma \setminus \mu, \Gamma^\prime \tee e : A \at \mu^\prime$, then $\Gamma, \lock_\mu, \Gamma^\prime \tee e : A \at \mu^\prime$.
\end{theorem}

\begin{proof}
  Proceed by induction over the structure of $\Gamma \setminus \mu, \Gamma^\prime \tee e : A \at \mu^\prime$.
  \begin{description}
    \item[Case VAR.]
    By assumption, have $\mu_1 \setminus \mu \ge \mathrm{locks}(\Gamma^\prime) \cdot \mu_2$, with $\mu_1 \setminus \mu$ being reducible. 
    
    By law \ref{law:mul-mono-left}, have $\mu \cdot (\mu_1 \setminus \mu) \ge \mu \cdot \mathrm{locks}(\Gamma^\prime) \cdot \mu_2$. 
    
    By law \ref{law:mul-inverse-div}, have $\mu_1 \ge \mu \cdot \mathrm{locks}(\Gamma^\prime) \cdot \mu_2$. 
    
    Thus, by VAR, $0 \cdot \Gamma, x : A \at \mu_1, \lock_\mu, 0 \cdot \Gamma^\prime \tee x : A \at \mu_2$, as required.
    \\
    \item[Case LAM.]
    By assumption, have \textbf{(1)} $\Gamma \setminus \mu, \Gamma^\prime, \lock_\mu^\prime, x : A \at \mu_1 \tee e : B \at \mu_2$. 
    
    By inductive hypothesis on \textbf{(1)}, have \textbf{(2)} $\Gamma, \lock_\mu, \Gamma^\prime, \lock_{\mu^\prime}, x : A \at \mu_1 \tee e : B \at \mu_2$.
    
    Thus, by LAM on \textbf{(2)}, $\Gamma, \lock_\mu, \Gamma^\prime \tee \lambda x.e : (A \at \mu_1 \rightarrow B \at \mu_2) \at \mu^\prime$, as required.
    \\
    \item[Case APP.]
    By assumption, have \textbf{(1)} $\Gamma_1 \setminus \mu, \Gamma^\prime \tee e_1 : (A \at \mu_1 \rightarrow B \at \mu_2) \at \_$, \textbf{(2)} $\Gamma_2 \setminus \mu, \Gamma^\prime \tee e_2 : A \at \mu_1$ and \textbf{(3)} $\mu_2 \ge \mu_3$. 
    
    By inductive hypothesis on each of \textbf{(1)} and \textbf{(2)}, have \textbf{(4)} $\Gamma_1, \lock_\mu, \Gamma^\prime_1 \tee e_1 : (A \at \mu_1 \rightarrow B \at \mu_2) \at \_$ and \textbf{(5)} $\Gamma_2, \lock_\mu, \Gamma^\prime_2 \tee e_2 : A \at \mu_1$.
    
    Thus, by APP on \textbf{(3)}, \textbf{(4)} and \textbf{(5)}, $\Gamma, \lock_\mu, \Gamma^\prime \tee e_1 e_2 : B \at \mu_3$, as required.
    \\
    \item[Case $\boxempty$I.]
    By assumption, have \textbf{(1)} $\Gamma \setminus \mu, \Gamma^\prime \tee e : A \at \nu \cdot \mu^\prime$. 
    
    By the inductive hypothesis on \textbf{(1)}, have \textbf{(2)} $\Gamma, \lock_\mu, \Gamma^\prime \tee e : A \at \nu \cdot \mu^\prime$. 
    
    Thus, by $\boxempty$I on \textbf{(2)}, $\Gamma, \lock_\mu, \Gamma^\prime \tee \mathrm{box}_v e : \boxempty^\nu A \at \cdot \mu^\prime$, as required.
    \\
    \item[Case $\boxempty$E.]
    By assumption, have \textbf{(1)} $\Gamma \setminus \mu, \Gamma^\prime \tee e : \boxempty^\nu A \at \mu^\prime$. 
    
    By the inductive hypothesis on \textbf{(1)}, have \textbf{(2)} $\Gamma, \lock_\mu, \Gamma^\prime \tee e : \boxempty^\nu A \at \mu^\prime$. 
    
    Thus, by $\boxempty$E on \textbf{(2)}, $\Gamma, \lock_\mu, \Gamma^\prime \tee \mathrm{unbox}_v e : A \at \nu \cdot \mu^\prime$, as required.
    \\
    \item[Case LET.]
    By assumption, have \textbf{(1)} $\Gamma_1 \setminus \mu, \Gamma^\prime_1 \tee e_1 : A \at \mu_1$ and \textbf{(2)} $\Gamma_2 \setminus \mu, \Gamma^\prime_2, x : A \at \mu_1 \tee e_2 : B \at \mu_2$. 
    
    By the inductive hypothesis on each of \textbf{(1)} and \textbf{(2)}, have \textbf{(3)} $\Gamma_1, \lock_\mu, \Gamma^\prime_1 \tee e_1 : A \at \mu_1$ and \textbf{(4)} $\Gamma_2, \lock_\mu, \Gamma^\prime_2, x : A \at \mu_1 \tee e_2 : B \at \mu_2$. 
    
    Thus, by LET on \textbf{(3)} and \textbf{(4)}, $\Gamma, \lock_\mu, \Gamma^\prime \tee \llet x = e_1 \lin e_2 : B \at \mu_2$, as required.
    \\
    \item[Case UNIT.]
    $\Gamma, \lock_\mu, \Gamma^\prime \tee () : \unit \at \mu^\prime$, as required.
    \\
    \item[Case PAIR.]
    By assumption, have \textbf{(1)} $\Gamma_1 \setminus \mu, \Gamma^\prime_1 \tee e_1 : A \at \mu^\prime$ and \textbf{(2)} $\Gamma_2 \setminus \mu, \Gamma^\prime_2 \tee e_2 : B \at \mu^\prime$. 
    
    By the inductive hypothesis on each of \textbf{(1)} and \textbf{(2)}, have \textbf{(3)} $\Gamma_1, \lock_\mu, \Gamma^\prime_1 \tee e_1 : A \at \mu^\prime$ and \textbf{(4)} $\Gamma_2, \lock_\mu, \Gamma^\prime_2 \tee e_2 : B \at \mu^\prime$. 
    
    Thus, by PAIR on \textbf{(3)} and \textbf{(4)}, $\Gamma, \lock_\mu, \Gamma^\prime \tee (e_1, e_2) : A \times B \at \mu^\prime$, as required.
    \\
    \item[Case SPLIT.]
    By assumption, have \textbf{(1)} $\Gamma_1 \setminus \mu, \Gamma^\prime_1 \tee e_1 : A \times B \at \mu_1$ and \textbf{(2)} $\Gamma_2 \setminus \mu, \Gamma^\prime_2, x : A \at \mu_1, y : B \at \mu_1 \tee e_2 : C \at \mu_2$. 
    
    By the inductive hypothesis on each of \textbf{(1)} and \textbf{(2)}, have \textbf{(3)} $\Gamma_1, \lock_\mu, \Gamma^\prime_1 \tee e_1 : A \times B \at \mu_1$ and \textbf{(4)} $\Gamma_2, \lock_\mu, \Gamma^\prime_2, x : A \at \mu_1, y : B \at \mu_1 \tee e_2 : C \at \mu_2$. 
    
    Thus, by SPLIT on \textbf{(3)} and \textbf{(4)}, $\Gamma, \lock_\mu, \Gamma^\prime \tee \llet_{\mu_1} (x, y) = e_1 \lin e_2 : C \at \mu_2$, as required.
    \\
    \item[Case INL.]
    By assumption, have \textbf{(1)} $\Gamma \setminus \mu, \Gamma^\prime \tee e : A \at \mu^\prime$. 
    
    By the inductive hypothesis on \textbf{(1)}, have \textbf{(2)} $\Gamma, \lock_\mu, \Gamma^\prime \tee e : A \at \mu^\prime$. 
    
    Thus, by INL on \textbf{(2)}, $\Gamma, \lock_\mu, \Gamma^\prime \tee \inl e : A + B \at \mu^\prime$, as required.
    \\
    \item[Case INR.]
    By assumption, have \textbf{(1)} $\Gamma \setminus \mu, \Gamma^\prime \tee e : B \at \mu^\prime$. 
    
    By the inductive hypothesis on \textbf{(1)}, have \textbf{(2)} $\Gamma, \lock_\mu, \Gamma^\prime \tee e : B \at \mu^\prime$. 
    
    Thus, by INR on \textbf{(2)}, $\Gamma, \lock_\mu, \Gamma^\prime \tee \inr e : A + B \at \mu^\prime$, as required.
    \\
    \item[Case CASE.]
    By assumption, have \textbf{(1)} $\Gamma_1 \setminus \mu, \Gamma^\prime_1 \tee e : A + B \at \mu_1$, \textbf{(2)} $\Gamma_2 \setminus \mu, \Gamma^\prime_2, x : A \at \mu_1 \tee e_1 : C \at \mu_2$ and \textbf{(3)} $\Gamma_2 \setminus \mu, \Gamma^\prime_2, y : B \at \mu_1 \tee e_2 : C \at \mu_2$. 
    
    By the inductive hypothesis on each of \textbf{(1)}, \textbf{(2)} and \textbf{(3)}, have \textbf{(4)} $\Gamma_1, \lock_\mu, \Gamma^\prime_1 \tee e : A + B \at \mu_1$, \textbf{(5)} $\Gamma_2, \lock_\mu, \Gamma^\prime_2, x : A \at \mu_1 \tee e_1 : C \at \mu_2$ and \textbf{(6)} $\Gamma_2, \lock_\mu, \Gamma^\prime_2, y : B \at \mu_1 \tee e_2 : C \at \mu_2$.
    
    Thus, by LET on \textbf{(4)}, \textbf{(5)} and \textbf{(6)}, $\Gamma, \lock_\mu, \Gamma^\prime \tee \mathrm{case}_{\mu_1} e \{x.e_1; y.e_2\} : C \at \mu_2$, as required.
    \\
    \item[Case ABSURD.]
    By assumption, have \textbf{(1)} $\Gamma \setminus \mu, \Gamma^\prime \tee e : \zero \at \mu^{\prime\prime}$. 
    
    By the inductive hypothesis on \textbf{(1)}, have \textbf{(2)} $\Gamma, \lock_\mu, \Gamma^\prime \tee e : \zero \at \mu^{\prime\prime}$. 
    
    Thus, by ABSURD on \textbf{(2)}, $\top \cdot \Gamma, \lock_\mu, \top \cdot \Gamma^\prime \tee e : \zero \at \mu^\prime$, as required.
  \end{description}
\end{proof}

\begin{theorem}[Graded Type Theory Values to Mode Qualifiers]
\label{thm:graded-modal-to-mode-qualifiers-vals}
  If\/ $\Gamma \tee v : A$, then $q \cdot \sem{\Gamma} \tee \sem{v} : \sem{A} \at q$ for all $q \neq 0$. 
\end{theorem}

\begin{proof}
  Proceed by induction over the structure of the derivation of $q \cdot \Gamma \tee v : A$.\\
  \begin{description}
    \item[Case VAR.]
    Have $1 \le 1$ by reflexivity of $\le$.
    
    Thus, by VAR, $\cdot, x : \sem{A} \at q \tee x : \sem{A} \at q$, as required.
    \\
    \item[Case SUB.]
    By the inductive hypothesis, have $q^\prime \cdot \sem{\Gamma}, x : \sem{A} \at q^\prime \cdot q_1, q^\prime \cdot \sem{\Gamma^\prime} \tee \sem{v} : \sem{B} \at q$. 
    
    Thus, by the assumption $q_1 \le q_2$ and weakening, $q^\prime \cdot \sem{\Gamma}, x : \sem{A} \at q_2, q^\prime \cdot \sem{\Gamma^\prime} \tee \sem{v} : \sem{B} \at q$, as required.
    \\
    \item[Case LAM.]
    By the inductive hypothesis, have $\sem{\Gamma}, x : \sem{A} \at q \tee \sem{e} : \sem{B} \at 1$. 

    By weakening using \ref{lem:mul-left-inverse-div-le-contexts}, have $(q^\prime \cdot \sem{\Gamma}) \setminus q^\prime, x : \sem{A} \at q \tee \sem{e} : B \at 1$. 
    
    By \ref{thm:mode-qualifiers-div-to-lock}, have $q^\prime \cdot \sem{\Gamma}, \lock_q, x : \sem{A} \at q \tee \sem{e} : B \at 1$. 
    
    Thus, by LAM, $q^\prime \cdot \sem{\Gamma} \tee \lambda x.\sem{e} : (\sem{A} \at q \rightarrow \sem{B} \at 1) \at q^\prime$ which, by the translation, is equivalent to $q^\prime \cdot \sem{\Gamma} \tee \sem{\lambda x.e} : \sem{\,^qA \rightarrow B} \at q^\prime$, as required.
    \\
    \item[Case UNIT.]
    By UNIT, $q \cdot \sem{\Gamma} \tee () : \unit \at q$ as required.
    \\ 
    \item[Case PAIR.]
    By assumption, have \textbf{(1)} $\Gamma_1 \tee v_1 : A$ and \textbf{(2)} $\Gamma_2 \tee v_2 : B$.
    
    By the inductive hypothesis on each of \textbf{(1)} and \textbf{(2)}, have \textbf{(3)} $q \cdot \sem{\Gamma_1} \tee \sem{v_1} : \sem{A} \at q$ and \textbf{(4)} $q \cdot \sem{\Gamma_2} \tee \sem{v_2} : \sem{B} \at q$.
    
    By PAIR on \textbf{(3)} and \textbf{(4)}, have \textbf{(5)} $q \cdot \sem{\Gamma_1} + q \cdot \sem{\Gamma_2} \tee (\sem{v_1}, \sem{v_2}) : \sem{A} \times \sem{B} \at q$
    
    Thus, by weakening using law \ref{law:mul-distr-add-left-1}, $q \cdot (\sem{\Gamma_1} + \sem{\Gamma_2}) \tee (\sem{v_1}, \sem{v_2}) : \sem{A} \times \sem{B} \at q$, which, by translation, is equivalent to $q \cdot (\sem{\Gamma_1} + \sem{\Gamma_2}) \tee \sem{(v_1, v_2)} : \sem{A \times B} \at q$, as required.
    \\ 
    \item[Case INL.]
    By the inductive hypothesis, have $q \cdot \sem{\Gamma} \tee \sem{e} : \sem{A} \at q$. 
    
    By INL on this, $q \cdot \sem{\Gamma} \tee \inl \sem{e} : \sem{A} + B^\prime \at q$. 
    
    Since, for all mode qualifier types $B^\prime$, there exists a graded modal type $B$ such that $B^\prime = \sem{B}$, we have $q \cdot \sem{\Gamma} \tee \inl \sem{e} : \sem{A} + \sem{B} \at q$, and thus, by the translation, $q \cdot \sem{\Gamma} \tee \sem{\inl e} : \sem{A + B} \at q$, as required.
    \\
    \item[Case INR.]
    By the inductive hypothesis, have $q \cdot \sem{\Gamma} \tee \sem{e} : \sem{B} \at q$. 
    
    By INR on this, $q \cdot \sem{\Gamma} \tee \inr \sem{e} : A^\prime + \sem{B} \at q$. 
    
    Since, for all mode qualifier types $A^\prime$, there exists a graded modal type $A$ such that $A^\prime = \sem{A}$, we have $q \cdot \sem{\Gamma} \tee \inr \sem{e} : \sem{A} + \sem{B} \at q$, and thus, by the translation, $q \cdot \sem{\Gamma} \tee \sem{\inr e} : \sem{A + B} \at q$, as required.
    \\
    \item[Case $\boxempty$I.]
    By the inductive hypothesis, have $q^\prime \cdot q \cdot \sem{\Gamma} \tee \sem{v} : \sem{A} \at q^\prime \cdot q$. 
    
    Thus, by $\boxempty$I, $q^\prime \cdot q \cdot \sem{\Gamma} \tee \mathrm{box}_q \sem{v} : \boxempty^q \sem{A} \at q^\prime$, which, by the translation, is equivalent to $q^\prime \cdot q \cdot \sem{\Gamma} \tee \sem{\mathrm{box}_q v} : \sem{\boxempty_q A} \at q^\prime$, as required.
  \end{description}
\end{proof}

\begin{theorem}[Graded Type Theory Expressions to Mode Qualifiers]
\label{thm:graded-modal-to-mode-qualifiers-exprs}
If $\Gamma \tee e : A$, then $\sem{\Gamma} \tee \sem{e} : \sem{A} \at 1$.
\end{theorem}

\begin{proof} 
  Proceed by induction over the structure of the derivation of  $\Gamma \tee e : A$.
  \begin{description}
    \item[Case APP.]
    By assumption, have \textbf{(1)} $\Gamma_1 \tee e : \,^qA \rightarrow B$ and \textbf{(2)} $\Gamma_2 \tee v : A$. 
    
    By inductive hypothesis on \textbf{(1)} and \ref{thm:graded-modal-to-mode-qualifiers-vals} on \textbf{(2)}, have \textbf{(3)} $\sem{\Gamma_1} \tee \sem{e} : (\sem{A} \at q \rightarrow \sem{B} \at 1) \at 1$ and \textbf{(4)} $q \cdot \sem{\Gamma_2} \tee \sem{v} : \sem{A} \at q$. 

    Thus, by APP on \textbf{(3)} and \textbf{(4)} and the statement $1 \le 1$, $\sem{\Gamma_1} + q \cdot \sem{\Gamma_2} \tee \sem{e} \sem{v} : \sem{B} \at 1$, which, by the translation, is equivalent to $\sem{\Gamma_1 + q \cdot \Gamma_2} \tee \sem{e \, v} : \sem{B} \at 1$, as required.
    \\
    \item[Case BIND.]
    By assumption, have \textbf{(1)} $\Gamma_1 \tee e_1 : A$ and \textbf{(2)} $\Gamma_2, x :^1 A \tee e_2 : B$. 
    
    By inductive hypothesis on each of \textbf{(1)} and \textbf{(2)}, have \textbf{(3)} $\sem{\Gamma_1} \tee \sem{e_1} : \sem{A} \at 1$ and \textbf{(4)} $\sem{\Gamma_2}, x : \sem{A} \at 1 \tee \sem{e_2} : \sem{B} \at 1$. 
    
    Thus, by LET on \textbf{(3)} and \textbf{(4)}, $\sem{\Gamma_1} + \sem{\Gamma_2} \tee \llet x = \sem{e_1} \lin \sem{e_2} : \sem{B} \at 1$, which, by the translation, is equivalent to $\sem{\Gamma_1 + \Gamma_2} \tee \sem{\llet x \leftarrow e_1 \lin e_2} : \sem{B} \at 1$, as required.
    \\
    \item[Case $\boxempty$E.]
    By assumption, have \textbf{(1)} $\Gamma_1 \tee v : \boxempty^{q_2} A$ and \textbf{(2)} $\Gamma_2, x :^{q_1 \cdot q_2} A \tee e : B$. 
    
    By \ref{thm:graded-modal-to-mode-qualifiers-vals} on \textbf{(1)} and inductive hypothesis on \textbf{(2)}, have \textbf{(3)} $q_1 \cdot \sem{\Gamma_1} \tee \sem{v} : \boxempty^{q_2} \sem{A} \at q_1$ and \textbf{(4)} $\sem{\Gamma_2}, x : \sem{A} \at q_1 \cdot q_2 \tee \sem{e} : \sem{B} \at 1$. 
    
    By $\boxempty$E on \textbf{(3)}, have \textbf{(5)} $q \cdot \sem{\Gamma_1}\tee \mathrm{unbox}_{q_2}\sem{v} : \sem{A} \at q_1 \cdot q_2$. 
    
    Thus, by LET on \textbf{(4)} and \textbf{(5)}, $q \cdot \sem{\Gamma_1} + \sem{\Gamma_2} \tee \llet x = \mathrm{unbox}_{q_2}\sem{v} \lin \sem{e} : \sem{B} \at 1$, which, by the translation, is equivalent to $\sem{q \cdot \Gamma_1 + \Gamma_2} \tee \sem{\llet_{q_1} \mathrm{box}_{q_2} x = v \lin e} : \sem{B} \at 1$, as required.
    \\
    \item[Case SPLIT.]
    By assumption, have \textbf{(1)} $\Gamma_1 \tee v : A_1 \times A_2$ and \textbf{(2)} $\Gamma_2, x :^q A_1, y :^q A_2 \tee e : B$. 
    
    By \ref{thm:graded-modal-to-mode-qualifiers-vals} on \textbf{(1)} and inductive hypothesis on \textbf{(2)}, have \textbf{(3)} $q \cdot \sem{\Gamma_1} \tee \sem{v} : \sem{A_1} \times \sem{A_2} \at q$ and \textbf{(4)} $\sem{\Gamma_2}, x : \sem{A_1} \at q, y : \sem{A_2} \at q \tee \sem{e} : \sem{B} \at 1$. 
    
    Thus, by SPLIT on \textbf{(3)} and \textbf{(4)}, $q \cdot \sem{\Gamma_1} + \sem{\Gamma_2} \tee \llet_q(x, y) = \sem{e_1} \lin \sem{e_2} : \sem{B} \at 1$, which, by the translation, is equivalent to $\sem{q \cdot \Gamma_1 + \Gamma_2} \tee \sem{\llet_q(x, y) = e_1 \lin e_2} : \sem{B} \at 1$, as required.
    \\
    \item[Case CASE.]
    By assumption, have \textbf{(1)} $\Gamma_1 \tee v : A + B$, \textbf{(2)} $\Gamma_2, x :^q A \tee e_1 : C$ and \textbf{(3)} $\Gamma_2, y :^q B \tee e_2 : C$. 
    
    By \ref{thm:graded-modal-to-mode-qualifiers-vals} on \textbf{(1)} and inductive hypothesis on each of \textbf{(2)} and \textbf{(3)}, have \textbf{(4)} $q \cdot \sem{\Gamma_1} \tee \sem{v} : \sem{A} + \sem{B} \at q$, \textbf{(5)} $\sem{\Gamma_2}, x : \sem{A} \at q \tee \sem{e_1} : \sem{C} \at 1$ and \textbf{(6)} $\sem{\Gamma_2}, y : \sem{B} \at q \tee \sem{e_2} : \sem{C} \at 1$. 
    
    Thus, by CASE on \textbf{(4)}, \textbf{(5)} and \textbf{(6)}, $q \cdot \sem{\Gamma_1} + \sem{\Gamma_2} \tee \mathrm{case}_q\sem{v}\{x.\sem{e_1}, y.\sem{e_2}\} : \sem{C} \at 1$, which, by the translation, is equivalent to $\sem{q \cdot \Gamma_1 + \Gamma_2} \tee \sem{\case_q\{x.e_1, y.e_2\}} : \sem{C} \at 1$, as required.
    \\
    \item[Case ABSURD.]
    By assumption, have \textbf{(1)} $\Gamma \tee e : \zero$.

    By the inductive hypothesis, have \textbf{(2)} $\sem{\Gamma} \tee \sem{e} : \sem{\zero} \at q_1$.

    Thus, by ABSURD, $\top \cdot \sem{\Gamma} \tee \absurd \, \sem{e} : \sem{A} \at q_2$, which, by the translation, is equivalent to $\sem{\top \cdot \Gamma} \tee \sem{\absurd \, e} : \sem{A} \at q_2$, as required.
  \end{description}
\end{proof}

\begin{lemma}
\label{lem:mode-qualifiers-box-cut}
  If $\Gamma_1 \tee e_1 : \boxempty^{q_1} A$ and $\Gamma_2, x^\prime :^{q_1} A \tee e_2 : \boxempty^{q_2} B$, then $\Gamma_1 + \Gamma_2 \tee \llet x \leftarrow e_1 \lin \llet \bbox_{q_1} x^\prime = x \lin e_2 : \boxempty^{q_2} B$.
\end{lemma}

\begin{proof}
  By assumption, have \textbf{(1)} $\Gamma_1 \tee e_1 : \boxempty^{q_1} A$ and \textbf{(2)} $\Gamma_2, x^\prime :^{q_1} A \tee e_2 : \boxempty^{q_2} B$.

  By VAR, have \textbf{(3)} $\cdot, x :^1 \boxempty^{q_1} A \tee x : \boxempty^{q_1} A$.

  By $\boxempty$E on \textbf{(2)} and \textbf{(3)}, have \textbf{(4)} $\cdot, x :^1 \boxempty^{q_1} A + \Gamma_2 \tee \llet \bbox_{q_1} x^\prime = x \lin e_2 : \boxempty^{q_2} B$.

  Thus, by BIND on \textbf{(1)} and \textbf{(5)}, $\Gamma_1 + \Gamma_2 \tee \llet x \leftarrow e_1 \lin \llet \bbox_{q_1} x^\prime = x \lin e_2 : \boxempty^{q_2} B$, as required.
  
\end{proof}

\begin{theorem}[Mode Qualifiers to Graded Type Theory]
\label{thm:mode-qualifiers-to-graded-modal}
  If $\Gamma \tee e : A \at q$, then $\sem{\Gamma} \tee \sem{e} : \boxempty^{q}\, \sem{A}$.
\end{theorem}

\begin{proof}
  Proceed by induction over the derivation of $\Gamma \tee e : A \at q$.
  \begin{description}
    \item[Case VAR.]
    By VAR, have \textbf{(1)} $\cdot, x :^1 \sem{A} \tee x : \sem{A}$.

    By weakening on \textbf{(1)}, using \ref{lem:div-identity}, have \textbf{(2)} $\cdot, x :^{q_2 \setminus q_2} \sem{A} \tee x : \sem{A}$.
    
    By weakening on \textbf{(2)}, using \ref{lem:mul-left-inverse-div-le}, have \textbf{(3)} $\cdot, x :^{((\mathrm{locks}(\Gamma^\prime) \cdot q_2) \setminus \mathrm{locks}(\Gamma^\prime)) \setminus q_2} \sem{A} \tee x : \sem{A}$.
    
    By weakening on \textbf{(3)} using the assumption $q_1 \ge \mathrm{locks}(\Gamma^\prime) \cdot q_2$, have \textbf{(4)} $\cdot, x :^{(q_1 \setminus \mathrm{locks}(\Gamma^\prime)) \setminus q_2} \sem{A} \tee x : \sem{A}$.
    
    By $\boxempty$I on \textbf{(4)}, have \textbf{(5)} $\cdot, x :^{q_2 \cdot ((q_1 \setminus \mathrm{locks}(\Gamma^\prime)) \setminus q_2)} \sem{A} \tee x : \boxempty^{q_2} \sem{A}$.
    
    By weakening using law \ref{law:mul-inverse-div} on \textbf{(6)}, have \textbf{(7)} $\cdot, x :^{q_1 \setminus \mathrm{locks}(\Gamma^\prime)} \sem{A} \tee x : \boxempty^{q_2} \sem{A}$.
    
    Thus, by weakening on \textbf{(7)}, $0 \cdot \sem{\Gamma} \setminus \mathrm{locks}(\Gamma^\prime), x :^{q_1 \setminus \mathrm{locks}(\Gamma^\prime)} \sem{A}, 0 \cdot \sem{\Gamma^\prime} \tee x : \boxempty^{q_2} \sem{A}$, which, by the translation, is equivalent to $\sem{0 \cdot \Gamma, x : A \at q_1, 0 \cdot \Gamma^\prime} \tee \sem{x} : \boxempty^{q_2} \sem{A}$, as required.
    \\
    \item[Case LAM.]
    By assumption, have \textbf{(1)} $\Gamma, \lock_q, x : A \at q_1 \tee e : B \at q_2$.
    
    By the inductive hypothesis on \textbf{(1)}, have \textbf{(2)} $\sem{\Gamma} \setminus q, x :^{q_1} \sem{A} \tee \sem{e} : \boxempty^{q_2} \sem{B}$.
    
    By LAM on \textbf{(2)}, have \textbf{(3)} $\sem{\Gamma} \setminus q \tee \lambda x.\sem{e} : \,^{q_1} \sem{A} \rightarrow \boxempty^{q_2} \sem{B}$.
    
    Thus, by $\boxempty$I on \textbf{(3)}, $\sem{\Gamma} \tee \bbox_q (\lambda x.\sem{e}) : \boxempty^q (\,^{q_1} \sem{A} \rightarrow \boxempty^{q_2} \sem{B})$, which, by the translation, is equivalent to $\sem{\Gamma} \tee \sem{\lambda x.e} : \boxempty^q \sem{A \at q_1 \rightarrow B \at q_2}$, as required.
    \\
    \item[Case APP.]
    By assumption, have \textbf{(1)} $\Gamma_1 \tee e_1 : (A \at q_1 \rightarrow B \at q_2) \at q$ and \textbf{(2)} $\Gamma_2 \tee e_2 : A \at q_1$.

    By the inductive hypothesis on each of \textbf{(1)} and \textbf{(2)}, have \textbf{(3)} $\sem{\Gamma_1} \tee \sem{e_1} : \boxempty^q (\,^{q_1} \sem{A} \rightarrow \boxempty^{q_2} \sem{B})$ and \textbf{(4)} $\sem{\Gamma_2} \tee \sem{e_2} : \boxempty^{q_1} \sem{A}$.

    By VAR, have \textbf{(5)} $\cdot, x_1^\prime :^1 \,^{q_1} \sem{A} \rightarrow \boxempty^{q_2} \sem{B} \tee x_1^\prime : \,^{q_1} \sem{A} \rightarrow \boxempty^{q_2} \sem{B}$ and \textbf{(6)} $\cdot, x_2^\prime :^1 \sem{A} \tee x_2^\prime : \sem{A}$.

    By APP on \textbf{(5)} and \textbf{(6)}, have \textbf{(7)} $\cdot, x_1^\prime :^1 \,^{q_1} \sem{A} \rightarrow \boxempty^{q_2} \sem{B}, x_2^\prime :^{q_1} \sem{A} \tee x_1^\prime \, x_2^\prime : \boxempty^{q_2} \sem{B}$.

    By \ref{lem:mode-qualifiers-box-cut} on \textbf{(2)} and \textbf{(7)}, have \textbf{(8)} $\sem{\Gamma_2}, x_1^\prime :^1 \,^{q_1} \sem{A} \rightarrow \boxempty^{q_2} \sem{B} \tee \llet x_2 \leftarrow \sem{e_2} \lin \llet \bbox_{q_1} x_2^\prime = x_2 \lin x_1^\prime \, x_2^\prime : \boxempty^{q_2} \sem{B}$.
    
    By weakening on \textbf{(8)}, have \textbf{(9)} $\sem{\Gamma_2}, x_1^\prime :^q \,^{q_1} \sem{A} \rightarrow \boxempty^{q_2} \sem{B} \tee \llet x_2 \leftarrow \sem{e_2} \lin \llet \bbox_{q_1} x_2^\prime = x_2 \lin x_1^\prime \, x_2^\prime : \boxempty^{q_2} \sem{B}$.
    
    Thus, by \ref{lem:mode-qualifiers-box-cut} on \textbf{(3)} and \textbf{(9)}, $\sem{\Gamma_1} + \sem{\Gamma_2} \tee \llet x_1 \leftarrow \sem{e_1} \lin \llet \bbox_q x_1^\prime = x_1 \lin \llet x_2 \leftarrow \sem{e_2} \lin \llet \bbox_{q_1} x_2^\prime = x_2 \lin x_1^\prime \, x_2^\prime : \boxempty^{q_2} \sem{B}$, which, by the translation, is equivalent to $\sem{\Gamma_1 + \Gamma_2} \tee \sem{x_1^\prime \, x_2^\prime} : \boxempty^{q_2} \sem{B}$, as required.
    \\
    \item[Case $\boxempty$I.]
    By assumption, have \textbf{(1)} $\Gamma \tee e : A \at q \cdot q^\prime$.
    
    By the inductive hypothesis on \textbf{(1)}, have \textbf{(2)} $\sem{\Gamma} \tee \sem{e} : \boxempty^{q \cdot q^\prime} \sem{A}$.
    
    By VAR, have \textbf{(3)} $\cdot, x^\prime :^1 \sem{A} \tee x^\prime : \sem{A}$.

    By $\boxempty$I twice on \textbf{(3)}, have \textbf{(4)} $\cdot, x^\prime :^{q \cdot q^\prime} \sem{A} \tee \bbox_q \bbox_{q^\prime} x^\prime : \boxempty^q \boxempty^{q^\prime} \sem{A}$.
    By $\boxempty$I twice on \textbf{(3)}, have \textbf{(4)} $\cdot, x^\prime :^{q \cdot q^\prime} \sem{A} \tee \bbox_q \bbox_{q^\prime} x^\prime : \boxempty^q \boxempty^{q^\prime} \sem{A}$.
    
    Thus, by \ref{lem:mode-qualifiers-box-cut} on \textbf{(1)} and \textbf{(4)}, $\sem{\Gamma} \tee \llet x \leftarrow \sem{e} \lin \llet \bbox_{q \cdot q^\prime} x^\prime = x \lin \bbox_q (\bbox_{q^\prime} x^\prime)$, which, by the translation, is equivalent to $\sem{\Gamma} \tee \sem{\bbox_{q^\prime} e} : \boxempty^{q_2} \sem{B}$, as required.
    \\
    \item[Case $\boxempty$E.]
    By assumption, have \textbf{(1)} $\Gamma \tee e : \boxempty^{q^\prime} \at q$.
    
    By inductive hypothesis on \textbf{(1)}, have \textbf{(2)} $\sem{\Gamma} \tee \sem{e} : \boxempty^q \boxempty^{q^\prime} \sem{A}$.

    By VAR, have \textbf{(3)} $\cdot, x^{\prime\prime} :^1 \sem{A} \tee x^{\prime\prime} : \sem{A}$, \textbf{(4)} $\cdot, x^\prime :^1 \boxempty^{q^\prime} \sem{A} \tee x^\prime : \boxempty^{q^\prime} \sem{A}$ , and \textbf{(5)} $\cdot, x :^1 \boxempty^q \boxempty^{q^\prime} \sem{A} \tee x : \boxempty^q \boxempty^{q^\prime} \sem{A}$.

    By $\boxempty$I on \textbf{(3)}, have \textbf{(6)} $\cdot, x^{\prime\prime} :^{q \cdot q^\prime} \sem{A} \tee \bbox_{q \cdot q^\prime} x^{\prime\prime} : \boxempty^{q \cdot q^\prime} \sem{A}$.
    
    By $\boxempty$E on \textbf{(4)} and \textbf{(6)}, have \textbf{(7)} $\cdot, x^\prime :^q \boxempty^{q^\prime} \sem{A} \tee \llet \bbox_{q^\prime} x^{\prime\prime} = x^\prime \lin \bbox_{q \cdot q^\prime} x^{\prime\prime} : \boxempty^{q \cdot q^\prime} \sem{A}$.
    
    By $\boxempty$E on \textbf{(5)} and \textbf{(7)}, have \textbf{(8)} $\cdot, x :^1 \boxempty^q \boxempty^{q^\prime} \sem{A} \tee \llet \bbox_q x^\prime = x \lin \llet \bbox_{q^\prime} x^{\prime\prime} = x^\prime \lin \bbox_{q \cdot q^\prime} x^{\prime\prime} : \boxempty^{q \cdot q^\prime} \sem{A}$.
    
    Thus, by BIND on \textbf{(2)} and \textbf{(8)}, have \textbf{(8)} $\sem{\Gamma} \tee \llet x \leftarrow \sem{e} \lin \llet \bbox_q x^\prime = x \lin \llet \bbox_{q^\prime} x^{\prime\prime} = x^\prime \lin \bbox_{q \cdot q^\prime} x^{\prime\prime} : \boxempty^{q \cdot q^\prime} A$, which, by the translation, is equivalent to $\sem{\Gamma} \tee \sem{\mathrm{unbox}_{q^\prime} e} : \boxempty_{q \cdot q^\prime} \sem{A}$, as required.
    \\
    \item[Case LET.]
    By assumption, have \textbf{(1)} $\Gamma_1 \tee e_1 : A \at q_1$ and \textbf{(2)} $\Gamma_2, x : A \at q_1 \tee e_2 : B \at q_2$.
    
    By the inductive hypothesis on each of \textbf{(1)} and \textbf{(2)}, have \textbf{(3)} $\sem{\Gamma_1} \tee \sem{e_1} : \boxempty^{q_1} \sem{A}$ and \textbf{(4)} $\sem{\Gamma_2}, x :^{q_1} \sem{A} \tee \sem{e_2} : \boxempty^{q_2} \sem{B}$.

    Thus, by \ref{lem:mode-qualifiers-box-cut} on \textbf{(3)} and \textbf{(4)}, $\sem{\Gamma_1} + \sem{\Gamma_2} \tee \llet x \leftarrow \sem{e_1} \lin \llet \bbox_{q_1} x^\prime = x \lin \sem{e_2} : \boxempty^{q_2} \sem{B}$, which, by the translation, is equivalent to $\sem{\Gamma_1 + \Gamma_2} \tee \sem{\llet x = e_1 \lin e_2} : \boxempty^{q_2} \sem{B}$, as required.
    \\
    \item[Case UNIT.]
    By UNIT, have \textbf{(1)} $\sem{\Gamma} \tee () : \unit$.      
    
    Thus, by $\boxempty$I on \textbf{(1)}, $\sem{\Gamma} \tee \mathrm{box}_q() : \boxempty^q \unit$, as required. 
    \\
    \item[Case PAIR.] 
    By assumption, have \textbf{(1)} $\Gamma_1 \tee e_1 : A \at q$ and \textbf{(2)} $\Gamma_2 \tee e_2 : B \at q$.

    By the inductive hypothesis on each of \textbf{(1)} and \textbf{(2)}, have \textbf{(3)} $\sem{\Gamma_1} \tee \sem{e_1} : \boxempty^q \sem{A}$ and \textbf{(4)} $\sem{\Gamma_2} \tee \sem{e_2} : \boxempty^q \sem{B}$.

    By VAR, have \textbf{(5)} $\cdot, x_1^\prime :^1 \sem{A} \tee x_1^\prime : \sem{A}$ and \textbf{(6)} $\cdot, x_2^\prime :^1 \sem{B} \tee x_2^\prime : \sem{B}$.

    By PAIR on \textbf{(5)} and \textbf{(6)}, have \textbf{(7)} $\cdot, x_1^\prime :^1 \sem{A}, x_2^\prime :^1 \sem{B} \tee (x_1^\prime, x_2^\prime) : \sem{A} \times \sem{B}$.

    By $\boxempty$I on \textbf{(7)}, have \textbf{(8)} $\cdot, x_1^\prime :^q \sem{A}, x_2^\prime :^q \sem{B} \tee \bbox_q (x_1^\prime, x_2^\prime) : \boxempty_q (\sem{A} \times \sem{B})$.

    By \ref{lem:mode-qualifiers-box-cut} on \textbf{(2)} and \textbf{(8)}, have \textbf{(9)} $\sem{\Gamma_2}, x_1^\prime :^q \sem{A} \tee \llet x_2 \leftarrow \sem{e_2} \lin \llet \bbox_q x_2^\prime = x_2 \lin \bbox_q (x_1^\prime, x_2^\prime) : \boxempty^q (\sem{A} \times \sem{B})$.
    
    Thus, by \ref{lem:mode-qualifiers-box-cut} on \textbf{(3)} and \textbf{(9)}, $\sem{\Gamma_1} + \sem{\Gamma_2} \tee \llet x_1 \leftarrow \sem{e_1} \lin \llet \bbox_q x_1^\prime = x_1 \lin \llet x_2 \leftarrow \sem{e_2} \lin \llet \bbox_q x_2^\prime = x_2 \lin \bbox_q (x_1^\prime, x_2^\prime) : \boxempty^q (\sem{A} \times \sem{B})$, which, by the translation, is equivalent to $\sem{\Gamma_1 + \Gamma_2} \tee \sem{(x_1^\prime, x_2^\prime)} : \boxempty^q \sem{A \times B}$, as required.
    \\
    \item[Case SPLIT.]
    By assumption, have \textbf{(1)} $\Gamma_1 \tee e_1 : A \times B \at q_1$ and \textbf{(2)} $\Gamma_2, x : A \at q_1, y : B \at q_1 \tee e_2 : C \at q_2$.
    
    By inductive hypothesis on each of \textbf{(1)} and \textbf{(2)}, have \textbf{(3)} $\sem{\Gamma_1} \tee \sem{e_1} : \boxempty^{q_1} (\sem{A} \times \sem{B})$ and \textbf{(4)} $\sem{\Gamma_2}, x :^{q_1} \sem{A}, y :^{q_1} \sem{B} \tee \sem{e_2} : \boxempty^{q_2} \sem{C}$.

    By VAR, have \textbf{(5)} $\cdot, z^\prime :^1 \sem{A} \times \sem{B} \tee z^\prime : \sem{A} \times \sem{B}$.

    By SPLIT on \textbf{(4)} and \textbf{(5)}, have \textbf{(6)} $\sem{\Gamma_2}, z^\prime :^{q_1} \sem{A} \times \sem{B} \tee \llet (x, y) = z^\prime \lin \sem{e_2} : \boxempty^{q_2} \sem{C}$.
    
    Thus, by \ref{lem:mode-qualifiers-box-cut} on \textbf{(3)} and \textbf{(6)}, $\sem{\Gamma_1} + \sem{\Gamma_2} \tee \llet z \leftarrow \sem{e_1} \lin \llet \bbox_{q_1} z^\prime = z \lin \llet (x, y) = z^\prime \lin \sem{e_2} : \boxempty^{q_2} \sem{C}$, which, by translation, is equivalent to $\sem{\Gamma_1 + \Gamma_2} \tee \sem{\llet_{q_1} (x, y) = e_1 \lin e_2} : \boxempty^{q_2} \sem{C}$, as required.
    \\
    \item[Case INL.]
    By assumption, have \textbf{(1)} $\Gamma \tee e : A \at q$.
    
    By the inductive hypothesis on \textbf{(1)}, have \textbf{(2)} $\sem{\Gamma} \tee \sem{e} : \boxempty^q \sem{A}$.
    
    By VAR, have \textbf{(3)} $\cdot, x^\prime :^1 \sem{A} \tee x^\prime : \sem{A}$.
    
    By INL on \textbf{(3)}, have \textbf{(4)} $\cdot, x^\prime :^1 \sem{A} \tee \inl x^\prime : \sem{A} + \sem{B}$.

    By $\boxempty$I on \textbf{(4)}, have \textbf{(5)} $\cdot, x^\prime :^q \sem{A} \tee \bbox_q(\inl x^\prime) : \boxempty^q(\sem{A} + \sem{B})$.

    Thus, by \ref{lem:mode-qualifiers-box-cut} on \textbf{(2)} and \textbf{(5)}, $\sem{\Gamma} \tee \llet x \leftarrow \sem{e} \lin \llet \bbox_q x^\prime = x \lin \bbox_q (\inl x^\prime) : \boxempty^q (\sem{A} + \sem{B})$, which, by the translation, is equivalent to $\sem{\Gamma} \tee \sem{\inl e} : \boxempty^q \sem{A + B}$, as required.
    \\
    \item[Case INR.] 
    By assumption, have \textbf{(1)} $\Gamma \tee e : B \at q$.
    
    By the inductive hypothesis on \textbf{(1)}, have \textbf{(2)} $\sem{\Gamma} \tee \sem{e} : \boxempty^q \sem{B}$.
    
    By VAR, have \textbf{(3)} $\cdot, x^\prime :^1 \sem{B} \tee x^\prime : \sem{B}$.
    
    By INR on \textbf{(3)}, have \textbf{(4)} $\cdot, x^\prime :^1 \sem{B} \tee \inr x^\prime : \sem{A} + \sem{B}$
    
    By $\boxempty$I on \textbf{(4)}, have \textbf{(5)} $\cdot, x^\prime :^q \sem{B} \tee \bbox_q(\inr x^\prime) : \boxempty^q(\sem{A} + \sem{B})$.

    Thus, by \ref{lem:mode-qualifiers-box-cut} on \textbf{(2)} and \textbf{(5)}, $\sem{\Gamma} \tee \llet x \leftarrow \sem{e} \lin \llet \bbox_q x^\prime = x \lin \bbox_q (\inr x^\prime) : \boxempty^q (\sem{A} + \sem{B})$, which, by the translation, is equivalent to $\sem{\Gamma} \tee \sem{\inr e} : \boxempty^q \sem{A + B}$, as required.
    \\
    \item[Case CASE.]
    By assumption, have \textbf{(1)} $\Gamma_1 \tee e_1 : A + B \at q_1$, \textbf{(2)} $\Gamma_2, x : A \at q_1 \tee e_2 : C \at q_2$, and \textbf{(3)} $\Gamma_2, y : B \at q_1 \tee e_3 : C \at q_2$.
    
    By inductive hypothesis on each of \textbf{(1)}, \textbf{(2)} and \textbf{(3)}, have \textbf{(4)} $\sem{\Gamma_1} \tee \sem{e_1} : \boxempty^{q_1} (\sem{A} + \sem{B})$, \textbf{(5)} $\sem{\Gamma_2}, x :^{q_1} \sem{A} \tee \sem{e_2} : \boxempty^{q_2} \sem{C}$ and \textbf{(6)} $\sem{\Gamma_2}, y :^{q_1} \sem{B} \tee \sem{e_3} : \boxempty^{q_2} \sem{C}$.

    By VAR, have \textbf{(7)} $\cdot, z^\prime :^1 \sem{A} + \sem{B} \tee z^\prime : \sem{A} + \sem{B}$.

    By CASE on \textbf{(5)}, \textbf{(6)} and \textbf{(7)}, have \textbf{(8)} $\sem{\Gamma_2}, z^\prime :^{q_1} \sem{A} + \sem{B} \tee \mathrm{case} \, z^\prime \{x.\sem{e_2}; y.\sem{e_3}\} : \boxempty^{q_2} \sem{C}$.
    
    Thus, by \ref{lem:mode-qualifiers-box-cut} on \textbf{(4)} and \textbf{(8)}, $\sem{\Gamma_1} + \sem{\Gamma_2} \tee \llet z \leftarrow \sem{e_1} \lin \llet \bbox_{q_1} z^\prime = z \lin \mathrm{case} \, z^\prime \{x.\sem{e_2}; y.\sem{e_3}\} : \boxempty^{q_2} \sem{C}$, which, by translation, is equivalent to $\sem{\Gamma_1 + \Gamma_2} \tee \sem{\mathrm{case}_{q_1} e_1 \{x.e_2; y.e_3\}} : \boxempty^{q_2} \sem{C}$, as required.
    \\
    \item[Case ABSURD.]
    By assumption, have \textbf{(1)} $\Gamma \tee e : \zero \at q^\prime$.

    By the inductive hypothesis on \textbf{(1)}, have \textbf{(2)} $\sem{\Gamma} \tee \sem{e} : \boxempty^q \zero$.

    By VAR, have \textbf{(3)} $0 \cdot \cdot, x^\prime :^1 \zero \tee x : \zero$.

    By ABSURD on \textbf{(3)}, have \textbf{(4)} $\top \cdot \cdot, x^\prime :^\top \zero \tee \absurd \, x : A$.
    
    By $\boxempty$I on \textbf{(4)}, have \textbf{(5)} $\top \cdot \cdot, x^\prime :^\top \zero \tee \bbox_q(\absurd \, x) : \boxempty^q A$.

    By weakening on \textbf{(5)}, have \textbf{(6)} $\cdot, x^\prime :^q \zero \tee \bbox_q(\absurd \, x) : \boxempty^q A$.

    Thus, by \ref{lem:mode-qualifiers-box-cut} on \textbf{(2)} and \textbf{(6)}, $\sem{\Gamma} \tee \llet x \leftarrow \sem{e} \lin \llet \bbox_q x^\prime = x \lin \bbox_q(\absurd x^\prime)$, which, by the translation, is equivalent to $\sem{\Gamma} \tee \sem{\absurd e} : \boxempty^q \sem{A}$, as required.
    \\
  \end{description}
\end{proof}

\subsection{Equational Laws for Graded Modal Type Theory}

Consider $e[v_1/x, v_2/y]$ as a shorthand for $e[v_2/y][v_1/x]$.

\paragraph{Beta reduction laws}

\begin{align*}
  (\lambda x.\, e)\, v &\red e[v/x] : A \\
  \llet x \leftarrow v \lin e &\red e[v/x] : A \\
  \llet_{q_1} \bbox_{q_2} x = \bbox_{q_2}\, v \lin e &\red e[v/x] : A \\
  \llet_q (x, y) = (v_1, v_2) \lin e &\red e[v_1/x, v_2/y] : A
    && (x \notin \mathrm{fv}(v_2)) \\
  \case_q\, (\inl\, v)\, \{x.e_1;\; y.e_2\} &\red e_1[v/x] : A \\
  \case_q\, (\inr\, v)\, \{x.e_1;\; y.e_2\} &\red e_2[v/y] : A
\end{align*}

\begin{lemma}[Forward Substitution for Graded Type Theory]
\label{lem:graded-modal-forward-substitution}
  If $\Gamma_1 \tee v : A$ and $\Gamma_2, x :^q A, \Gamma_2^\prime \tee e : B$, then $q \cdot \Gamma_1 + \Gamma_2, \Gamma_2^\prime \tee e[v/x] : B$.
\end{lemma}

\begin{proof}
  Proceed by the induction over the structure of the derivation of $\Gamma_2, x :^q A, \Gamma_2^\prime \tee e : B$.
  \begin{description}
    \item[Case VAR.]
    Since $\Gamma_2 = \Gamma_2^\prime = \cdot$, and $q = 1$, by assumption, $\Gamma_1 \tee v : A$, as required.
    \\
    \item[Case SUB.]
    By assumption, have \textbf{(1)} $\Gamma_1 \tee v : A$ and \textbf{(2)} $\Gamma_2, x :^q A, y :^{q^{\prime\prime}} B, \Gamma_2^\prime \tee v^\prime : C$.

    By SUB on \textbf{(2)}, have \textbf{(3)} $q^\prime \le q^{\prime\prime}$ and \textbf{(4)} $\Gamma_2, x :^q A, y :^{q^\prime} B, \Gamma_2^\prime \tee v^\prime : C$.

    By the inductive hypothesis on \textbf{(1)} and \textbf{(4)}, have \textbf{(5)} $q \cdot \Gamma_1 + \Gamma_2, y :^{q^\prime} B, \Gamma_2^\prime \tee v^\prime[v/x] : C$.
    
    Thus, by SUB on \textbf{(3)} and \textbf{(5)}, $q \cdot \Gamma_1 + \Gamma_2, y :^{q^{\prime\prime}} B, \Gamma_2^\prime \tee v^\prime[v/x] : C$, as required.
    \\
    \item[Case LAM.] 
    By assumption, have \textbf{(1)} $\Gamma_1 \tee v : A$ and \textbf{(2)} $\Gamma_2, x :^q A, \Gamma_2^\prime \tee \lambda y.e : \,^{q^\prime} B_1 \rightarrow B_2$.

    By LAM on \textbf{(2)}, have \textbf{(3)} $\Gamma_2, x :^q A, \Gamma_2^\prime, y :^{q^\prime} B_1 \tee e : B_2$.

    By the inductive hypothesis on \textbf{(1)} and \textbf{(3)}, have \textbf{(4)} $q \cdot \Gamma_1 + \Gamma_2, \Gamma_2^\prime, y :^{q^\prime} B_1 \tee e[v/x] : B_2$.

    Thus, by LAM on \textbf{(4)}, $q \cdot \Gamma_1 + \Gamma_2, \Gamma_2^\prime \tee \lambda y.e[v/x] : \,^{q^\prime} B_1 \rightarrow B_2$, as required.
    \\
    \item[Case UNIT.]
    By assumption, have \textbf{(1)} $\Gamma_1 \tee v : A$ and \textbf{(2)} $\Gamma_2, x :^q A, \Gamma_2^\prime \tee () : \unit$.

    Thus, by UNIT, $q \cdot \Gamma_1 + \Gamma_2, \Gamma_2^\prime \tee () : \unit$, as required.
    \\
    \item[Case PAIR.] 
    By assumption, have \textbf{(1)} $\Gamma_1 \tee v : A$ and \textbf{(2)} $\Gamma_2, x :^q A, \Gamma_2^\prime \tee (v_1, v_2) : B_1 \times B_2$.
    
    By PAIR on \textbf{(2)}, have \textbf{(3)} $\Gamma_{2_1}, x :^{q_1} A, \Gamma_{2_1}^\prime \tee v_1 : B_1$ and \textbf{(4)} $\Gamma_{2_2}, x : ^{q_2} A, \Gamma_{2_2}^\prime \tee v_2 : B_2$, where $\Gamma_{2_1} + \Gamma_{2_2} = \Gamma_2$, $\Gamma_{2_1}^\prime + \Gamma_{2_2}^\prime = \Gamma_2^\prime$, and $q_1 + q_2 = q$.

    By the inductive hypothesis on each of \textbf{(3)} and \textbf{(4)} with \textbf{(1)}, have \textbf{(5)} $q_1 \cdot \Gamma_1 + \Gamma_{2_1}, \Gamma_{2_1}^\prime \tee v_1[v/x] : B_1$ and \textbf{(6)} $q_2 \cdot \Gamma_1 + \Gamma_{2_2}, \Gamma_{2_2}^\prime \tee v_2[v/x] : B_2$.

    By PAIR on \textbf{(5)} and \textbf{(6)}, have \textbf{(7)} $q_1 \cdot \Gamma_1 + q_2 \cdot \Gamma_1 + \Gamma_2, \Gamma_2^\prime \tee (v_1[v/x], v_2[v/x]) : B_1 \times B_2$.

    Thus, by weakening using law \ref{law:mul-distr-add-right-1}, $q \cdot \Gamma_1 + \Gamma_2, \Gamma_2^\prime \tee (v_1[v/x], v_2[v/x]) : B_1 \times B_2$, as required.
    \\
    \item[Case INL.] 
    By assumption, have \textbf{(1)} $\Gamma_1 \tee v : A$ and \textbf{(2)} $\Gamma_2, x :^q A, \Gamma_2^\prime \tee \inl v^\prime : B_1 + B_2$.
    
    By INL on \textbf{(2)}, have \textbf{(3)} $\Gamma_2, x :^q A, \Gamma_2^\prime \tee v^\prime : B_1$.
    
    By the inductive hypothesis on \textbf{(1)} and \textbf{(3)}, have \textbf{(4)} $q \cdot \Gamma_1 + \Gamma_2, x :^q A, \Gamma_2^\prime \tee v^\prime[v/x] : B_1$.

    Thus, by INL on \textbf{(4)}, $q \cdot \Gamma_1 + \Gamma_2, x :^q A, \Gamma_2^\prime \tee \inl v^\prime[v/x] : B_1 + B_2$, as required.
    \\
    \item[Case INR.] 
    By assumption, have \textbf{(1)} $\Gamma_1 \tee v : A$ and \textbf{(2)} $\Gamma_2, x :^q A, \Gamma_2^\prime \tee \inr v^\prime : B_1 + B_2$.
    
    By INR on \textbf{(2)}, have \textbf{(3)} $\Gamma_2, x :^q A, \Gamma_2^\prime \tee v^\prime : B_2$.
    
    By the inductive hypothesis on \textbf{(1)} and \textbf{(3)}, have \textbf{(4)} $q \cdot \Gamma_1 + \Gamma_2, x :^q A, \Gamma_2^\prime \tee v^\prime[v/x] : B_2$.

    Thus, by INR on \textbf{(4)}, $q \cdot \Gamma_1 + \Gamma_2, x :^q A, \Gamma_2^\prime \tee \inr v^\prime[v/x] : B_1 + B_2$, as required.
    \\
    \item[Case $\boxempty$I.] 
    By assumption, have \textbf{(1)} $\Gamma_1 \tee v : A$ and \textbf{(2)} $q^\prime \cdot \Gamma_2, x :^{q^\prime \cdot q} A, q^\prime \cdot \Gamma_2^\prime \tee \bbox_{q^\prime} v^\prime : \boxempty^{q^\prime} B$.

    By $\boxempty$I on \textbf{(2)}, have \textbf{(3)} $\Gamma_2, x :^q A, \Gamma_2^\prime \tee v^\prime : B$.

    By the inductive hypothesis on \textbf{(1)} and \textbf{(3)}, have \textbf{(4)} $q \cdot \Gamma_1 + \Gamma_2, \Gamma_2^\prime \tee v^\prime[v/x] : B$.
    
    By $\boxempty$I on \textbf{(4)}, have \textbf{(5)} $q^\prime \cdot (q \cdot \Gamma_1 + \Gamma_2), x :^{q^\prime \cdot q} A, q^\prime \cdot \Gamma_2^\prime \tee v^\prime[v/x] : B$.
    
    By weakening using law \ref{law:mul-distr-add-left-2} on \textbf{(5)}, have \textbf{(6)} $q^\prime \cdot (q \cdot \Gamma_1) + q^\prime \cdot \Gamma_2, x :^{q^\prime \cdot q} A, q^\prime \cdot \Gamma_2^\prime \tee v^\prime[v/x] : B$.
    
    Thus, by weakening using law \ref{law:mul-assoc-1} on \textbf{(6)}, $(q^\prime \cdot q) \cdot \Gamma_1 + q^\prime \cdot \Gamma_2, x :^{q^\prime \cdot q} A, q^\prime \cdot \Gamma_2^\prime \tee v^\prime[v/x] : B$, as required.
    \\
    \item[Case APP.] 
    By assumption, have \textbf{(1)} $\Gamma_1 \tee v : A$ and \textbf{(2)} $\Gamma_2, x :^q A, \Gamma_2^\prime \tee e \, v^\prime : B_2$.
    
    By APP on \textbf{(2)}, have \textbf{(3)} $\Gamma_{2_1}, x :^{q_1} A, \Gamma_{2_1}^\prime \tee e : \,^q B_1 \rightarrow B_2$ and \textbf{(4)} $\Gamma_{2_2}, x :^{q_2}, \Gamma_{2_2}^\prime \tee v^\prime : B_2$, where $\Gamma_{2_1} + q^\prime \cdot \Gamma_{2_2} = \Gamma_2$, $\Gamma_{2_1}^\prime + q^\prime \cdot \Gamma_{2_2}^\prime = \Gamma_2^\prime$, and $q_1 + q^\prime \cdot q_2 = q$.

    By the inductive hypothesis on each of \textbf{(3)} and \textbf{(4)} with \textbf{(1)}, have \textbf{(5)} $q_1 \cdot \Gamma_1 + \Gamma_{2_1}, \Gamma_{2_1}^\prime \tee e[v/x] : \,^q B_1 \rightarrow B_2$ and \textbf{(6)} $q_2 \cdot \Gamma_1 + \Gamma_{2_2}, \Gamma_{2_2}^\prime \tee v^\prime[v/x] : B_2$.

    By APP on \textbf{(5)} and \textbf{(6)}, have \textbf{(7)} $q_1 \cdot \Gamma_1 + q^\prime \cdot (q_2 \cdot \Gamma_1 + \Gamma_{2_2}) + \Gamma_{2_1}, \Gamma_2^\prime \tee e[v/x] \, v^\prime[v/x] : B_2$.
    
    By weakening on \textbf{(7)} using law \ref{law:mul-distr-add-left-2}, have \textbf{(8)} $q_1 \cdot \Gamma_1 + q^\prime \cdot (q_2 \cdot \Gamma_1) + \Gamma_2, \Gamma_2^\prime \tee e[v/x] \, v^\prime[v/x] : B_2$.
    
    By weakening using law \ref{law:mul-assoc-1} on \textbf{(8)}, have \textbf{(9)} $q_1 \cdot \Gamma_1 + (q^\prime \cdot q_2) \cdot \Gamma_1 + \Gamma_2, \Gamma_2^\prime \tee e[v/x] \, v^\prime[v/x] : B_2$.

    Thus, by weakening using law \ref{law:mul-distr-add-right-1} on \textbf{(9)}, $q \cdot \Gamma_1 + \Gamma_2, \Gamma_2^\prime \tee e[v/x] \, v^\prime[v/x] : B_2$, as required.
    \\
    \item[Case BIND.] 
    By assumption, have \textbf{(1)} $\Gamma_1 \tee v : A$ and \textbf{(2)} $\Gamma_2, x :^q A, \Gamma_2^\prime \tee \llet y \leftarrow e_1 \lin e_2 : B_2$.

    By BIND on \textbf{(2)}, have \textbf{(3)} $\Gamma_{2_1}, x :^{q_1}, \Gamma_{2_1}^\prime \tee e_1 : B_1$ and \textbf{(4)} $\Gamma_{2_2}, x :^{q_2} A, \Gamma_{2_2}^\prime, y :^1 B_1 \tee e_2 : B_2$, where $\Gamma_{2_1} + \Gamma_{2_2} = \Gamma_2$, $\Gamma_{2_1}^\prime + \Gamma_{2_2}^\prime = \Gamma_2^\prime$, and $q_1 + q_2 = q$.

    By the inductive hypothesis on each of \textbf{(3)} and \textbf{(4)} with \textbf{(1)}, have \textbf{(5)} $q_1 \cdot \Gamma_1 + \Gamma_{2_1}, \Gamma_{2_1}^\prime \tee e_1[v/x] : B_1$ and \textbf{(6)} $q_2 \cdot \Gamma_1 + \Gamma_{2_2}, \Gamma_{2_2}^\prime, y :^1 B_1 \tee e_2[v/x] : B_2$.

    By BIND on \textbf{(5)} and \textbf{(6)}, have \textbf{(7)} $q_1 \cdot \Gamma_1 + q_2 \cdot \Gamma_1 + \Gamma_2, \Gamma_2^\prime \tee \llet y \leftarrow e_1 \lin e_2 : B_2$.

    Thus, by weakening using law \ref{law:mul-distr-add-right-1}, $q \cdot \Gamma_1 + \Gamma_2, \Gamma_2^\prime \tee \llet y \leftarrow e_1[v/x] \lin e_2[v/x] : B_2$, as required.
    \\
    \item[Case $\boxempty$E.]
    By assumption, have \textbf{(1)} $\Gamma_1 \tee v : A$ and \textbf{(2)} $\Gamma_2, x :^q A, \Gamma_2^\prime \tee \llet_{q_1^\prime} \bbox_{q_2^\prime} y = v^\prime \lin e : B_2$.
    
    By $\boxempty$E on \textbf{(2)}, have \textbf{(3)} $\Gamma_{2_1}, x :^{q_1}, \Gamma_{2_1}^\prime \tee v^\prime : \boxempty^{q_2^\prime} B_1$ and \textbf{(4)} $\Gamma_{2_2}, x :^{q_2} A, \Gamma_{2_2}^\prime, y :^{q_1^\prime \cdot q_2^\prime} B_1 \tee e_2 : B_2$, where $q_1^\prime \cdot \Gamma_{2_1} + \Gamma_{2_2} = \Gamma_2$, $q_1^\prime \cdot \Gamma_{2_1}^\prime + \Gamma_{2_2}^\prime = \Gamma_2^\prime$, and $q_1^\prime \cdot q_1 + q_2 = q$.
    
    By the inductive hypothesis on each of \textbf{(3)} and \textbf{(4)} with \textbf{(1)}, have \textbf{(5)} $q_1 \cdot \Gamma_1 + \Gamma_{2_1}, \Gamma_{2_1}^\prime \tee v^\prime[v/x] : \boxempty^{q_2} B_1$ and \textbf{(6)} $q_2 \cdot \Gamma_1 + \Gamma_{2_2}, \Gamma_{2_2}^\prime, y :^{q_1^\prime \cdot q_2^\prime} B_1 \tee e[v/x] : B_2$.

    By $\boxempty$E on \textbf{(5)} and \textbf{(6)}, have \textbf{(7)} $q_1^\prime \cdot (q_1 \cdot \Gamma_1 + \Gamma_{2_1}) + q_2 \cdot \Gamma_1 + \Gamma_{2_2}, \Gamma_2^\prime \tee \llet_{q_1} \bbox_{q_2} y = v^\prime[v/x] \lin e[v/x] : B_2$.
    
    By weakening using law \ref{law:mul-distr-add-left-2} on \textbf{(7)}, have \textbf{(8)} $q_1^\prime \cdot (q_1 \cdot \Gamma_1) + q_2 \cdot \Gamma_1 + \Gamma_2, \Gamma_2^\prime \tee \llet_{q_1} \bbox_{q_2} y = v^\prime[v/x] \lin e[v/x] : B_2$.
    
    By weakening using law \ref{law:mul-assoc-1} on \textbf{(8)}, have \textbf{(9)} 
    $(q_1^\prime \cdot q_1) \cdot \Gamma_1 + q_2 \cdot \Gamma_1 + \Gamma_2, \Gamma_2^\prime \tee \llet_{q_1} \bbox_{q_2} y = v^\prime[v/x] \lin e[v/x] : B_2$.

    Thus, by weakening using law \ref{law:mul-distr-add-right-1} on \textbf{(9)}, $q \cdot \Gamma_1 + \Gamma_2, \Gamma_2^\prime \tee \llet_{q_1} \bbox_{q_2} y = v^\prime[v/x] \lin e[v/x] : B_2$, as required.
    \\
    \item[Case SPLIT.] 
    By assumption, have \textbf{(1)} $\Gamma_1 \tee v : A$ and \textbf{(2)} $\Gamma_2, x :^q A, \Gamma_2^\prime \tee \llet_{q^\prime} (y, z) = v^\prime \lin e : C$.
    
    By SPLIT on \textbf{(2)}, have \textbf{(3)} $\Gamma_{2_1}, x :^{q_1}, \Gamma_{2_1}^\prime \tee v^\prime : B_1 \times B_2$ and \textbf{(4)} $\Gamma_{2_2}, x :^{q_2} A, \Gamma_{2_2}^\prime, y :^{q^\prime} B_1, z :^{q^\prime} B_2 \tee e : C$, where $q^\prime \cdot \Gamma_{2_1} + \Gamma_{2_2} = \Gamma_2$, $q^\prime \cdot \Gamma_{2_1}^\prime + \Gamma_{2_2}^\prime = \Gamma_2^\prime$, and $q^\prime \cdot q_1 + q_2 = q$.
    
    By the inductive hypothesis on each of \textbf{(3)} and \textbf{(4)} with \textbf{(1)}, have \textbf{(5)} $q_1 \cdot \Gamma_1 + \Gamma_{2_1}, \Gamma_{2_1}^\prime \tee v^\prime[v/x] : B_1 \times B_2$ and \textbf{(6)} $q_2 \cdot \Gamma_1 + \Gamma_{2_2}, \Gamma_{2_2}^\prime, y :^{q^\prime} B_1, z :^{q^\prime} B_2 \tee e[v/x] : C$.

    By SPLIT on \textbf{(5)} and \textbf{(6)}, have \textbf{(7)} $q^\prime \cdot (q_1 \cdot \Gamma_1 + \Gamma_{2_1}) + q_2 \cdot \Gamma_1 + \Gamma_{2_2}, \Gamma_2^\prime \tee \llet_{q} (y, z) = v^\prime[v/x] \lin e[v/x] : C$.
    
    By weakening using law \ref{law:mul-distr-add-left-2} on \textbf{(7)}, have \textbf{(8)} $q^\prime \cdot (q_1 \cdot \Gamma_1) + q_2 \cdot \Gamma_1 + \Gamma_2, \Gamma_2^\prime \tee \llet_{q} (y, z) = v^\prime[v/x] \lin e[v/x] : C$.
    
    By weakening using law \ref{law:mul-assoc-1} on \textbf{(8)}, have \textbf{(9)}  $(q^\prime \cdot q_1) \cdot \Gamma_1 + q_2 \cdot \Gamma_1 + \Gamma_2, \Gamma_2^\prime \tee \llet_{q} (y, z) = v^\prime[v/x] \lin e[v/x] : C$

    Thus, by weakening using law \ref{law:mul-distr-add-right-1} on \textbf{(9)}, $q \cdot \Gamma_1 + \Gamma_2, \Gamma_2^\prime \tee \llet_{q} (y, z) = v^\prime[v/x] \lin e[v/x] : C$, as required.
    \\
    \item[Case CASE.] 
    By assumption, have \textbf{(1)} $\Gamma_1 \tee v : A$ and \textbf{(2)} $\Gamma_2, x :^q A, \Gamma_2^\prime \tee \case_{q^\prime} v^\prime\{y.e_1; z.e_2\} : C$.
    
    By CASE on \textbf{(2)}, have \textbf{(3)} $\Gamma_{2_1}, x :^{q_1}, \Gamma_{2_1}^\prime \tee v^\prime : B_1 + B_2$, \textbf{(4)} $\Gamma_{2_2}, x :^{q_2} A, \Gamma_{2_2}^\prime, y :^{q^\prime} B_1 \tee e_1 : C$, and \textbf{(5)} $\Gamma_{2_2}, x :^{q_2} A, \Gamma_{2_2}^\prime, z :^{q^\prime} B_2 \tee e_2 : C$,  where $q^\prime \cdot \Gamma_{2_1} + \Gamma_{2_2} = \Gamma_2$, $q^\prime \cdot \Gamma_{2_1}^\prime + \Gamma_{2_2}^\prime = \Gamma_2^\prime$, and $q^\prime \cdot q_1 + q_2 = q$.
    
    By the inductive hypothesis on each of \textbf{(3)}, \textbf{(4)} and \textbf{(5)} with \textbf{(1)}, have \textbf{(6)} $q_1 \cdot \Gamma_1 + \Gamma_{2_1}, \Gamma_{2_1}^\prime \tee v^\prime[v/x] : B_1 + B_2$, \textbf{(7)} $q_2 \cdot \Gamma_1 + \Gamma_{2_2}, \Gamma_{2_2}^\prime, y :^{q^\prime} B_1 \tee e_1[v/x] : C$ and \textbf{(8)} $q_2 \cdot \Gamma_1 + \Gamma_{2_2}, \Gamma_{2_2}^\prime, z :^{q^\prime} B_2 \tee e_2[v/x] : C$.

    By CASE on \textbf{(6)}, \textbf{(7)} and \textbf{(8)}, have \textbf{(9)} $q^\prime \cdot (q_1 \cdot \Gamma_1 + \Gamma_{2_1}) + q_2 \cdot \Gamma_1 + \Gamma_{2_2}, \Gamma_2^\prime \tee \case_{q^\prime} v^\prime[v/x] \{y.e_1[v/x]; z.e_2[v/x]\} : C$.
    
    By weakening using law \ref{law:mul-distr-add-left-2} on \textbf{(9)}, have \textbf{(10)} $q^\prime \cdot (q_1 \cdot \Gamma_1) + q_2 \cdot \Gamma_1 + \Gamma_2, \Gamma_2^\prime \tee \case_{q^\prime} v^\prime[v/x] \{y.e_1[v/x]; z.e_2[v/x]\} : C$.
    
    By weakening using law \ref{law:mul-assoc-1} on \textbf{(10)}, have \textbf{(11)}  $(q^\prime \cdot q_1) \cdot \Gamma_1 + q_2 \cdot \Gamma_1 + \Gamma_2, \Gamma_2^\prime \tee \case_{q^\prime} v^\prime[v/x] \{y.e_1[v/x]; z.e_2[v/x]\} : C$

\begin{lemma}[Type Preservation for Beta Reduction Laws]
\label{lem:graded-modal-type-preservation-beta-reduction-laws}
  If\/ $e_1 \red e_2 : A$ and $\Gamma \tee e_1 : B$, then $\Gamma \tee e_2 : B$.
\end{lemma}

\begin{proof}
  Proceed by induction over the structure of $e_1 \red e_2 : A$. 
  \begin{description}
    \item[\mbox{Case $(\lambda x.e) v \red e[v/x] : A$}.]
    By assumption, have \textbf{(1)} $\Gamma \tee (\lambda x.e) v : B$.

    By APP on \textbf{(1)}, have \textbf{(2)} $\Gamma_1 \tee \lambda x.e : \,^q A \rightarrow B$ and \textbf{(3)} $\Gamma_2 \tee v : A$, where $\Gamma_1 + q \cdot \Gamma_2 = \Gamma$.
    
    By LAM on \textbf{(2)}, have \textbf{(4)} $\Gamma_1, x :^q A \tee e : B$.

    Thus, by \ref{lem:graded-modal-forward-substitution} on \textbf{(3)} and \textbf{(4)}, $\Gamma \tee e[v/x] : B$, as required.
    \\
    \item[\mbox{Case $\llet x \leftarrow v \lin e \red e[v/x]$}.]
    By assumption, have \textbf{(1)} $\Gamma \tee \llet x \leftarrow v \lin e : B$.

    By BIND on \textbf{(1)}, have \textbf{(2)} $\Gamma_1 \tee v : A$ and \textbf{(3)} $\Gamma_2, x :^1 A \tee e : B$, where $\Gamma_1 + \Gamma_2 = \Gamma$.

    Thus, by \ref{lem:graded-modal-forward-substitution}, $\Gamma \tee e[v/x] : B$, as required.
    \\
    \item[\mbox{Case $\llet_{q_1} \bbox_{q_2} x = \bbox_{q_2} v \lin e \red e[v/x] : A$}.] 
    By assumption, have \textbf{(1)} $\Gamma \tee \llet_{q_1} \bbox_{q_2} x = \bbox_{q_2} v \lin e : B$.

    By $\boxempty$E on \textbf{(1)}, have \textbf{(2)} $q_2 \cdot \Gamma_1 \tee v : \boxempty^{q_2} A$ and \textbf{(3)} $\Gamma_2, x :^{q_1 \cdot q_2} A \tee e : B$, where $q_1 \cdot (q_2 \cdot \Gamma_1) + \Gamma_2 = \Gamma$.

    By $\boxempty$I on \textbf{(2)}, have \textbf{(4)} $\Gamma_1 \tee v : A$.

    By \ref{lem:graded-modal-forward-substitution} on \textbf{(3)} and \textbf{(4)}, have \textbf{(5)} $(q_1 \cdot q_2) \cdot \Gamma \tee e[v/x] : B$.
    
    Thus, by weakening using law \ref{law:mul-assoc-2}, $q_1 \cdot (q_2 \cdot \Gamma) \tee e[v/x] : B$, as required.
    \\
    \item[\mbox{Case $\llet_q (x, y) = (v_1, v_2) \red e[v_1/x, v_2/y] : A$}.]
    By assumption, have \textbf{(1)} $\Gamma \tee \llet_q (x, y) = (v_1, v_2) \lin e : C$.

    By SPLIT on \textbf{(1)}, have \textbf{(2)} $\Gamma_1 \tee (v_1, v_2) : A \times B$ and \textbf{(3)} $\Gamma_2, x :^q A, y :^q B \tee e : C$, where $q \cdot \Gamma_1 + \Gamma_2 = \Gamma$.

    By PAIR on \textbf{(2)}, have \textbf{(4)} $\Gamma_{1_1} \tee v_1 : A$ and \textbf{(5)} $\Gamma_{1_2} \tee v_2 : B$, where $\Gamma_{1_1} + \Gamma_{1_2} = \Gamma_1$.

    By \ref{lem:graded-modal-forward-substitution} on \textbf{(3)} and \textbf{(5)}, then \textbf{(4)}, have \textbf{(6)} $q \cdot \Gamma_{1_1} + q \cdot \Gamma_{1_2} + \Gamma_2 \tee e[v_2/y][v_1/x] : C$.

    Thus, by weakening using law \ref{law:mul-distr-add-left-1} on \textbf{(6)}, $q \cdot \Gamma_1 + \Gamma_2 \tee e[v_2/y][v_1/x] : C$, as required.
    \\
    \item[\mbox{Case $\case_q (\inl v) \{x.e_1; y.e_2\} \red e_1[v/x] : A$}.] 
    By assumption, have \textbf{(1)} $\Gamma \tee \case_q (\inl v) \{x.e_1; y.e_2\} : C$.

    By CASE on \textbf{(1)}, have \textbf{(2)} $\Gamma_1 \tee \inl v : A + B$, \textbf{(3)} $\Gamma_2, x :^q A \tee e_1 : C$, and \textbf{(4)} $\Gamma_2, y :^q B \tee e_2 : C$, where $q \cdot \Gamma_1 + \Gamma_2 = \Gamma$.

    By INL on \textbf{(2)}, have \textbf{(5)} $\Gamma_1 \tee v : A$.

    Thus, by \ref{lem:graded-modal-forward-substitution} on \textbf{(3)} and \textbf{(5)}, $\Gamma \tee e_1[v/x] : C$, as required.
    \\
    \item[\mbox{Case $\case_q (\inr v) \{x.e_1; y.e_2\} \red e_2[v/x] : A$}.] 
    By assumption, have \textbf{(1)} $\Gamma \tee \case_q (\inr v) \{x.e_1; y.e_2\} : C$.

    By CASE on \textbf{(1)}, have \textbf{(2)} $\Gamma_1 \tee \inr v : A + B$, \textbf{(3)} $\Gamma_2, x :^q A \tee e_1 : C$, and \textbf{(4)} $\Gamma_2, y :^q B \tee e_2 : C$, where $q \cdot \Gamma_1 + \Gamma_2 = \Gamma$.

    Thus, by \ref{lem:graded-modal-forward-substitution} on \textbf{(3)} and \textbf{(5)}, $\Gamma \tee e_2[v/x] : C$, as required.
  \end{description}
\end{proof}

\begin{lemma}[Backward Substitution for Graded Type Theory]
\label{lem:graded-modal-backward-substitution}

\end{lemma}

\begin{proof}
  Proceed by induction over the structure of $\Gamma \tee e[v/x]$.
  \begin{description}

    By \ref{lem:graded-modal-equal-contexts}, $\Gamma = \Gamma_1 = \cdot, 1 \cdot \Gamma_1, \cdot$, as required.
    
    By VAR, $\cdot, x :^1 A \tee x : A$, as required.
    \\
    \item[Case SUB.]
    By assumption, have \textbf{(1)} $\Gamma, y :^{q_2} B, \Gamma^\prime \tee e[v/x] : C$ and \textbf{(2)} $\Gamma_1 \tee v : A$.

    By SUB on \textbf{(1)}, have \textbf{(3)} $q_1 \le q_2$ and \textbf{(4)} $\Gamma, y :^{q_1} B, \Gamma^\prime \tee e[v/x] : C$.

\begin{theorem}[Type Preservation for Beta Expansion Laws]
\label{lem:graded-modal-type-preservation-beta-expansion-laws}
  Given an appropriate number of well-typed values $\Gamma_{1_i} \tee v_i : A_i$, if\/ $e_1 \equiv e_2 : A$ and $\Gamma \tee e_1 : B$, then $\Gamma \tee e_2 : B$, where $\Gamma$ and each $\Gamma_{1_i}$ have the same variables.
\end{theorem}

\begin{proof}
  Proceed by induction over the structure of $e_1 \red e_2 : A$. 
  
  Note that, since the length of $\Gamma$ and $\Gamma_{1_i}$ are the same, $\Gamma_{2_i}^\prime = \cdot.$
  \begin{description}
    \item[\mbox{Case $e[v/x] \red (\lambda x.e) v : A$}.]
    By assumption, have \textbf{(1)} $\Gamma_1 \tee v : A$ and \textbf{(2)} $\Gamma \tee e[v/x] : B$.

    By \ref{lem:graded-modal-backward-substitution} on \textbf{(1)} and \textbf{(2)}, have $q$ and $\Gamma_2$ such that \textbf{(4)} $q \cdot \Gamma_1 + \Gamma_2 = \Gamma$ and \textbf{(5)} $\Gamma_2, x :^q A \tee e : B$.

    By LAM on \textbf{(5)}, have \textbf{(6)} $\Gamma_2 \tee \lambda x.e : \,^q A \rightarrow B$.

    Thus, by APP on \textbf{(4)} and \textbf{(6)}, $q \cdot \Gamma_1 + \Gamma_2 \tee (\lambda x.e) v : B$, as required.
    \\
    \item[\mbox{Case $e[v/x] \red \llet x \leftarrow v \lin e : A$}.]
    By assumption, have \textbf{(1)} $\Gamma_1 \tee v : A$ and \textbf{(2)} $\Gamma \tee e[v/x] : B$.
    
    By \ref{lem:graded-modal-backward-substitution} on \textbf{(1)} and \textbf{(2)}, have $q$ and $\Gamma_2$ such that \textbf{(4)} $q \cdot \Gamma_1 + \Gamma_2 = \Gamma$ and \textbf{(5)} $\Gamma_2, x :^q A \tee e : B$.

    Since $x \in \mathrm{fv}(e) \at 1$ and considering \textbf{(5)}, have $q = 1$.

    Thus, by BIND on \textbf{(1)} and \textbf{(5)}, $\Gamma_1 + \Gamma_2 \tee \llet x \leftarrow v \lin e : B$, as required.
    \\
    \item[\mbox{Case $e[v/x] \red \llet_{q_1} \bbox_{q_2} x = \bbox_{q_2} v \lin e : A$}.]
    By assumption, have \textbf{(1)} $\Gamma_1 \tee v : A$ and \textbf{(2)} $\Gamma \tee e[v/x] : B$.

    By \ref{lem:graded-modal-backward-substitution} on \textbf{(1)} and \textbf{(2)}, have $q$ and $\Gamma_2$ such that \textbf{(4)} $q \cdot \Gamma_1 + \Gamma_2 = \Gamma$ and \textbf{(5)} $\Gamma_2, x :^q A \tee e : B$.

    Since $x \in \mathrm{fv}(e) \at 1$ and considering \textbf{(5)}, have $q = q_1 \cdot q_2$.

    By $\boxempty$I on \textbf{(1)}, have \textbf{(6)} $q_2 \cdot \Gamma_1 \tee \bbox_{q_2} v : \boxempty^{q_2} A$.

    By $\boxempty$E on \textbf{(5)} and \textbf{(6)}, have \textbf{(7)} $q_1 \cdot (q_2 \cdot \Gamma_1) + \Gamma_2 \tee \llet_{q_1} \bbox_{q_2} x = \bbox_{q_2} v \lin e : B$.

    Thus, by weakening using law \ref{law:mul-assoc-1}, $(q_1 \cdot q_2) \cdot \Gamma_1 + \Gamma_2 \tee \llet_{q_1} \bbox_{q_2} x = \bbox_{q_2} v \lin e : B$, as required.
    \\
    \item[\mbox{Case $e[v_1/x, v_2/y] \red \llet_q (x, y) = (v_1, v_2) \lin e : A$}.]
    By assumption, have \textbf{(1)} $\Gamma_{1_1} \tee v_1 : A_1$, \textbf{(2)} $\Gamma_{1_2} \tee v_2 : A_2$ and \textbf{(3)} $\Gamma \tee e[v_2/y][v_1/x] : B$.

    By \ref{lem:graded-modal-backward-substitution} on \textbf{(1)} and \textbf{(3)}, have $q_1$ and $\Gamma_{2_1}$ such that \textbf{(4)} $q_1 \cdot \Gamma_{1_1} + \Gamma_{2_1} = \Gamma$ and \textbf{(5)} $\Gamma_{2_1}, x :^{q_1} A \tee e[v_2/y] : B$.

    By \ref{lem:graded-modal-backward-substitution} on \textbf{(2)} and \textbf{(5)}, have $q_2$ and $\Gamma_{2_2}$ such that \textbf{(6)} $q_2 \cdot \Gamma_{1_2} + \Gamma_{2_2} = \Gamma_{2_1}, x :^{q_1}$ and \textbf{(7)} $\Gamma_{2_2}, y :^{q_2} A \tee e : B$.

    Since $x, y \in \mathrm{fv}(e) \at q$, have $q = q_1 = q_2$.

    Since $x \notin \mathrm{fv}(v_2)$, have $x \notin \Gamma_{1_2}$ and so $\Gamma_{2_2} = \Gamma_{2_2}^\prime, x :^q A_1$ and $q \cdot \Gamma_{1_1} + q \cdot \Gamma_{1_2} + \Gamma_{2_2}^\prime = \Gamma$.

    From this, have \textbf{(8)} $\Gamma_2^\prime, x :^q A_1, y :^q A_2 \tee e : B$.

    By PAIR on \textbf{(1)} and \textbf{(2)}, have \textbf{(9)} $\Gamma_{1_1} + \Gamma_{1_2} \tee (v_1, v_2) : A_1 \times A_2$.

    By SPLIT on \textbf{(8)} and \textbf{(9)}, have $q \cdot (\Gamma_{1_1} + \Gamma_{1_2}) + \Gamma_2^\prime \tee \llet_q (x, y) = (v_1, v_2) \lin e : B$.

    Thus, by weakening using law \ref{law:mul-distr-add-left-2}, $q \cdot \Gamma_{1_1} + q \cdot \Gamma_{1_2} + \Gamma_2^\prime \tee \llet_q (x, y) = (v_1, v_2) \lin e : B$, as required.
    \\
    \item[\mbox{Case $e[v/x] \red \case_q v \{x.e; \, y.\absurd \, y\} : A$}.]
    By assumption, have \textbf{(1)} $\Gamma_1 \tee v : A$ and \textbf{(2)} $\Gamma \tee e[v/x] : B$.
    
    By INL on \textbf{(1)}, have \textbf{(3)} $\Gamma_1 \tee \inl v : A + \zero$.

    By \ref{lem:graded-modal-backward-substitution} on \textbf{(1)} and \textbf{(2)}, have $q$ and $\Gamma_2$ such that $q \cdot \Gamma_1 + \Gamma_2 = \Gamma$ and \textbf{(4)} $\Gamma_2, x :^q A \tee e : B$.

    By VAR, have \textbf{(5)} $0 \cdot \Gamma_2, y :^1 \zero \tee y : \zero$.

    By ABSURD on \textbf{(5)}, have \textbf{(6)} $\top \cdot \Gamma_2, y :^\top \zero \tee \absurd \, y : B$.

    By weakening on \textbf{(6)} using $\top \le q$ for any $q$, have \textbf{(7)} $\Gamma_2, y :^q \zero \tee \absurd \, y : B$.

    Thus, by CASE on \textbf{(3)}, \textbf{(4)} and \textbf{(7)}, $q \cdot \Gamma_1 + \Gamma_2 \tee \case_q v \{ x.e; \, y.\absurd \, y \}$, as required.
    \\
    \item[\mbox{Case $e[v/y] \red \case_q v \{x.\absurd \, x; y.e\} : A$}.]
    By assumption, have \textbf{(1)} $\Gamma_1 \tee v : A$ and \textbf{(2)} $\Gamma \tee e[v/y] : B$.
    
    By INL on \textbf{(1)}, have \textbf{(3)} $\Gamma_1 \tee \inl v : A + \zero$.

    By \ref{lem:graded-modal-backward-substitution} on \textbf{(1)} and \textbf{(2)}, have $q$ and $\Gamma_2$ such that $q \cdot \Gamma_1 + \Gamma_2 = \Gamma$ and \textbf{(4)} $\Gamma_2, y :^q A \tee e : B$.

    By VAR, have \textbf{(5)} $0 \cdot \Gamma_2, x :^1 \zero \tee x : \zero$.

    By ABSURD on \textbf{(5)}, have \textbf{(6)} $\top \cdot \Gamma_2, x :^\top \zero \tee \absurd \, x : B$.

    By weakening on \textbf{(6)} using $\top \le q$ for any $q$, have \textbf{(7)} $\Gamma_2, x :^q \zero \tee \absurd \, x : B$.

    Thus, by CASE on \textbf{(3)}, \textbf{(4)} and \textbf{(7)}, $q \cdot \Gamma_1 + \Gamma_2 \tee \case_q v \{ x.\absurd \, x; \, y.e \}$, as required.
  \end{description}
\end{proof}

\paragraph{Eta laws}

\begin{align*}
  \lambda x.\, v\, x &\equiv v : \,^q A \rightarrow B
    && (x \notin \mathrm{fv}(v)) \\
  \llet x \leftarrow v \lin () &\equiv v : \unit \\
  \llet_1 (x, y) = v \lin (x, y) &\equiv v : A \times B \\
  \case_1\, v\, \{x.\, \inl\, x;\; y.\, \inr\, y\} &\equiv v : A + B \\
  \llet_{1} \bbox_{q} x = v \lin \bbox_{q}\, x &\equiv v : \boxempty^{q} A
\end{align*}

\begin{lemma}[Type Preservation for Eta Laws]
\label{lem:graded-modal-type-preservation-eta-laws}
  If\/ $e_1 \equiv e_2 : A$, then $\Gamma \tee e_1 : A$ if and only if $\Gamma \tee e_2 : A$.
\end{lemma}

\begin{proof}
  Proceed by induction over the structure of $e_1 \equiv e_2 : A$.
  \begin{description}
    \item[Case $\lambda x.v \, x \equiv v : \,^q A \rightarrow B$.]
    By assumption, have \textbf{(1)} $\Gamma \tee \lambda x.v \, x : \,^q A \rightarrow B$.
    
    By LAM on \textbf{(1)}, have \textbf{(2)} $\Gamma, x :^q A \tee v \, x : B$.
    
    Thus, by APP on \textbf{(2)}, $\Gamma \tee v : \,^q A \rightarrow B$, as required.
    \\
    
    Conversely, by assumption, have \textbf{(1)} $\Gamma \tee v : \,^q A \rightarrow B$.

    By VAR, have \textbf{(2)} $\cdot, x :^1 A \tee x : A$.

    By APP on \textbf{(1)} and \textbf{(2)}, have \textbf{(3)} $\Gamma, x :^q A \tee v \, x : B$.
    
    Thus, by LAM on \textbf{(3)}, $\Gamma \tee \lambda x.v \, x : \,^q A \rightarrow B$, as required.
    \\
    \item[Case $\llet x \leftarrow v \lin \, () \equiv v : \unit$.]
    By UNIT, $\Gamma \tee v : \unit$, where $v = ()$, as required.
    \\

    Conversely, by assumption, have \textbf{(1)} $\Gamma \tee v : \unit$.
    
    By VAR, have \textbf{(2)} $\cdot, x :^1 \unit \tee x : \unit$.

    Thus, by BIND on \textbf{(1)} and \textbf{(2)}, $\Gamma \tee \llet x \leftarrow v \lin \, () : \unit$, as required.
    \\
    \item[Case $\llet_1 (x, y) = v \lin \, (x, y) \equiv v : A \times B$.]
    By assumption, have \textbf{(1)} $\Gamma \tee \llet_1 (x, y) = v \lin \, (x, y) : A \times B$.
    
    Thus, by SPLIT on \textbf{(1)}, $\Gamma \tee v : A \times B$, as required.
    \\

    Conversely, by assumption, have \textbf{(1)} $\Gamma \tee v : A \times B$.

    By VAR, have \textbf{(2)} $\cdot, x :^1 A \tee x : A$ and \textbf{(3)} $\cdot, y :^1 B \tee y : B$.

    By PAIR on \textbf{(2)} and \textbf{(3)}, have \textbf{(4)} $\cdot, x :^1 A, y :^1 B \tee (x, y) : A \times B$.

    Thus, by SPLIT on \textbf{(1)} and \textbf{(4)}, $\Gamma, x :^1 A, y :^1 B \tee \llet_1 (x, y) = v \lin \, (x, y) : A \times B$, as required.
    \\
    \item[Case $\case_1 v \{x.\inl x; y.\inr y\} \equiv v : \,^q A + B$.]
    By assumption, have \textbf{(1)} $\Gamma \tee \case_1 v \{x.\inl x, y.\inr y\} : A + B$.
    
    Thus, by CASE on \textbf{(1)}, $\Gamma \tee v : A + B$, as required.
    \\

    Conversely, by assumption, have \textbf{(1)} $\Gamma \tee v : A + B$.
    
    By VAR, have \textbf{(2)} $\cdot, x :^1 A \tee x : A$ and \textbf{(3)} $\cdot, y :^1 B \tee y : B$.

    By INL on \textbf{(2)}, have \textbf{(4)} $\cdot, x :^1 A \tee \inl x : A + B$.
    
    By INR on \textbf{(3)}, have \textbf{(5)} $\cdot, y :^1 B \tee \inr y : A + B$.
    \\
    \item[Case $\llet_1 \bbox_q x = v \lin \bbox_q x \equiv v : \boxempty^q A$.]
    By assumption, have \textbf{(1)} $\Gamma \tee \llet_1 \bbox_q x = v \lin \bbox_q x : \boxempty^q A$.

    Thus, by $\boxempty$E, $\Gamma \tee v : \boxempty^q A$, as required.
    \\

    Conversely, by assumption, have \textbf{(1)} $\Gamma \tee v : \boxempty^q A$.

    By VAR, have \textbf{(2)} $\cdot, x :^1 A \tee x : A$.

    By $\boxempty$I on \textbf{(2)}, have \textbf{(3)} $\cdot, x :^q A \tee \bbox_q v : \boxempty^q A$.

    Thus, by $\boxempty$E on \textbf{(1)} and \textbf{(3)}, $\Gamma \tee \llet_1 \bbox_q x = v \lin \bbox_q x : \boxempty^q A$, as required.
  \end{description}
\end{proof}

\paragraph{Commuting conversions}

\begin{align*}
  E[\llet x \leftarrow e_1 \lin e_2]
    &\equiv \llet x \leftarrow e_1 \lin E[e_2] : A
    && (x \notin \mathrm{fv}(E)) \\
  E[\llet_{q_1} \bbox_{q_2} x = v \lin e]
    &\equiv \llet_{q_1} \bbox_{q_2} x = v \lin E[e] : A
    && (x \notin \mathrm{fv}(E)) \\
  E[\llet_q (x, y) = v \lin e]
    &\equiv \llet_q (x, y) = v \lin E[e] : A
    && (x, y \notin \mathrm{fv}(E)) \\
  E[\case_q\, v\, \{x.e_1;\; y.e_2\}]
    &\equiv \case_q\, v\, \{x.\, E[e_1];\; y.\, E[e_2]\} : A
    && (x, y \notin \mathrm{fv}(E))
\end{align*}

\begin{lemma}[Type Preservation for Commuting Conversions]
\label{lem:graded-modal-type-preservation-commuting-conversions}
  If\/ $e_1 \equiv e_2 : A$, then $\Gamma \tee e_1 : A$ if and only if $\Gamma \tee e_2 : A$.
\end{lemma}

\begin{theorem}[Type Preservation for Graded Modal Type Theory]
\label{lem:graded-modal-type-preservation}
  If\/ $e_1 \equiv e_2 : A$, then $\Gamma \tee e_1 : A$ if and only if $\Gamma \tee e_2 : A$.
\end{theorem}
\begin{proof}
  By Lemmas~\ref{lem:graded-modal-type-preservation-beta-reduction-laws}, \ref{lem:graded-modal-type-preservation-beta-expansion-laws}, \ref{lem:graded-modal-type-preservation-eta-laws} and \ref{lem:graded-modal-type-preservation-commuting-conversions}.
\end{proof}

\begin{lemma}[Forward Substitution for Mode Qualifiers]
\label{lem:mode-qualifiers-forward-substitution}
  If $\Gamma_1 \tee v : A \at \mu_1$ and $\Gamma_2, x : A \at \mu_1, \Gamma_2^\prime \tee e : B \at \mu_2$, then $\Gamma_1 + \Gamma_2, \Gamma_2^\prime \tee e[v/x] : B \at \mu_2$.
\end{lemma}

\begin{proof}
  Proceed by induction over the structure of the derivation of $\Gamma_2, x : A \at \mu_1, \Gamma_2^\prime \tee e : B$.

  Most cases will be omitted for brevity.
  \begin{description}
    \item[Case VAR.] 
    Since $\Gamma_2 = \Gamma_2^\prime = \cdot$, by assumption, $\Gamma_1 \tee v : A \at \mu_1$, as required.
    \\
    \item[Case LAM.] 
    By assumption, have \textbf{(1)} $\Gamma_1 \tee v : A \at \mu_1$ and \textbf{(2)} $\Gamma_2, x : A \at \mu_1, \Gamma_2^\prime \tee \lambda y.e : (B_1 \at \mu_2 \rightarrow B_2 \at \mu_3) \at \mu_4$.

    By LAM on \textbf{(2)}, have \textbf{(3)} $\Gamma_2, x : A \at \mu_1, \Gamma_2^\prime, \lock_{\mu_4}, y : B_1 \at \mu_2 \tee e : B_2 \at \mu_3$.

    By the inductive hypothesis on \textbf{(1)} and \textbf{(3)}, have \textbf{(4)} $\Gamma_1 + \Gamma_2, \Gamma_2^\prime, \lock_{\mu_4}, y : B_1 \at \mu_2 \tee e[v/x] : B_2 \at \mu_3$.

    Thus, by LAM on \textbf{(4)}, $\Gamma_1 + \Gamma_2, \Gamma_2^\prime \tee \lambda y.e[v/x] : (B_1 \at \mu_2 \rightarrow B_2 \at \mu_3) \at \mu_4$, as required.
    \\
    \item[Case APP.]
    By assumption, have \textbf{(1)} $\Gamma_1 \tee v : A \at \mu_1$ and \textbf{(2)} $\Gamma_2, x : A \at \mu_1, \Gamma_2^\prime \tee e_1 \, e_2 : B_2 \at \mu_2$.

    By APP on \textbf{(2)}, have \textbf{(3)} $\Gamma_{2_1}, x : A \at \mu_{1_1}, \Gamma_{2_1}^\prime \tee e_1 : (B_1 \at \mu_3 \rightarrow B_2 \at \mu_4) \at \mu_5$, \textbf{(4)} $\Gamma_{2_2}, x : A \at \mu_{1_2}, \Gamma_{2_2}^\prime \tee e_2 : B_1 \at \mu_3$ and \textbf{(5)} $\mu_4 \ge \mu_2$, where $\mu_{1_1} + \mu_{1_2} = \mu_1$, $\Gamma_{2_1} + \Gamma_{2_2} = \Gamma_2$ and $\Gamma_{2_1}^\prime + \Gamma_{2_2}^\prime = \Gamma_2^\prime$.

    By \ref{lem:mode-qualifiers-mode-split} on \textbf{(1)}, have \textbf{(6)} $\Gamma_{1_1} \tee v : A \at \mu_{1_1}$ and \textbf{(7)} $\Gamma_{1_2} \tee v : A \at \mu_{1_2}$, where $\Gamma_{1_1} + \Gamma_{1_2} = \Gamma_1$.

    By the inductive hypothesis on \textbf{(3)} with \textbf{(5)} and  \textbf{(4)} with \textbf{(6)}, have \textbf{(8)} $\Gamma_{1_1} + \Gamma_{2_1}, \Gamma_{2_1}^\prime \tee e_1[v/x] : (B_1 \at \mu_3 \rightarrow B_2 \at \mu_4) \at \mu_5$ and \textbf{(9)} $\Gamma_{1_2} + \Gamma_{2_2}, \Gamma_{2_2}^\prime \tee e_2[v/x] : B_1 \at \mu_3$.

    Thus, by APP on \textbf{(5)}, \textbf{(8)} and \textbf{(9)}, $\Gamma_1 + \Gamma_2, \Gamma_2^\prime \tee e_1[v/x] \, e_2[v/x] : B_2 \at \mu_2$, as required.
    \\
    \item[Case $\boxempty$I.]
    By assumption, have \textbf{(1)} $\Gamma_1 \tee v : A \at \mu_1$ and \textbf{(2)} $\Gamma_2, x : A \at \mu_1, \Gamma_2^\prime \tee \bbox_{\mu_3} e : \boxempty^{\mu_3} A \at \mu_2$.

    By $\boxempty$I on \textbf{(2)}, have \textbf{(3)} $\Gamma_2, x : A \at \mu_1, \Gamma_2^\prime \tee e : A \at \mu_2 \cdot \mu_3$.

    By the inductive hypothesis on \textbf{(1)} and \textbf{(3)}, have \textbf{(4)} $\Gamma_1 + \Gamma_2, \Gamma_2^\prime \tee e[v/x] : A \at \mu_2 \cdot \mu_3$.

    Thus, by $\boxempty$I on \textbf{(4)}, $\Gamma_1 + \Gamma_2, \Gamma_2^\prime \tee \bbox_{\mu_3} e[v/x] : \boxempty^{\mu_3} A \at \mu_2$, as required.
    \\
    \item[Case $\boxempty$E.]
    By assumption, have \textbf{(1)} $\Gamma_1 \tee v : A \at \mu_1$ and \textbf{(2)} $\Gamma_2, x : A \at \mu_1, \Gamma_2^\prime \tee \unbox_{\mu_3} e : A \at \mu_2 \cdot \mu_3$.

    By $\boxempty$E on \textbf{(2)}, have \textbf{(3)} $\Gamma_2, x : A \at \mu_1, \Gamma_2^\prime \tee e : \boxempty^{\mu_3} A \at \mu_2$.

    By the inductive hypothesis on \textbf{(1)} and \textbf{(3)}, have \textbf{(4)} $\Gamma_1 + \Gamma_2, \Gamma_2^\prime \tee e[v/x] : \boxempty^{\mu_3} A \at \mu_2$.
    
    Thus, by $\boxempty$E on \textbf{(4)}, $\Gamma_1 + \Gamma_2, \Gamma_2^\prime \tee e[v/x] : \at \mu_2 \cdot \mu_3$, as required.
    \\
    \item[Case LET.]
    By assumption, have \textbf{(1)} $\Gamma_1 \tee v : A \at \mu_1$ and \textbf{(2)} $\Gamma_2, x : A \at \mu_1, \Gamma_2^\prime \tee \llet y = e_1 in e_2 : B_2 \at \mu_2$.

    By LET on \textbf{(2)}, have \textbf{(3)} $\Gamma_{2_1}, x : A \at \mu_{1_1}, \Gamma_{2_1}^\prime \tee e_1 : B_1 \at \mu_3$ and \textbf{(4)} $\Gamma_{2_2}, x : A \at \mu_{1_2}, \Gamma_{2_2}^\prime, y : B_1 \at \mu_3 \tee e_2 : B_2 \at \mu_2$, where $\mu_{1_1} + \mu_{1_2} = \mu_1$, $\Gamma_{2_1} + \Gamma_{2_2} = \Gamma_2$ and $\Gamma_{2_1}^\prime + \Gamma_{2_2}^\prime = \Gamma_2^\prime$.

    By \ref{lem:mode-qualifiers-mode-split} on \textbf{(1)}, have \textbf{(5)} $\Gamma_{1_1} \tee v : A \at \mu_{1_1}$ and \textbf{(6)} $\Gamma_{1_2} \tee v : A \at \mu_{1_2}$, where $\Gamma_{1_1} + \Gamma_{1_2} = \Gamma_1$.

    By the inductive hypothesis on \textbf{(3)} with \textbf{(5)} and  \textbf{(4)} with \textbf{(6)}, have \textbf{(7)} $\Gamma_{1_1} + \Gamma_{2_1}, \Gamma_{2_1}^\prime \tee e_1[v/x] : B_1$ and \textbf{(8)} $\Gamma_{1_2} + \Gamma_{2_2}, \Gamma_{2_2}^\prime, y : B_1 \at \mu_3 \tee e_2[v/x] : B_2 \at \mu_2$.

    Thus, by LET on \textbf{(7)} and \textbf{(8)}, $\Gamma_1 + \Gamma_2, \Gamma_2^\prime \tee \llet y = e_1[v/x] \lin e_2[v/x] : B_2 \at \mu_2$, as required.
  \end{description}
\end{proof}

\begin{lemma}[Type Preservation for Beta Reduction Laws]
\label{lem:mode-qualifiers-type-preservation-beta-reduction-laws}
  If\/ $e_1 \equiv e_2 : A \at \mu$ and $\Gamma \tee e_1 : A \at \mu$, then $\Gamma \tee e_2 : A \at \mu$.
\end{lemma}

\begin{proof}
  Proceed by induction over the structure of $e_1 \red e_2 : A$. 
  \begin{description}
    \item[Case \mbox{$(\lambda x.e) v \red e[v/x] : A \at \mu_2$}.]
    By assumption, have \textbf{(1)} $\Gamma_1 + \Gamma_2 \tee (\lambda x.e) v : B \at \mu_2$.

    By APP on \textbf{(1)}, have \textbf{(2)} $\Gamma_1 \tee \lambda x.e : (A \at \mu_1 \rightarrow B \at \mu_3) \at \mu_4$, \textbf{(3)} $\Gamma_2 \tee v : A \at \mu_1$ and \textbf{(4)} $\mu_3 \ge \mu_2$

    By LAM on \textbf{(2)}, have \textbf{(5)} $\Gamma_1, \lock_{\mu_4}, x : A \at \mu_1 \tee e : B \at \mu_3$.

    By \ref{lem:mode-qualifiers-forward-substitution} on \textbf{(3)} and \textbf{(5)}, have \textbf{(6)} $(\Gamma_1, \lock_{\mu_4}) + \Gamma_2 \tee e[v/x] : B \at \mu_3$.

    Thus, by weakening, \textbf{(4)} and \ref{law:add-mono} on \textbf{(6)}, $\Gamma_1 + \Gamma_2 \tee e[v/x] : B \at \mu_2$, as required.
    \\
    \item[Case \mbox{$\llet x = v \lin e \red e[v/x] : A \at \mu_2$}.]
    By assumption, have \textbf{(1)} $\Gamma_1 + \Gamma_2 \tee \llet x = v \lin e : B \at \mu_2$.

    By LET on \textbf{(1)}, have \textbf{(2)} $\Gamma_1 \tee v : A \at \mu_1$ and \textbf{(3)} $\Gamma_2, x : A \at \mu_1 \tee e : B \at \mu_2$.

    Thus, by \ref{lem:mode-qualifiers-forward-substitution} on \textbf{(2)} and \textbf{(3)}, $\Gamma_1 + \Gamma_2 \tee e[v/x] : B \at \mu_2$, as required.
    \\
    \item[Case \mbox{$\unbox_\nu (\bbox_\nu v) \red v : A \at \nu \cdot \mu$}.] 
    \item[Case \mbox{$\llet_{\mu_1} (x, y) = (v_1, v_2) in e \red e[v_1/x, v_2/y]$}.] 
    \item[Case \mbox{$(\lambda x.e) v \red e[v/x]$}.] 
    \item[Case \mbox{$(\lambda x.e) v \red e[v/x]$}.] 
  \end{description}
\end{proof}

\paragraph{Eta laws}

\begin{align*}
  \llet x = e \lin \lambda y.\, x\, y &\equiv e : (A \at \mu_1 \rightarrow B \at \mu_2) \at \mu \\
  \llet x = e \lin () &\equiv e : \unit \at \mu \\
  \llet_1 (x, y) = e \lin (x, y) &\equiv e : A \times B \at \mu \\
  \case_1\, e\, \{x.\, \inl\, x;\; y.\, \inr\, y\} &\equiv e : A + B \at \mu \\
  \bbox_\nu\, (\unbox_\nu\, e) &\equiv e : \boxempty^\nu A \at \mu
\end{align*}

\begin{lemma}[Type Preservation for Eta Laws]
\label{lem:mode-qualifiers-type-preservation-eta-laws}
  If\/ $e_1 \equiv e_2 : A \at \mu$, then $\Gamma \tee e_1 : A \at \mu$ if and only if $\Gamma \tee e_2 : A \at \mu$.
\end{lemma}

\paragraph{Evaluation contexts}
Recall:
\begin{align*}
  E &::= [\cdot] \mid \bbox_\nu\, E \mid (E, e) \mid (v, E) \mid \inl\, E \mid \inr\, E \\
    &\mid E\, e_2 \mid v_1\, E \mid \unbox_\nu\, E \\
    &\mid \llet x = E \lin e \mid \llet_\mu (x, y) = E \lin e \\
    &\mid \case_\mu\, E\, \{x.e_1;\; y.e_2\}
\end{align*}

\paragraph{Commuting conversions}

\begin{align*}
  E[\llet x = e_1 \lin e_2]
    &\equiv \llet x = e_1 \lin E[e_2] : A \at \mu_2
    && (x \notin \mathrm{fv}(E)) \\
  E[\llet_{\mu_1} (x, y) = e_1 \lin e_2]
    &\equiv \llet_{\mu_1} (x, y) = e_1 \lin E[e_2] : A \at \mu_2
    && (x, y \notin \mathrm{fv}(E)) \\
  E[\case_{\mu_1}\, e\, \{x.e_1;\; y.e_2\}]
    &\equiv \case_{\mu_1}\, e\, \{x.\, E[e_1];\; y.\, E[e_2]\} : A \at \mu_2
    && (x, y \notin \mathrm{fv}(E))
\end{align*}

\begin{lemma}[Type Preservation for Commuting Conversions]
\label{lem:mode-qualifiers-type-preservation-commuting-conversions}
  If\/ $e_1 \equiv e_2 : A \at \mu$, then $\Gamma \tee e_1 : A \at \mu$ if and only if $\Gamma \tee e_2 : A \at \mu$.
\end{lemma}

\begin{theorem}[Type Preservation for Mode Qualifiers]
  If\/ $e_1 \equiv e_2 : A \at \mu$, then $\Gamma \tee e_1 : A \at \mu$ if and only if $\Gamma \tee e_2 : A \at \mu$.
\end{theorem}
\begin{proof}
  By Lemmas~\ref{lem:mode-qualifiers-type-preservation-beta-laws}, \ref{lem:mode-qualifiers-type-preservation-eta-laws} and \ref{lem:mode-qualifiers-type-preservation-commuting-conversions}.
\end{proof}
